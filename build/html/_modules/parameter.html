from superclass import Base, Broadcast
from abc import ABC, abstractmethod
import util
import numpy as np


class Parameter(ABC, Base, Broadcast):
    
    _value = None
    
    # Properties
    
    @property
    def value(self):
        """
        array or None: Value of the parameter. The parameter is not initialized
        if its value is set to None.
        """
        
        return self._value
    
    # Setter
    
    @value.setter
    def value(self, value):
        if value is None:
            self._value = None
        elif isinstance(value, Parameter):
            self._value = value.value
        else:
            self._value = np.asarray(value)
        
    # Abstract methods
    
    @abstractmethod
    def control(self, value, name):
        """
        Abstract method to control wether the parameter is valid or not.
        If the parameter is not valid, an error should be raised.
        """
        
        raise NotImplementedError()
        
    # Methods
    
    def broadcast(self, *args):
        """
        Mimics the broadcast of the parameter's value with input arguments.
        
        Parameters
        ----------
        *args : array-like or object with shape attribute
            The arguments to be broadcast with the value of the parameter.
            It can be an array-like or any object having a 'shape' attribute.

        Returns
        -------
        np.broadcast
            Broadcast result.
        """
        
        return util.broadcast(self.value, *args)
    
    def reset(self):
        """
        Reset the value of the parameter. The value of the parameter will be set
        to None meaning that it is not initialized.
        
        Returns
        -------
        Instance
        """
        
        self.value = None
        return self
        
    def update(self, value):
        """
        Update the value of the parameter.
        
        Parameters
        ----------
        value : array-like or Parameter or None
            The new value of the parameter.
            
        Returns
        -------
        Instance
        """
        
        self.value = value
        return self

    def __init__(self, dtype = None, min = None, max = None, strict = False):
        
        # Control the lower bound
        
        if min is not None:
            min = np.asarray(min)[()]
            if min.size != 1:
                raise ValueError(f"'min' must be a scalar: not an array with shape {min.shape}")
                
        self._min = min
                
        # Control the upper bound
                
        if max is not None:
            max = np.asarray(max)[()]
            if max.size != 1:
                raise ValueError(f"'max' must be a scalar: not an array with shape {max.shape}")
                
        self._max = max
                
        # Control if the upper bound is greater than the lower bound
        
        strict = bool(strict)
        
        if min is not None and max is not None:
            if strict and not np.all(min < max):
                raise ValueError("'max' must be greater than 'min'")
            elif not np.all(min <= max):
                raise ValueError("'max' must be greater than or equal to 'min'")
            
        self._strict = strict
        
        # Control if dtype can be interpreted as a valid data type
        
        if dtype is not None:
            dtype = np.dtype(dtype)
            
        self._dtype = dtype
            
    # Properties
    
    @property
    def bounds(self):
        """
        tuple: Tuple with the lower bound and upper bound.
        """
        
        return (self.min, self.max)
    
    @property
    def dtype(self):
        """
        np.dtype or None: Data type of the parameter's value. If None, all data

        """

[docs]    def control(self, value = None, name = "value"):
        """
        Control wether the parameter is valid or not.
        
        The parameter is considered as valid if all the following conditions are
        satisfied:
            
            * The parameter is initialized (its value is not set to None).
            * The parameter has a valid dtype.
            * The parameter is greater than the lower bound.
            * The parameter is less than the upper bound.

        Parameters
        ----------
        value : array_like or None, optional
            The value to control. If None, the instance value will be used.
            The default is None.
        name : str, optional
            Name of the parameter. If an error is raised, this name will be used.
            The default is "value".

        Raises
        ------
        ValueError
            If the parameter is not initialized (set to None).
            If the parameter has not a valid dtype.
            If the parameter is not between the lower bound and upper bound.
        """
        
        # Value to control
        
        if value is None:
            value = self.value
        else:
            value = np.asarray(value)
        
        # Control if the value is initialized
        
        if value is None:
            raise ValueError(f"'{name}' is not initialized (set to None)")
            
        # Control if the value has a valid dtype
        
        if self.dtype is not None:
            if not np.issubdtype(value.dtype, self.dtype):
                raise ValueError(f"'{name}' must have the following dtype: {type(self.dtype).__name__}")
            
        # Control if the value is greater than the lower bound
        
        if self.min is not None:
            if self.strict and not np.all(value > self.min):
                raise ValueError(f"'{name}' must be greater than {self.min}")
            elif not np.all(value >= self.min):
                raise ValueError(f"'{name}' must be greater than or equal to {self.min}")
                
        # Control if the value is less than the upper bound
        
        if self.max is not None:
            if self.strict and not np.all(value < self.max):
                raise ValueError(f"'{name}' must be less than {self.max}")
            elif not np.all(value <= self.max):
                raise ValueError(f"'{name}' must be less than or equal to {self.max}")             


   
class Parameters(Base, Broadcast, dict):
    
    # Constructor    
    
    def __init__(self, **kwargs):
        
        self._control_kwargs(**kwargs)
        dict.__init__(self, **kwargs)
        
    # Properties    
    
    @property
    def names(self):
        """
        list of str: Name of each parameter.
        """
        
        return list(self.keys())
    
    @property
    def shapes(self):
        """
        dict of tuple: Shape of each parameter.
        """
        
        return {key: parameter.shape for key, parameter in self.items()}
    
    @property
    def values(self):
        """
        list: Value of each parameter.
        """
        
        return [parameter.value for parameter in self]
        
    # Methods
    
    def _control_keys(self, *keys):
        """
        Control wether the provided keys (parameter names) are defined in the instance.
        If a parameter doesn't exist, an error will be raised.
        """
        
        for key in keys:
            if key not in self:
                raise KeyError(f"parameter '{key}' is not found")
                
    def _control_kwargs(self, **kwargs):
        """
        Control wether the provided arguments are 'Parameter' instances.
        If an argument is not a 'Parameter' instance, an error will be raised.
        """
        
        for key, parameter in kwargs.items():
            if not isinstance(parameter, Parameter):
                raise TypeError(f"'{key}' must be 'Parameter' instance")
                
    def add(self, **kwargs):
        """
        Add new parameters.

        Parameters
        ----------
        **kwargs : Parameter
            New parameters to add. The keyword is the name of the parameter to
            add and the argument must be a 'Parameter' instance. If the name
            already exists, the parameter will be replaced by the provided argument.
        
        Raises
        ------
        TypeError
            If an argument is not a 'Parameter' instance.
            
        Returns
        -------
        Instance

        """
        
        self._control_kwargs(**kwargs)
        
        for key, parameter in kwargs.items():
            self[key] = parameter
            
        return self
        
    def broadcast(self, *args):
        """
        Mimics the broadcast of parameters with input arguments.

        Parameters
        ----------
        *args : array_like or object with shape attribute
            Inputs to be broadcast with the parameters. It can be an array-like
            or any object having a 'shape' attribute. If no argument is provided,
            only the parameters are broadcast together.

        Returns
        -------
        np.broadcast
            Broadcast result.

        """
        
        return util.broadcast(*self, *args)
    
    def control(self):
        """
        Control the value of the parameters. If a parameter is not valid, an
        error will be raised.

        Raises
        ------
        ValueError
            If a parameter is not valid.

        """
        
        # Control each individual parameter
        
        for key, parameter in self.items():
            parameter.control(name = key)
            
        # Control min-max parameters such as min < max
            
        if "min" in self and "max" in self:
            if np.any(self["min"].value >= self["max"].value):
                raise ValueError("'max' must be greater than 'min'")
    
    def pop(self, keys = -1):
        """
        Remove in-place the specified parameters. The removed parameters are
        returned by the method.

        Parameters
        ----------
        keys : str or int or slice or list of str, optional
            Name of parameters that must be returned and removed. It can be a
            single parameter if a string or integer is provided, or multiple
            parameters if a slice or list of strings is provided. In the last case,
            a 'Parameters' instance will be returned. The default is the last parameter.

        Raises
        ------
        TypeError
            If the keys argument is not a string, integer, slice or list.
        IndexError
            If the index is out of range when an integer is provided.
        KeyError
            If the name of a parameter is not found when a string or list of strings is provided.

        Returns
        -------
        Parameter or Parameters
            Return the removed parameters. If a string or integer is provided
            (single parameter to be removed), a 'Parameter' instance is returned
            while a 'Parameters' instance is returned if a slice or list
            of strings is provided (multiple parameters to be removed).

        """
        
        pop = lambda key: dict.pop(self, key)
            
        if isinstance(keys, str):
            return pop(keys)
        elif isinstance(keys, int):
            return pop(self.names[keys])
        elif isinstance(keys, slice):
            return Parameters(**{key: pop(key) for key in self.names[keys]})
        elif isinstance(keys, (list, tuple)):
            self._control_keys(*keys)
            return Parameters(**{key: pop(key) for key in keys})
        else:
            raise TypeError(f"cannot pop items from a '{type(keys).__name__}' instance: should be a string, integer, slice or list of strings")
            
    def reset(self):
        """
        Reset the value of the parameters. The value of each parameter will be
        set to None meaning that they are not initialized.  
        
        Returns
        -------
        Instance
        
        """
        
        for parameter in self:
            parameter.reset()
            
        return self
    
    def update(self, *args, **kwargs):
        """
        Update the value of the parameters.

        Parameters
        ----------
        *args : array_like
            New parameter values. The first argument is the new value
            of the first parameter, the second argument is the new value of the
            second parameter, ... The number of arguments provided must be equal
            to the number of parameters of the instance.
        **kwargs : array_like
            New parameter values given as key-value pairs. The keyword is the
            name of the parameter to be updated. If the name doesn't exist,
            an error will be raised.
            
        Raises
        ------
        ValueError
            If the number of arguments provided by args is not the same as the
            number of parameters of the instance.
            If the name of a parameter doesn't exist.

        Returns
        -------
        Instance
        
        """
        
        # Control the keys from kwargs argument
        
        self._control_keys(*kwargs.keys())
        
        # Update parameters given by args
        
        if args:
            if len(args) == len(self):
                for key, value in zip(self.keys(), args):
                    self[key].update(value)
            else:
                raise ValueError(f"{len(args)} values are given through args instead of {len(self)} associated with the following parameters: {', '.join(self.keys())}")
        
        # Update parameters given by kwargs
            
        for key, value in kwargs.items():
            self[key].update(value)
            
        # Return output
            
        return self
    
    # Special methods
    
    def __getitem__(self, keys):
        """
        Return a single parameter ('Parameter' instance) if a string or integer
        is provided, and return a 'Parameters' instance if a slice or list
        of strings is provided.
        """
        
        get = lambda key: dict.__getitem__(self, key)

        if isinstance(keys, str):
            return get(keys)
        elif isinstance(keys, int):
            return get(self.names[keys])
        elif isinstance(keys, slice):
            return Parameters(**{key: get(key) for key in self.names[keys]})
        elif isinstance(keys, (list, tuple)):
            return Parameters(**{key: get(key) for key in keys})
        else:
            raise TypeError(f"cannot get items from a '{type(keys).__name__}' instance: should be a string, integer, slice or list of strings")
            
    def __iter__(self):
        """
        Iterate over the parameters.     
        """
        
        return iter(dict.values(self))
    
    def __repr__(self): 
        return self.__module__ + "." + type(self).__name__ + "\n" + util.dict_to_string(self, bullet = "\t")
