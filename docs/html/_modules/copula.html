    def log_likelihood(self, u, weights = None): # REVIEW
        """
        Compute the log-likelihood.

        Parameters
        ----------
        u : 2D array-like
            Observed values from which the log-likelihood will be calculated.
            Because each observation is *n*-dimensional, `u` must be a matrix with
            *n* columns meaning that each row represents an observed *n*-dimensional
            sample.
        weights : 1D array-like or None, optional
            Weight associated with each *n*-dimensional sample. If None, the weight
            is assumed to be 1 for each sample. The default is None.
            
        Returns
        -------
        scalar:
            Computed log-likelihood. The returned scalar has a float dtype.
        """
                
        if weights is None:
            return np.sum(np.log(self.pdf(u)))
        else:
            return np.sum(weights*np.log(self.pdf(u)))
        
    # Decorators    
    
    class Decorators:
        """
        Container class containing decorators to use for the implemented
        methods of `Copula` instances.
        """
        
        def cdf(func):
            """
            Decorator for the implemented `cdf` method of a copula.
            
            This decorator implements the following:
                
                * Control wether the values for which the cdf should be evaluated
                  match the dimensionality of the copula along the last axis and
                  are between 0 and 1.
                * Ensure that the returned output is an array with a float dtype
                  and reduce it to a scalar when possible.
    
            Parameters
            ----------
            func : callable
                Method evaluating the cdf. The first argument of this method
                must be the values for which the cdf should be evaluated.
                Other additional arguments are allowed.
    
            Returns
            -------
            callable
                Decorator.
            """
            
            @wraps(func)
            def wrapper(self, u, *args, **kwargs):
                
                # Control if u matches the dimensionality of the copula and contains values between 0 and 1

                u = np.asarray(u)
                
                if not util.is_along_axis(u, -1, self.dim):
                    raise ValueError(f"'u' doesn't match the dimensionality of the copula: must have a size {self.dim} along its last index")
                elif np.any(u < 0) or np.any(u > 1):
                    raise ValueError("'u' must contain values between 0 and 1")
                    
                # Evaluate the cdf
                
                cdf = func(self, u, *args, **kwargs)
                
                # Return output
                
                return np.array(cdf, dtype = float)[()]
                
            wrapper._undecorated = func            
            return wrapper
        
        def pdf(func):
            """
            Decorator for the implemented `pdf` method of a copula.
            
            This decorator implements the following:
                
                * Control wether the values for which the pdf should be evaluated
                  match the dimensionality of the copula along the last axis and
                  are between 0 and 1.
                * Ensure that the returned output is an array with a float dtype

                u = np.asarray(u)

                pdf = func(self, u, *args, **kwargs)
                
                # Return output
                
                return np.array(pdf, dtype = float)[()]
                
            wrapper._undecorated = func            
            return wrapper
        
        def random(func):
            """
            Decorator for the implemented `random` method of a copula.
            
            This decorator implements the following:
                
                * Convert the size of the samples to a tuple of integers.
                * Ensure that the returned output is an array with a float dtype.
    
            Parameters
            ----------
            func : callable
                Method generating the random samples. This method must have two
                arguments: the size of the samples and the seed to use.
                No additional arguments are allowed.
    
            Returns
            -------
            callable
                Decorator.
            """

            @wraps(func)
            def wrapper(self, size = None, seed = -1):

                # Convert size to a tuple of integers
    
                if size:
                    try:
                        size = (int(size),)
                    except TypeError:
                        size = tuple([int(s) for s in size])
                else:
                    size = ()
                    
                # Generate random samples of the copula
                
                random = func(self, size, seed)
                
                # Return output
                
                return np.array(random, dtype = float)
            
            wrapper._undecorated = func            
            return wrapper

    def cdf(self, x, keepdims = False):
        r"""
        Evaluate the cumulative distribution function (cdf).
                
        Note
        ----
        The cdf of the Clayton copula is given by
        
        .. math::                
            C(u_1, ..., u_n) = \max\left(\sum_{i=1}^n u_i^{-\theta} + 1 - n, 0 \right)^{-1/\theta}
            
        where :math:`(u_1, ..., u_n) \in [0,1]^n`. In the bivariate case (:math:`n=2`),
        we have therefore
        
        .. math::
            C(u_1, u_2) = \max\left( u_1^{-\theta} + u_2^{-\theta} - 1, 0 \right)^{-1/\theta}.

        Parameters
        ----------
        u : array-like
            Values (between 0 and 1) for which the cdf should be evaluated.
            The size along the last axis must be equal to the dimensionality *n*
            of the copula.
        keepdims : bool, optional
            If True, the last axis which is reduced is left in the result as dimension
            with size one (otherwise, it will be removed). With this option, the
            result will broadcast correctly against the input array.
            The default is False.
        
        Returns
        -------
        scalar or array
            Evaluated cdf. Array with the same shape as `u` except that the last
            axis has a size 1 if `keepdims` is True or is removed if it is False.
            
        Raises
        ------
        ValueError        
            If the size along the last axis of `u` doesn't match the dimensionality
            of the copula.
            If `u` contains values not between 0 and 1.
        """
        
        theta, d = self.theta, self.dim  
        return np.maximum((np.sum(x**-theta, axis = -1, keepdims = keepdims) + 1 - d), 0.)**(-1/theta)

[docs]    @Copula.Decorators.random
    def random(self, size = None, seed = -1):
        """
        Draw *n*-dimensional random samples of joint uniform distributions.
        
        Note
        ----
        The following methodology is used to generate a *n*-dimensional random
        sample of a comonotonic copula:
            
            1. Draw a single realization :math:`u` from an uniform distribution.
            2. Return :math:`(u_1, ..., u_n) = (u, ..., u)`.

        Parameters
        ----------
        size : int or tuple of int or None, optional
            Number of samples to be drawn. Because each sample is *n*-dimensional,
            the output shape will have an additional axis with size *n* in last
            position. The default returns a single *n*-dimensional sample.
        seed : int or array-like of int or SeedSequence or Generator or None, optional
            Seed to use for the random number generator. The default uses the
            random number generator associated with this instance.

        Returns
        -------
        array
            Drawn *n*-dimensional random samples containing values between 0 and 1.
            Array with the same shape as `size` except that an additional axis
            with size *n* has been added in last position.
        """
        
        rng = self.get_rng(seed)
        u = rng.random(size + (1,))
        return np.broadcast_to(u, size + (self.dim,))

    def cdf(self, u, keepdims = False):
        """
        Evaluate the cumulative distribution function (cdf).
        
        Note
        ----
        The cdf of the Gaussian copula is given by
        
        .. math::
            C(u_1, ..., u_n) = \Phi_R(\Phi^{-1}(u_1), ..., \Phi^{-1}(u_n))
            
        where :math:`(u_1, ..., u_n) \in [0,1]^n` and :math:`\Phi_R` is the joint
        cumulative distribution of a multivariate normal distribution with mean
        vector zero and covariance matrix equal to the correlation matrix :math:`R`.

        Parameters
        ----------
        u : array-like
            Values (between 0 and 1) for which the cdf should be evaluated.
            The size along the last axis must be equal to the dimensionality *n*
            of the copula.
        keepdims : bool, optional
            If True, the last axis which is reduced is left in the result as dimension
            with size one (otherwise, it will be removed). With this option, the
            result will broadcast correctly against the input array.
            The default is False.
        
        Returns
        -------
        scalar or array
            Evaluated cdf. Array with the same shape as `u` except that the last
            axis has a size 1 if `keepdims` is True or is removed if it is False.
            
        Raises
        ------
        ValueError        
            If the size along the last axis of `u` doesn't match the dimensionality
            of the copula.
            If `u` contains values not between 0 and 1.
        """

        cdf = multivariate_normal.cdf(norm.ppf(u), mean = np.full(self.dim, 0), cov = self.correlation_matrix) # allow_singular option ?
        shape = util.reduce_shape(u.shape, axis = -1, keepdims = keepdims)
        return cdf.reshape(shape)

    

[docs]    @Copula.Decorators.pdf
    def pdf(self, u, keepdims = False):
        """
        Evaluate the probability density function (pdf).
        
        Note
        ----
        The pdf of the Gaussian copula is given
        
        .. math::
            c(u_1, ..., u_n) = \dfrac{\phi_R(\Phi^{-1}(u_1), ..., \Phi^{-1}(u_n))}{\prod_{i=1}^n \phi(\Phi^{-1}(u_i))}
            
        where :math:`(u_1, ..., u_n) \in [0,1]^n` and :math:`\phi_R` is the joint
        density of a multivariate normal distribution with mean vector zero and
        covariance matrix equal to the correlation matrix :math:`R`.

        Parameters
        ----------
        u : array-like
            Values (between 0 and 1) for which the pdf should be evaluated.
            The size along the last axis must be equal to the dimensionality *n*
            of the copula.
        keepdims : bool, optional
            If True, the last axis which is reduced is left in the result as dimension
            with size one (otherwise, it will be removed). With this option, the
            result will broadcast correctly against the input array.
            The default is False.
        
        Returns
        -------
        scalar or array
            Evaluated pdf. Array with the same shape as `u` except that the last
            axis has a size 1 if `keepdims` is True or is removed if it is False.
            
        Raises
        ------
        ValueError        
            If the size along the last axis of `u` doesn't match the dimensionality
            of the copula.
            If `u` contains values not between 0 and 1.
        """
        
        x = norm.ppf(u)
        log_pdf = multivariate_normal.logpdf(x, mean = np.full(self.dim, 0), cov = self.correlation_matrix) - np.sum(norm.logpdf(x), axis = -1).squeeze() # work in log-scale to avoid over/under-flow
        shape = util.reduce_shape(u.shape, axis = -1, keepdims = keepdims)
        return np.exp(log_pdf).reshape(shape)

    def random(self, size = None, seed = -1):
        """
        Draw *n*-dimensional random samples of joint uniform distributions.
        
        Note
        ----
        The following methodology is used to generate a *n*-dimensional random
        sample of a Gaussian copula:
        
            1. Draw a random sample :math:`(x_1, ..., x_n)` from a multivariate
               normal distribution with mean vector zero and covariance matrix
               equal to the correlation matrix :math:`R`.
            2. Return :math:`(u_1, ..., u_n) = (\Phi(x_1), ..., \Phi(x_n))` where
               :math:`\Phi` denotes the cumulative distribution function of the
               standard normal distribution.

        Parameters
        ----------
        size : int or tuple of int or None, optional
            Number of samples to be drawn. Because each sample is *n*-dimensional,
            the output shape will have an additional axis with size *n* in last
            position. The default returns a single *n*-dimensional sample.
        seed : int or array-like of int or SeedSequence or Generator or None, optional
            Seed to use for the random number generator. The default uses the
            random number generator associated with this instance.

        Returns
        -------
        array
            Drawn *n*-dimensional random samples containing values between 0 and 1.
            Array with the same shape as `size` except that an additional axis
            with size *n* has been added in last position.
        """
        
        rng = self.get_rng(seed)
        return norm.cdf(rng.multivariate_normal(mean = np.full(self.dim, 0), cov = self.correlation_matrix, size = size))

            random number generator associated with this instance.

        Returns
        -------
        array
            Drawn *n*-dimensional random samples containing values between 0 and 1.
            Array with the same shape as `size` except that an additional axis
            with size *n* has been added in last position.
        """
        
        rng = self.get_rng(seed)
        return rng.random(size + (self.dim,))
