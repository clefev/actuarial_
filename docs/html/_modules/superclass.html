class Active:
    
    # Class methods
    
    active = True
        
    # Methods
    
    def activate(self):
        """
        Activate the instance. The instance will become active and its 'active'
        attribute will be set to True. All methods requiring an active instance
        to run will execute.

        Returns
        -------
        Instance.

        """
        
        self.active = True
        return self
        
    def deactivate(self):
        """
        Deactivate the instance. The instance will become inactive and its 'active'
        attribute will be set to False. All methods requiring an active instance
        to run will not execute.

        Returns
        -------
        Instance.

        """
        
        self.active = False
        return self
        
    # Decorators
    
    class Decorators:
        """
        Container class containing decorators to be used for the implemented methods
        of Active instances.
        """        
        
        def active(func):
            """
            Decorator for the implemented methods requiring an active instance
            to run. If the instance is active, the decorated methods will execute
            normaly. If the instance is inactive, the decorated methods will
            not execute.
    
            Parameters
            ----------
            func : callable
                Method requiring an active instance to run.
    
            Returns
            -------
            callable
                Decorator.
    
            """
            
            @wraps(func)
            def wrapper(self, *args, **kwargs):
                
                if self.active:
                    return func(self, *args, **kwargs)
                else:
                    return self
                
            return wrapper

[docs]    def _dict_to_print(self):
        """
        Return a dictionary with attributes and properties to print when calling
        the `print` function on the object.
        
        Note
        ----
        The dictionary will contain all attributes and properties mentionned in
        the `_print` attribute of the class. Therefore, the `_print` attribute
        must be a tuple containing the name of all attributes and properties of
        the instance to print. If a name doesn't exist as an attribute or
        property, it will be skipped.
        
        Returns
        -------
        dict
            Dictionary with the attributes and properties of the instance to print.
        """
        
        keys = self._print if self._print else [key.strip("_") for key in self.__dict__] # get the list of properties to be printed
        keys = [key for key in keys if hasattr(self, key) and not isinstance(getattr(self, key), type(getattr(self, "__repr__")))] # check if the selected properties exist and are not methods
        return {key: getattr(self, key) for key in keys}

            return copy.copy(self)

[docs]class Mixture:
    """
    .. rubric:: Description
    
    Superclass for mixture model.
    
    A mixture model is described by a collection of weighted components.
    The individual components that are combined to form the mixture model are
    called the *mixture components* and the weights associated with each component
    are called the *mixture weights*.
    
    A mixture distribution is an example of mixture model. The mixture
    components are the individual distributions :math:`X_i` and the mixture weights
    are the probabilities :math:`w_i` associated with each distribution.
    """
    
    # Class attributes
    
    _print = ("components", "weights")
    
    # Constructor
    
    def __init__(self, components, weights = None, component_type = None, relative_weights = False):
        """
        .. rubric:: Initialization

        Parameters
        ----------
        components : list
            List containing the mixture components.
        weights : list of array-like or None, optional
            Weights associated with the mixture components. The list must have the same
            length as the number of components. If None, a weight of 1 is associated
            with each component. The default is None.
        component_type : type, optional
            Type of the mixture components. Each component should have this type.
            If None, components can have any type. The default is None.
        relative_weights : bool, optional
            If True, the weights are considered as probabilities and are normalized
            to get a sum equal to 1. The default is False.

        Raises
        ------
        TypeError
            If `components` don't have the correct type given by `component_type`.
        ValueError
            If `weights` is a list with a different length than the length of `components`.
            If `weights` cannot be broadcast together.
        """
        
        # Control if components have the correct type
        
        if not isinstance(components, (list, tuple)):
            raise TypeError("'components' must be a list containing the mixture components")
            
        if component_type is not None:
            for i, component in enumerate(components):
                if not isinstance(component, component_type):
                    raise TypeError(f"component {i} is not of type '{component_type.__name__}'")
        
        # Initialize the weights if not provided
        
        if weights is None:
            weights = [1]*len(components)
            
        # Control the weights
            
        weights = [np.asarray(weight) for weight in weights]
        
        if len(weights) != len(components):
            raise ValueError(f"'weights' must have the same length as the number of mixture components: their respective lengths are {len(weights)} and {len(components)}")
        elif not util.is_broadcastable(*weights):
            raise ValueError(f"'weights' cannot be broadcast together: their respective shapes are {', '.join([str(weight.shape) for weight in weights])}")
            
        # Define the instance attributes
        
        self._components = list(components)
        self._weights = [weight/sum(weights) for weight in weights] if relative_weights else weights
        self._relative_weights = bool(relative_weights)   
        self._component_type = component_type
    
    # Properties
    
    @property
    def components(self):
        """
        list: Mixture components.
        """
        
        return self._components
    
    @property
    def component_type(self):
        """
        type or None: Type of the mixture components. Each component should have
        this type. If None, components can have any type.
        """
        
        return self._component_type
    
    @property
    def relative_weights(self):
        """
        bool: Indicates whether the weights should be considered as probabilities
        with sum 1 or not.
        """
        
        return self._relative_weights
    
    @property
    def weights(self):
        """
        list of array: Weights associated with the mixture components.
        """
        
        return self._weights
    
    # Setter
    
    @components.setter
    def components(self, value):
        self._components = value
        
    @weights.setter
    def weights(self, value):
        self._weights = value
        
    # Methods

        dictionary["components"] = {"component " + str(i): component for i, component in enumerate(self.components)}

    """
    .. rubric:: Description
    
    Superclass for parametric model.
    
    A parametric model is described by a set of parameters through a `Parameters`
    instance. Therefore, each parameter has a name, type and potential constraints.
    
    The normal distribution is an example of parametric model. The parameters
    are the mean :math:`\mu` and the standard deviation :math:`\sigma` of the
    distribution. Both parameters are scalar type. The mean has no constraints
    while the standard deviation must be strictly positive.
    """
    
    # Constructor
    
    def __init__(self, parameters):
        """
        .. rubric:: Initialization

        Parameters
        ----------
        parameters : Parameters
            Parameters of the instance. The parameters must be initialized and
            broadcast together.

        Raises
        ------
        TypeError
            If `parameters` is not a `Parameters` instance.
        ValueError
            If parameters cannot be broadcast together.
            If parameters are not valid when using the `control` method.
        """
        
        from parameter import Parameters
        
        # Control the parameters

        if not isinstance(parameters, Parameters):
            raise TypeError("'parameters' must be a 'Parameters' instance")
        elif not parameters.is_broadcastable():
            raise ValueError(f"parameters cannot be broadcast together: their respective shapes are {parameters.shapes}")
                       
        parameters.control()
        self._parameters = parameters
            
    # Properties

        class. The parametrization is given through a `Parameters` instance where
        the parameters are not initialized. The `parameters` attribute of an
        instance should be an initialized copy of this parametrization.
        """
        
        raise NotImplementedError()
    
    @property
    def parameters(self):
        """
        Parameters: Parameters of the instance.
        """

        """

        Parameters
        ----------

            Seed to use for the random number generator. If the seed is -1, the

            if the rng propagation is enabled (i.e. if the `propagate_rng`

        """

        Parameters
        ----------
        maturities : array-like
            Maturities of the term structure. Must be along the first axis.
        values : array-like
            Values associated with the maturities of the term structure. The size
            of `values` along the term structure axis must be equal to the size of
            `maturities`.

        Raises
        ------
        ValueError
            If `maturities` is not along the first axis.
            If `values` has not the same size as `maturities` along the term structure axis.
        """
        
        # Control if maturities is along the first axis and contains positive values
        
        maturities = np.asarray(maturities)
        
        if not util.is_along_axis(maturities):
            raise ValueError(f"'maturities' must be along axis 0: its shape is {maturities.shape}")
        elif np.any(maturities < 0):
            raise ValueError("'maturities' must be positive")
            
        self._maturities = maturities
            
        # Control if values has the same size as maturities along the axis
        
        values = np.asarray(values)

        if not util.is_along_axis(values, self.axis, maturities.size):
            raise ValueError(f"'values' must have the same size as 'maturities' along the axis {self.axis}: its shape is {values.shape}")
            
        self._values = values

    @property
    def axis(self):
        """
        int: Axis along which the maturities are (given as a negative integer).
        """
        
        return -self.maturities.ndim

    def maturities(self):
        """
        array: Maturties.
        """
        
        return self._maturities

    @property
    def values(self):
        """
        array: Values associated with the maturities.
        """
        
        return self._values

    def maturities(self, value):
        self._maturities = np.asarray(value)
        
    @values.setter
    def values(self, value):
        self._values = np.asarray(value)

        Note that if the maturity to append already exists, the appending

[docs]    def sort(self, decreasing = False):
        """
        Sort in-place the maturities and values by increasing maturities.
        
        Parameters
        ----------
        decreasing : bool
            If True, the maturities and values will be sorted by decreasing maturities.
            The default is False.

        Returns
        -------
        TermStructure
            Instance.
        """
        
        indices = np.argsort(self.maturities.ravel())
        indices = indices[::-1] if decreasing else indices        
        self.maturities, self.values = util.take(self.maturities, indices, axis = self.axis), util.take(self.values, indices, axis = self.axis)
        return self
