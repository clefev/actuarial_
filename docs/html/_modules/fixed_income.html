[docs]    def aggregate(self, maturities, strict = False):
        """
        Aggregate in-place cash-flows by maturity buckets. The value of all
        cash-flows belonging to a same maturity bucket will be aggregated in 
        one single cash-flow amount. If no cash flow belongs to a maturity bucket,
        the cash-flow amount will be set to 0 for this bucket.

        Parameters
        ----------
        maturities : 1D sorted array-like
            Sorted maturities defining the maturity buckets as follows:
                * Bucket 0: [0, maturities[0]]
                * Bucket 1: (maturities[0], maturities[1]]
                * Bucket 2: (maturities[1], maturities[2]]
                * ...
                * Bucket i: (maturities[i-1], maturities[i]]
        strict : bool, optional
            If True, the inequalities of buckets will change so that bucket i
            becomes [maturities[i-1], maturities[i]) i.e. left closed and right
            open interval. The default is False.

        Returns
        -------
        Instance
        """
        
        # Input maturities must be a 1D array
        
        maturities = np.asarray(maturities).ravel()
        
        # Append the instance final maturity to the input maturities
        
        maturity_max = np.max(self.maturities)
        
        if maturities[-1] < maturity_max:
            maturities = np.append(maturities, maturity_max)
            
        # Map the instance maturities to the maturity buckets
            
        indices = util.upper_bound(self.maturities.ravel(), maturities, index = True, strict = strict) # 1D array
        maturity_buckets = pd.Categorical(maturities[indices], maturities, ordered = True)
        
        # Aggregate the cash-flows by maturity buckets
        
        self.maturities, self.values = util.aggregate_by(self.values, maturity_buckets, axis = self.axis, sort = True)
        
        # Return output
        
        return self

    

[docs]    def cumulate(self):
        """
        Cumulate the cash-flows over the maturities.
        
        Note
        ----
        If the cash-flows are already cumulated ('cumulative' attribute set to True),
        then the instance is returned as is. 

        Returns
        -------
        Instance
            Instance with cumulated cash-flow values over the maturities.

        """
        
        if not self.cumulative:
            self.values = np.cumsum(self.values, axis = self.axis)
            self.cumulative = True
            
        return self

        """
        Decumulate the cash-flows over the maturities.
        
        Note
        ----
        If the cash-flows are already decumulated ('cumulative' attribute set to False),
        then the instance is returned as is. 

        Returns
        -------
        Instance
            Instance with decumulated cash-flow values over the maturities.

        """
        
        if self.cumulative:
            self.values = np.diff(self.values, axis = self.axis, prepend = 0)
            self.cumulative = False
            
        return self

 

[docs]    def sum(self, keepdims = False):
        """
        Compute the sum of cash-flows over the maturities.

        Parameters
        ----------
        keepdims : bool, optional
            If True, the axis which is reduced is left in the result as dimension
            with size one (otherwise, it will be removed). With this option, the
            result will broadcast correctly against the initial cash-flows.
            The default is False.
            
        Returns
        -------
        scalar or array
            Summed cash-flows.
        """
        
        return self.values.sum(axis = self.axis, keepdims = keepdims)

    

[docs]    def weights(self, cumulative = False):
        """
        Compute the relative weight (or percentage) of each cash-flow.
        
        Note
        ----
        The weight of a cash-flow corresponds to its relative size compared
        to the total sum of cash-flows over the maturities:

        Parameters
        ----------
        cumulative : bool, optional
            If True, the weights are cumulated to reach 100% at the final maturity.
            The default is False.

        Returns
        -------
        array
            Array with a float dtype and the same shape as the `values` attribute.
        """
        
        if cumulative:
            return (np.cumsum(self.values, axis = self.axis) / self.sum(keepdims = True)).astype(float)
        else:
            return (self.values / self.sum(keepdims = True)).astype(float)

class FixedIncome(Base, Broadcast):
    
    # Class attributes
    
    _print = ("maturities", "cash_flows", "yield_curve", "kind")
    kinds = ("compound_rate", "continuous_rate", "simple_rate")
    
    # Constructor
    
    def __init__(self, maturities, cash_flows, yield_curve = None, kind = "compound_rate"):
        
        # Control the kind
                
        if kind not in self.kinds:
            raise ValueError(f"kind must be one of the following: {', '.join(self.kinds)}")
        
        # Control if yield_curve is a YieldCurve instance
        
        if yield_curve is None:
            yield_curve = FlatYieldCurve(0.)
        elif not isinstance(yield_curve, YieldCurve):
            raise TypeError("yield_curve must be a YieldCurve instance")
            
        # Control if maturities are along the first axis
        
        maturities = np.asarray(maturities)
        
        if not util.is_along_axis(maturities):
            raise ValueError(f"maturities must be along axis 0: its shape is {maturities.shape}")
            
        # Control if maturities, cash_flows and yield_curve can be broadcast
        
        cash_flows = np.asarray(cash_flows)
        
        if not util.is_broadcastable(maturities, cash_flows, yield_curve):
            raise ValueError(f"maturities, cash_flows and yield_curve cannot be broadcast together: their respective shapes are {maturities.shape}, {cash_flows.shape} and {yield_curve.shape}")
            
        # Define the instance attributes
        
        self._maturities = maturities
        self._cash_flows = cash_flows
        self._yield_curve = yield_curve
        self._kind = kind
        
    # Properties

    def asymetry(self, dollar = False, keepdims = False):
        """
        Compute the modified asymetry.

        Parameters
        ----------
        dollar : bool, optional
            If True, compute the modified dollar asymetry. The default is False.
        keepdims : bool, optional
            If True, the axis along which the maturities are is kept with a size one.
            Otherwise, it is reduced. The default is False.

        Returns
        -------
        scalar or array
            Modified asymetry.

        """
        
        # Compute the modified dollar asymetry
        
        if self.kind == "compound_rate":
            dollar_asymetry = (self.maturities**3 + 3*self.maturities**2 + 2*self.maturities) * self.cash_flows * self.discount_factors**(1 + np.divide(3, self.maturities, where = (self.maturities != 0)))
        elif self.kind == "continuous_rate":
            dollar_asymetry = self.maturities**3 * self.cash_flows * self.discount_factors
        elif self.kind == "simple_rate":
            dollar_asymetry = 6 * self.maturities**3 * self.cash_flows * self.discount_factors**4
            
        # Return output
        
        if dollar:
            return np.sum(dollar_asymetry, axis = self.axis, keepdims = keepdims)
        else:
            return np.sum(dollar_asymetry / self.present_value(keepdims = True), axis = self.axis, keepdims = keepdims)
    
    def broadcast(self, *args):
        """
        Mimics the broadcast of the maturities, cash-flows and yield curve with
        input arrays.

        Parameters
        ----------
        *args : array_like
            The arrays to broadcast with the maturities, cash-flows and yield curve.
            If no array is provided, only the maturities, cash-flows and yield curve
            are broadcast together.
            
        Returns
        -------
        np.broadcast
            Broadcast result.

        """
        
        return util.broadcast(self.maturities, self.cash_flows, self.yield_curve, *args)
    
    def convexity(self, dollar = False, keepdims = False):
        """
        Compute the modified convexity.

        Parameters
        ----------
        dollar : bool, optional
            If True, compute the modified dollar convexity. The default is False.
        keepdims : bool, optional
            If True, the axis along which the maturities are is kept with a size one.
            Otherwise, it is reduced. The default is False.

        Returns
        -------
        scalar or array
            Modified convexity.

        """
        
        # Compute the modified dollar convexity
        
        if self.kind == "compound_rate":
            dollar_convexity = (self.maturities**2 + self.maturities) * self.cash_flows * self.discount_factors**(1 + np.divide(2, self.maturities, where = (self.maturities != 0)))
        elif self.kind == "continuous_rate":
            dollar_convexity = self.maturities**2 * self.cash_flows * self.discount_factors
        elif self.kind == "simple_rate":
            dollar_convexity = 2 * self.maturities**2 * self.cash_flows * self.discount_factors**3
            
        # Return output
        
        if dollar:
            return np.sum(dollar_convexity, axis = self.axis, keepdims = keepdims)
        else:
            return np.sum(dollar_convexity / self.present_value(keepdims = True), axis = self.axis, keepdims = keepdims)
     
    def duration(self, dollar = False, keepdims = False):
        """
        Compute the modified duration.

        Parameters
        ----------.
        dollar : bool, optional
            If True, compute the modified dollar duration. The default is False.
        keepdims : bool, optional
            If True, the axis along which the maturities are is kept with a size one.
            Otherwise, it is reduced. The default is False.

        Returns
        -------
        scalar or array
            Modified duration.

        """
        
        # Compute the modified dollar duration
        
        if self.kind == "compound_rate":
            dollar_duration = self.maturities * self.cash_flows * self.discount_factors**(1 + np.divide(1, self.maturities, where = (self.maturities != 0)))
        elif self.kind == "continuous_rate":
            dollar_duration = self.maturities * self.cash_flows * self.discount_factors
        elif self.kind == "simple_rate":
            dollar_duration = self.maturities * self.cash_flows * self.discount_factors**2
                    
        # Return output
        
        if dollar:
            return np.sum(dollar_duration, axis = self.axis, keepdims = keepdims)
        else:
            return np.sum(dollar_duration / self.present_value(keepdims = True), axis = self.axis, keepdims = keepdims)
    
    def present_value(self, aggregate = True, keepdims = False):
        """
        Compute the present value of cash-flows.

        Parameters
        ----------
        aggregate : bool, optional
            If False, the present value of individual cash-flows is returned (not summed).
            Otherwise, they are aggregated along the maturities (summed). The default is True.
        keepdims : bool, optional
            When aggregating and if True, the axis along which the maturities are
            is kept with a size one. Otherwise, it is reduced when aggregating.
            This option has no impact if there is no aggregation. The default is False.
            
        Returns
        -------
        scalar or array
            Present value.

        """
        
        if aggregate:
            return np.sum(self.cash_flows * self.discount_factors, axis = self.axis, keepdims = keepdims)
        else:
            return self.cash_flows * self.discount_factors
    
    def yield_to_maturity(self, present_value, tol = 1e-4, keepdims = False):
        """
        Compute the yield to maturity (YTM).
        
        Note
        ----
        The yield to maturity is the constant rate allowing to get the present market
        price when discounting the cash-flows with this rate. The type of the rate
        is given by the 'kind' attribute of the instance. While the present market
        price is provided in the 'present_value' argument of the method.
        Since no close formula exists to compute the yield to maturity, it is calculated
        using the Newton-Raphson method.

        Parameters
        ----------
        present_value : array_like
            Present market price.
        tol : float, optional
            Absolute tolerance for the yield to maturity. The maximum absolute error between
            the computed yield to maturity and the exact one will be equal to this tolerance.
            The default is 1e-4 (1bp).
        keepdims : bool, optional
            If True, the axis along which the maturities are is kept with a size one.
            Otherwise, it is reduced. The default is False.
            
        Raises
        ------
        ValueError
            If 'present_value' has a size greater than one along the 'axis' attribute of the instance.
            If 'present_value' cannot be broadcast with the FixedIncome instance.

        Returns
        -------
        scalar or array
            Yield to maturity.

        """
        
        return self.copy().set_attributes(yield_curve = FlatYieldCurve(0.)).z_spread(present_value, tol = tol, keepdims = keepdims)
        
    def z_spread(self, present_value, tol = 1e-4, keepdims = False):
        """
        Compute the z-spread.
        
        Note
        ----
        The z-spread is a spread or a parallel shift over the yield curve in order
        to get the present market price when discounting the cash-flows. The yield
        curve used is the one from the 'yield_curve' attribute converted to the
        rate given by the 'kind' attribute. While the present market price is provided
        in the 'present_value' argument of the method.
        Since no close formula exists to compute the z-spread, it is calculated
        using the Newton-Raphson method.

        Parameters
        ----------
        present_value : array_like
            Present market price.
        tol : float, optional
            Absolute tolerance for the z-spread. The maximum absolute error between
            the computed z-spread and the exact one will be equal to this tolerance.
            The default is 1e-4 (1bp).
        keepdims : bool, optional
            If True, the axis along which the maturities are is kept with a size one.
            Otherwise, it is reduced. The default is False.
            
        Raises
        ------
        ValueError
            If 'present_value' has a size greater than one along the 'axis' attribute of the instance.
            If 'present_value' cannot be broadcast with the FixedIncome instance.

        Returns
        -------
        scalar or array
            Z-spread.

        """
        
        # Control if present_value can be broadcast correctly
        
        present_value = np.asarray(present_value)
        
        if present_value.ndim >= -self.axis and present_value.shape[self.axis] != 1:
            raise ValueError(f"'present_value' must have a size 1 along the axis {self.axis}")
        elif not self.is_broadcastable(present_value):
            raise ValueError(f"'present_value' cannot be broadcast with the FixedIncome instance: their respective shapes are {present_value.shape} and {self.shape}")
            
        # Copy of the FixedIncome instance where the z-spread can be added without modifying the original instance
        
        fixed_income = self.copy()
        
        # Define the target function for which the root (z-spread) is sought
          
        def f(z_spread):
            fixed_income.yield_curve = MixtureYieldCurve([self.yield_curve, FlatYieldCurve(z_spread, kind = self.kind)], kind = self.kind) # add the z-spread to the original yield curve
            return fixed_income.present_value(keepdims = True) - present_value
        
        # Define the derivative of the target function (opposite of the modified dollar duration)
        
        def df(z_spread):
            return -fixed_income.duration(dollar = True, keepdims = True)
        
        # Newton-Raphson method to find the root (z-spread) of the target function
        
        z_spread = math.newton_raphson(f, df, 0., tol = tol)["root"]
        
        # Return output
        
        if keepdims:
            return z_spread
        else:
            return np.squeeze(z_spread, self.axis)[()]

    def __len__(self):
        """
        Number of maturities (length of the 'maturities' attribute).
        """
        
        return len(self.maturities)
