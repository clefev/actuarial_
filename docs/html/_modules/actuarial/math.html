def argmax(x, axis = None, keepdims = False, skipna = False):
        
    if skipna:
        argmax = np.nanargmax(x, axis = axis)
    else:
        argmax = np.argmax(x, axis = axis)
        
    if not keepdims:
        return argmax
    elif axis is None:
        return np.broadcast_to(argmax, shape = (1,)*np.asarray(x).ndim)
    elif axis is not None:
        return np.expand_dims(argmax, axis = axis)

[docs]def bisection(f, lower, upper, maxiter = 100, tol = 1e-4, f_tol = 0., kwargs = {}):
    r"""
    Bisection algorithm to find the root of a continuous function in an interval
    :math:`[a,b]`.
    
    The algorithm supports vectorization.
    It means that if the function returns *n* values, then *n* roots will be found.
    Note that the algorithm will stop as soon as one of the criteria given by
    `maxiter`, `tol` or `f_tol` is satisfied (not when both are satisfied at the
    same time).
    
    Parameters
    ----------
    f : callable
        Function whose root must be found.
        The function must be continuous and return a scalar or an array.
        Moreover, :math:`f(a)` and :math:`f(b)` must have opposite signs.
        If an array is returned, a root will be found for each returned component
        of the function (vectorization).
        The function will be called as `f(x, **kwargs)` where `x` is the value
        for which the function should be evaluated and `kwargs` is a dictionary
        of additional arguments that can be passed to the function (see below).
    lower : array_like
        Lower bound :math:`a` of the interval in which the root is.
    upper : array_like
        Upper bound :math:`b` of the interval in which the root is.
    maxiter : int, optional
        Maximum number of iterations.
        The algorithm will stop and return the current result after this number
        of iterations.
        The default is 100.
    tol : float, optional
        Absolute tolerance for the root :math:`x^*`.
        The maximum absolute error between the computed root :math:`x^*` and the
        exact root :math:`x_{exact}` will be equal to this tolerance.
        The default is 1e-4.
    f_tol : float, optional
        Absolute tolerance for :math:`f(x^*)`.
        The maximum absolute value for :math:`f(x^*)` will be equal to this tolerance.
        The default is 0.
    kwargs : dict, optional
        Additional keyword arguments for the function `f`.

    Raises
    ------
    ValueError
        If `upper` is not greater than `lower`.
        If `f(lower)` and `f(upper)` don't have opposite signs.

    Returns
    -------
    root : scalar or array
        Computed root.
    f_root : scalar or array
        Value of the function for the computed root.
    iter : int
        Number of iterations needed to find the root.
    """
    
    # Control if upper is greater than lower
    
    lower, upper = np.asarray(lower, dtype = float), np.asarray(upper, dtype = float)
    
    if not util.is_broadcastable(lower, upper):
        raise ValueError(f"'lower' and 'upper' cannot be broadcast together: their respective shapes are {lower.shape} and {upper.shape}")        
    elif not np.all(lower <= upper):
        raise ValueError("'upper' must be greater than 'lower'")
        
    # Evaluate the function f(lower) and f(upper)
    
    f_lower, f_upper = np.asarray(f(lower, **kwargs), dtype = float), np.asarray(f(upper, **kwargs), dtype = float)
    
    # Control if f(lower) and f(upper) have different signs
    
    if not util.is_broadcastable(f_lower, f_upper):
        raise ValueError(f"'f(lower)' and 'f(upper)' cannot be broadcast together: their respective shapes are {f_lower.shape} and {f_upper.shape}")        
    elif not np.all(f_lower*f_upper <= 0):
        raise ValueError("'f(lower)' and 'f(upper)' must have opposite signs")
        
    # Broadcast the arrays
    
    shape = np.broadcast(f_lower, f_upper).shape
    lower, upper = np.broadcast_to(lower, shape).copy(), np.broadcast_to(upper, shape).copy()
    f_lower, f_upper = np.broadcast_to(f_lower, shape).copy(), np.broadcast_to(f_upper, shape).copy()
    
    # Bisection algorithm
      
    for i in range(int(maxiter)):
        
        # Compute a candidate between lower and upper and evaluate the function at this point
        
        candidate = np.asarray(0.5*(lower + upper), dtype = float) # middle point between lower and upper
        f_candidate = np.asarray(f(candidate, **kwargs), dtype = float)
        
        # Stop if the candidate approximates sufficiently well the root
        
        if np.all((np.abs(f_candidate) <= f_tol) | ((candidate - lower) <= tol)):
          break
        
        # Update the interval [lower, upper]
        
        I = (f_lower*f_candidate > 0) # indicate if f_lower and f_candidate have the same sign i.e. if the second interval is retained or not
        lower[I], upper[~I] = candidate[I], candidate[~I]
        f_lower[I], f_upper[~I] = f_candidate[I], f_candidate[~I]
        
    # Return output
    
    return {"root": candidate[()], "f_root": f_candidate[()], "iter": i + 1}

    
    
    


def cummax(x, axis = None, dtype = None, skipna = False):
    """
    Cumulative maximum of array elements.

    Parameters
    ----------
    x : array_like
        Input array.
    axis : int or None, optional
        Axis along which the cumulative maximum is computed. If None, the cumulative maximum is computed
        over the flattened array. The default is None.
    dtype : dtype, optional
        The type of the returned array. If None, the dtype of x is used. The default is None.
    skipna : bool, optional
        If True, NaN values will be ignored  (otherwise, a NaN will be returned).
        The default is False.
        
    Returns
    -------
    maximum : array
        If axis is None, a 1D array is returned with the same size as x.
        Otherwise, an array is returned with the same shape as x.

    """

    if axis is None:
        x = x.ravel()
        axis = 0
        
    if skipna:
        return np.fmax.accumulate(x, axis = axis, dtype = dtype)
    else:
        return np.maximum.accumulate(x, axis = axis, dtype = dtype)

def diff(x, lag = 1, axis = None):
    """
    Calculate the lagged differences along an axis.

    Parameters
    ----------
    x : array_like
        Input array.
    lag : int or 1D array_like of int, optional
        Lag to use to compute the differences.
        If an integer is provided, the difference is given by diff[i] = x[i+lag] - x[i] along the axis.
        If a one-dimensional array is provided, the difference is given by diff[i] = x[lag[i+1]] - x[lag[i]] along the axis.
        Therefore, the returned array will have the same shape as x except along the axis where the dimension will be
        smaller by lag if an integer is provided and equal to len(lag)-1 if a one-dimensional array is provided.
        The default is 1.
    axis : int or None, optional
        The axis along which the difference is taken. If None, the input array
        will be flattened. The default is None.

    Returns
    -------
    diff : array
        Lagged differences.

    """
    
    # Convert x to an array
    
    x = np.asanyarray(x)
    
    # Flatten x if no axis is provided
    
    if axis is None:
        x = x.ravel()
        axis = 0
        
    # Compute the lagged differences
        
    if isinstance(lag, (int, float)):
        differences = np.take(x, range(int(lag), x.shape[axis]), axis = axis) - np.take(x, range(0, x.shape[axis] - int(lag)), axis = axis)
    else:
        differences = np.take(x, lag[1:], axis = axis) - np.take(x, lag[:-1], axis = axis)
        
    # Return output
    
    return differences

[docs]def newton_raphson(f, df, x0, maxiter = 100, tol = 1e-4, f_tol = 0., kwargs = {}):
    r"""
    Newton-Raphson algorithm to find the root of a continuous function.
    
    The algorithm supports vectorization.
    It means that if the function returns *n* values, then *n* roots will be found.
    Note that the algorithm will stop as soon as one of the criteria given by
    `maxiter`, `tol` or `f_tol` is satisfied (not when both are satisfied at the
    same time).

    Parameters
    ----------
    f : callable
        Function whose root must be found.
        The function must be continuous and return a scalar or an array.
        If an array is returned, a root will be found for each returned component
        of the function (vectorization).
        The function will be called as `f(x, **kwargs)` where `x` is the value
        for which the function should be evaluated and `kwargs` is a dictionary
        of additional arguments that can be passed to the function (see below).
    df : callable
        Derivative of the function `f`.
        The derivative will be called as `df(x, **kwargs)`.
    x0 : array-like
        Initial guess :math:`x_0` for the root.
    maxiter : int, optional
        Maximum number of iterations.
        The algorithm will stop and return the current result after this number
        of iterations.
        The default is 100.
    tol : float, optional
        Absolute tolerance for the root :math:`x^*`.
        The maximum absolute error between the computed root :math:`x^*` and the
        exact root :math:`x_{exact}` will be equal to this tolerance.
        The default is 1e-4.
    f_tol : float, optional
        Absolute tolerance for :math:`f(x^*)`.
        The maximum absolute value for :math:`f(x^*)` will be equal to this tolerance.
        The default is 0.
    kwargs : dict, optional
        Additional keyword arguments for the function `f` and its derivative `df`.

    Returns
    -------
    root : scalar or array
        Computed root.
    f_root : scalar or array
        Value of the function for the computed root.
    iter : int
        Number of iterations needed to find the root.
    """

    # Initialize
    
    x0 = np.asarray(x0, dtype = float)
    f_x0 = np.asarray(f(x0, **kwargs), dtype = float)
    df_x0 = np.asarray(df(x0, **kwargs), dtype = float) 
    
    # Newton-Raphson algorithm
    
    for i in range(int(maxiter)):
        
        # Compute the new candidate x1 and evaluate the function at this point
        
        x1 = np.asarray(x0 - f_x0/df_x0, dtype = float)
        f_x1 = np.asarray(f(x1, **kwargs), dtype = float)
        
        # Stop if the candidate approximates sufficiently well the root
        
        if np.all((np.abs(f_x1) <= f_tol) | (np.abs(x1 - x0) <= tol)):
            break
        
        # Update the point x0, the value of the function at this point and the value of its derivative
        
        x0 = x1
        f_x0 = f_x1
        df_x0 = np.asarray(df(x1, **kwargs), dtype = float)
        
    # Return output
    
    return {"root": x1[()], "f_root": f_x1[()], "iter": i + 1}



    
### norm



def norm(x, p = 2, axis = 0, keepdims = False, skipna = False):
    
    # p-norm
    
    if p == 0:
        norm = sum(x != 0, axis = axis, keepdims = keepdims, skipna = skipna)
    elif p == np.inf:
        norm = max(np.abs(x), axis = axis, keepdims = keepdims, skipna = skipna)
    elif p == -np.inf:
        norm = min(np.abs(x), axis = axis, keepdims = keepdims, skipna = skipna)
    else:
        norm = sum(np.abs(x)**p, axis = axis, keepdims = keepdims, skipna = skipna)**(1/p)
        
    # Return output (array with the same shape as x with the specified axis removed if keepdims is set to False or of size 1 if keepdims is set to True)
    
    return norm
        
        

### pareto_frontier



def pareto_frontier(x, y, bottom = False, right = False):
    
    # Control
    
    assert len(x) == len(y), "x and y must have the same length"
    
    # Flatten x and y
    
    x = np.asarray(x).ravel()
    y = np.asarray(y).ravel()
    
    # Sort x and y
    
    order = util.argsort(x, decreasing = right)
    x = x[order]
    y = y[order]
    
    # Find efficient points
    
    if bottom:
        *_, index = np.unique(cummin(y), return_index = True)
    else:
        *_, index = np.unique(cummax(y), return_index = True)
        
    # Return output (dictionary)
    
    return {"x": x[index], "y": y[index], "index": order[index]}




    
def pmax(*args, skipna = False):
    """
    Element-wise maximum of array elements.

    Parameters
    ----------
    *args : array_like
        Input arrays. The arrays must be broadcastable to a common shape
        (which becomes the shape of the output).
    skipna : bool, optional
        If True, NaN values will be ignored (otherwise, a NaN will be returned).
        The default is False.
        
    Returns
    -------
    pmax : scalar or array
        A scalar is returned if all arguments are scalars. Otherwise, an array
        is returned.

    """
    
    max = np.fmax if skipna else np.maximum
    
    if len(args) == 0:
        return None
    if len(args) == 1:
        return args[0] if np.isscalar(args[0]) else np.asanyarray(args[0])
    elif len(args) == 2:
        return max(*args)
    else:
        return pmax(*(max(*args[:2]),) + args[2:], skipna = skipna)
