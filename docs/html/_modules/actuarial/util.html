[docs]def aggregate_by(x, by, func = "sum", sort = False, axis = 0, kwargs = {}):
    """
    Aggregates data by groups along an axis. The data belonging to a same
    group will be aggregated together using the `func` function.

    Parameters
    ----------
    x : array-like
        Input array containing the data to be aggregated by groups.
    by : 1D array-like or pd.Categorical
        The values determine the groups along the axis of the input array.
        Therefore, the length must correspond to the size of `x` along the axis.
        Data belonging to a same group will be aggregated together.
    func : string or callable, optional
        Function to use to aggregate data belonging to a same group. It can
        be a callable or a string such as "sum", "mean", "std", "var", "min", "max",
        "median", "prod" or "count". The default aggregation is a sum.
    sort : bool, optional
        If True, the output groups will be sorted in ascending order.
        Otherwise, they are given in their order of appearance in the input `by`.
        The default is False.
    axis : int, optional
        Axis along which the data are aggregated. The default is the first axis
        of `x`.
    kwargs : dict, optional
        Additional keyword arguments for the `func` function when a callable is
        passed.

    Returns
    -------
    tuple    
        groups : array
            Output groups after aggregation. The groups are sorted in ascending
            order if the `sort` option is activated. The returned array is along
            the aggregation axis and can be broadcast with the `aggregate` output.
        aggregate : array
            Aggregated values by groups. The shape is the same as the input array `x`
            except that the size along the axis is equal to the number of output
            groups `len(groups)`.
        
    Raises
    ------
    ValueError
        If the length of `by` doesn't correspond to the size of `x` along the axis.
    """
    
    # Convert x to array
    
    x = np.asarray(x)
    ndim = x.ndim
    shape = x.shape
    size = shape[axis]
    
    # Convert by to categorical variable
    
    by = pd.Categorical(by)
    
    if len(by) != size:
        raise ValueError(f"the length of 'by' argument must be equal to {size} and not {len(by)}")
        
    # Reshape x to a matrix such that rows represent the axis to be aggregated
    
    x = np.moveaxis(x, axis, 0) # move the axis to the first position
    x = x.reshape(size, -1) # matrix where rows will be aggregated
    
    # Aggregate the values

    x = pd.DataFrame(x).groupby(by, sort = sort).agg(func, **kwargs)
    groups, aggregate = x.index.to_numpy(), x.to_numpy()
    del x
    
    # Reshape aggregated values such that its shape is the same as the initial shape of x except that the size along the axis is equal to len(groups)
    
    aggregate = aggregate.reshape(-1, *np.delete(shape, axis))
    aggregate = np.moveaxis(aggregate, 0, axis)
    
    # Return output

    return reshape_along_axis(groups, -normalize_axis(axis, ndim, negative = True)), aggregate

    """

    Note
    ----
    NaN values will be put in last position if decreasing is False and in first position if
    it is True. It means that NaN values are greater than all other values (even np.inf).

    -------
    indices : array of int

        has the same shape as x if `axis` is an integer and is flattened if `axis`
        is None.
        
    """

        return take(np.argsort(x, axis = axis), slice(None, None, -1), axis = axis)
    else:
        return np.argsort(x, axis = axis)

[docs]def broadcast(*args):
    """
    Mimics the broadcast of input arguments.
    
    The broadcast is based on the `shape` attribute of each argument. If an argument
    has no `shape` attribute, it will be converted to an array in order to get its shape.

    Parameters
    ----------
    *args : array-like or object with shape attribute
        Inputs to be broadcast. It can be array-like or any object having a
        `shape` attribute.

    Returns
    -------
    np.broadcast
        Broadcast result.
        
    Raises
    ------
    ValueError
        If the inputs cannot be broadcast together.
    """
    
    return np.broadcast(*[np.empty(shape(arg), dtype = []) for arg in args])




[docs]def concatenate(*args, axis = 0):
    """
    Concatenate the input arrays into a single array.

    Note
    ----
    The arrays can be concatenated if they can be broadcast against each other
    except along the given axis.

    Parameters
    ----------
    *args : array_like
        The arrays to be concatenated against each other.
    axis : int or None, optional
        The axis along which the arrays will be concatenated (compared to the array with the greatest number of dimensions).
        If None, the input arrays will be flattened. The default is 0.
        
    Raises
    ------
    ValueError
        If the arrays cannot be concatenated.

    Returns
    -------
    concatenated_array : array
        The concatenated array has the same number of dimensions as the greatest
        number of dimensions of the input arrays. The size along each dimension
        corresponds to the broadcast one except along the given axis where it
        is the cumulated one.

    """
    
    # Flat concatenation if axis is None
    
    if axis is None:
        return np.concatenate(args, axis = None)
    
    # Initialize
    
    args = [np.asarray(arg) for arg in args]
    shapes = [arg.shape for arg in args]
    concatenated_shape = concatenate_shapes(*shapes, axis = axis)
    axis = normalize_axis(axis, ndim = len(concatenated_shape), negative = True)
    
    # Broadcast each argument to reach the correct number of dimensions
    
    for i, arg in enumerate(args):
        
        new_shape = list(concatenated_shape)
        
        try:
            new_shape[axis] = shapes[i][axis]
        except IndexError:
            new_shape[axis] = 1
            
        new_shape = tuple(new_shape)
            
        if new_shape != shapes[i]:
            args[i] = np.broadcast_to(args[i], shape = new_shape)
        
    # Return output
    
    return np.concatenate(args, axis = axis)

    
    
     

[docs]def concatenate_shapes(*args, axis = 0):
    """
    Concatenate the input shapes into a single shape.
    
    Note
    ----
    The shapes can be concatenated if they can be broadcast against each other
    except along the given axis.

    Parameters
    ----------
    *args : tuple of int
        The shapes to be concatenated against each other.
    axis : int or None, optional
        The axis along which the shapes will be concatenated (compared to the shape with the greatest number of dimensions).
        If None, the input shapes will be flattened. The default is 0.

    Raises
    ------
    ValueError
        If the shapes cannot be concatenated.

    Returns
    -------
    concatenated_shape : tuple of int
        The concatenated shape has the same number of dimensions as the greatest
        number of dimensions of the input shapes. The size along each dimension
        corresponds to the broadcast one except along the given axis where it
        is the cumulated one.

    """
    
    # Flat concatenation if axis is None
    
    if axis is None:
        return (np.sum([np.prod(shape, dtype = int) for shape in args]),)
    
    # Define a function to check if two shapes can be concatenated
    
    def is_concatenable(shape1, shape2):
            
        for i, (a, b) in enumerate(zip(shape1[::-1], shape2[::-1])):
            if a == 1 or b == 1 or a == b or axis == -(i+1):
                pass
            else:
                return False
            
        return True
    
    # Initialize
    
    ndim = max([len(shape) for shape in args]) # greatest number of dimensions of the input shapes
    axis = normalize_axis(axis, ndim = ndim, negative = True)
    
    # Concatenate the shapes
    
    concatenated_shape = [1]*ndim
    concatenated_shape[axis] = 0
        
    for shape in args:
        
        if not is_concatenable(concatenated_shape, shape):
            raise ValueError(f"shape {shape} cannot be concatenated with the previous concatenated ones {tuple(concatenated_shape)} along the axis {axis}")
            
        for i, (a, b) in enumerate(zip(concatenated_shape[::-1], shape[::-1])):
            if -(i+1) != axis:
                concatenated_shape[-(i+1)] = max(a, b)
            
        try:
            concatenated_shape[axis] += shape[axis]
        except IndexError:
            concatenated_shape[axis] += 1
            
    # Return output
    
    return tuple(concatenated_shape)          

        


[docs]def count_bool(x, axis = None, keepdims = False, invert = False):
    """
    Counts the number of True values in an array.

    Parameters
    ----------
    x : array-like or MaskedArray
        Input array. If a masked array is provided, the masked values will be
        ignored in the count.
    axis : int or tuple of int or None, optional
        Axis or tuple of axes along which to count True values. If None, counts
        the total number of True values in the input array (along all axes).
        The default is None.
    keepdims : bool, optional
        If True, the axes which are reduced are left in the result as dimensions
        with size one (otherwise, they will be removed). With this option, the
        result will broadcast correctly against the input array.
        The default is False.
    invert : bool, optional
        If True, counts the number of False values instead of the number of True
        values. The default is False.

    Returns
    -------
    int or array of int
        Number of True values.
    """
    
    # Get the mask of x
    
    x = np.asanyarray(x)
    mask = np.ma.getmask(x)
    
    # Count the number of True values (or False values if invert is True) along the given axis
    
    if mask is np.ma.nomask: # x has no mask
    
        if invert:
            count = size(x, axis = axis) - np.count_nonzero(x, axis = axis, keepdims = keepdims)
        else:
            count = np.count_nonzero(x, axis = axis, keepdims = keepdims)
            
    else: # x has a mask
    
        particular_case = axis is None and not keepdims # particular case because the function np.count_nonzero doesn't take into account the mask in this situation (we need to force keepdims to True)
        
        if invert:
            count = np.count_nonzero(~x, axis = axis, keepdims = True if particular_case else keepdims)
        else:
            count = np.count_nonzero(x, axis = axis, keepdims = True if particular_case else keepdims)
        
        if particular_case:
            count = int(count.data)
            
    # Return output
    
    return np.ma.getdata(count)[()]




[docs]def count_mask(x, axis = None, keepdims = False, invert = False):
    """
    Counts the number of masked values in an array.

    Parameters
    ----------
    x : array-like or MaskedArray
        Input array. If the array has no mask, the number of masked values will be 0.
    axis : int or tuple of int or None, optional
        Axis or tuple of axes along which to count the number of masked values.
        If None, counts the total number of masked values in the input array
        (along all axes). The default is None.
    keepdims : bool, optional
        If True, the axes which are reduced are left in the result as dimensions
        with size one (otherwise, they will be removed). With this option, the
        result will broadcast correctly against the input array.
        The default is False.
    invert : bool, optional
        If True, counts the number of unmasked values instead of the number of
        masked values. The default is False.
        
    Returns
    -------
    int or array of int
        Number of masked values.
    """
    
    # Get the mask of x
    
    x = np.asanyarray(x)
    mask = np.ma.getmask(x)
    
    # Count the number of masked values along the given axis
    
    if mask is np.ma.nomask: # x has no mask
        count = 0 if axis is None else np.full(np.delete(x.shape, axis), 0)
    else: # x has a mask
        count = np.count_nonzero(mask, axis = axis)
        
    # Count the number of unmasked values if invert = True
    
    if invert:
        count = size(x, axis = axis) - count
        
    # Return output
        
    if not keepdims:
        return count
    elif axis is None:
        return np.broadcast_to(count, shape = (1,)*x.ndim)
    else:
        return np.expand_dims(count, axis = axis)

[docs]def dict_to_string(dictionary, bullet = "", quote = False):
    """
    Convert a dictionary to a string. The returned string will be pretty printed
    with a new line for each key-value pair and a correct indent level.

    Parameters
    ----------
    dictionary : dict
        The dictionary to convert to a string.
    bullet : str, optional
        A string to add before the key (typically, a bullet point). The default is "".
    quote : bool, optional
        If True, the keys will be in quotes. The default is False.

    Returns
    -------
    string : str
        Print this string will give a console-friendly dictionary output.

    """
    
    # Initialize an empty string
    
    string = ""
        
    # Add each dictionary item to the string
        
    for key, value in dictionary.items():
        
        # Add the key
        
        key = f"'{key}'" if quote else key # add quotation marks
        key_string = f"{bullet}{key}: "
        string += key_string
        
        # Add the value
        
        if isinstance(value, dict): # nested dictionary
            string += "\n" + dict_to_string(value, bullet = "\t" + bullet, quote = quote) + "\n"
        else:
            value_string = str(value).replace("\n", "\n" + "\t"*key_string.count("\t") + " "*len(key_string.replace("\t", ""))) # if the value is printed on multiple lines, add a correct indent level for each line
            string += value_string + "\n"
                            
    # Return output 
        
    return string.rstrip()

    """
    Return the values of a dictionary for the given keys.

    Parameters
    ----------
    dictionary : dict
        Input dictionary.
    key : dict_key or list of dict_key
        Key or list of keys to be searched in the dictionary.
        If a single key is provided, the output will be the value corresponding to this key.
        If a list of keys is provided, the output will be a dictionary with the corresponding key-value pairs.
    default : any, optional
        Defaut value to use if a key doesn't exist and if errors is set to "fill".
        The default is None.
    errors : {"raise", "fill", "remove"}, optional
        If "raise", invalid keys will raise a KeyError.
        If "fill", invalid keys will return the default argument as corresponding value.
        If "remove", invalid keys will be removed.
        The default is "raise".

    Returns
    -------
    dict_value or dict
        A dictionary value will be returned if a single key is provided and a
        dictionary will be returned if multiple keys are provided.
        
    Raises
    ------
    TypeError
        If dictionary is not a dict instance.
    ValueError
        If the argument errors has not a valid value.

    """
    
    # Control if dictionary is a dict instance
    
    if not isinstance(dictionary, dict):
        raise TypeError("dictionary must be a dict instance")
        
    # Control if the argument errors has a valid value
    
    errors = errors.lower()
    _errors = ("raise", "fill", "remove") # list of valid errors
    
    if errors not in _errors:
        raise ValueError(f"errors must be one of the following: {', '.join(_errors)}")
        
    # Get item
            
    try:
        if errors == "raise":
            return dictionary[key]
        elif errors == "fill":
            return dictionary.get(key, default)
        elif errors == "remove":
            return dictionary.get(key)
    except TypeError:
        if errors == "raise":
            return {k: dictionary[k] for k in key}
        elif errors == "fill":
            return {k: dictionary.get(k, default) for k in key}
        elif errors == "remove":
            return {k: dictionary[k] for k in key if k in dictionary}

[docs]def is_sorted(x, strict = False, decreasing = False):
    """
    Indicates wether an input array is sorted or not.

    Parameters
    ----------
    x : 1D array-like
        Input array.
    strict : bool, optional
        If True, controls wether the input array is strictly sorted.
        The default is False.
    decreasing : bool, optional
        If True, controls if the input array is sorted in decreasing order instead
        of in increasing order. The default is False.

    Returns
    -------
    bool
        True if the array is sorted and False otherwise.
    """

    # Return True if x has only one value
    
    x = np.asarray(x)
    
    if x.size <= 1:
        return True
    
    # Control if x is sorted
    
    diff = np.diff(x, axis = 0)
    
    if decreasing:
        if strict:
            return np.all(diff < 0)
        else:
            return np.all(diff <= 0)
    else:
        if strict:
            return np.all(diff > 0)
        else:
            return np.all(diff >= 0)

        
        

[docs]def is_subshape(shape1, shape2):
    """
    Indicates wether a shape is a subshape of another shape.
    
    Note
    ----
    Shape1 is a subshape of shape2 if its number of dimensions (length) is less
    than or equal to the number of dimensions of shape2 and if the size along
    each axis of shape1 is equal to 1 or the size of shape2 along the corresponding
    axis. Equivalently, shape1 is a subshape of shape2 if they can be broadcast
    together to a single shape and if the resulting shape is the same as shape2.
    Finally, note that shape1 is a subshape of shape2 is equivalent to shape2
    is a supershape of shape1.

    Parameters
    ----------
    shape1 : tuple of int
        The shape to control if it is a subshape.
    shape2 : tuple of int
        The shape against which `shape1` is tested as a subshape.

    Returns
    -------
    bool
        True if `shape1` is a subshape of `shape2` and False otherwise.
    """
    
    if len(shape1) > len(shape2):
        return False
    
    for a, b in zip(shape1[::-1], shape2[::-1]):
        if a != 1 and a != b:
            return False
        
    return True

    axis. Equivalently, shape1 is a supershape of shape2 if they can be broadcast
    together to a single shape and if the resulting shape is the same as shape1.
    Finally, note that shape1 is a supershape of shape2 is equivalent to shape2
    is a subshape of shape1.

    shape1 : tuple of int
        The shape to control if it is a supershape.
    shape2 : tuple of int
        The shape against which `shape1` is tested as a supershape.

    Returns
    -------
    bool
        True if `shape1` is a supershape of `shape2` and False otherwise.
    """
    
    return is_subshape(shape2, shape1)

    """
    Given a vector of non-decreasing breakpoints in table, find the lower bound of each element of x in table.
    The lower bound index of the value x in table is the index i satisfying table[i] <= x < table[i+1].

    table = np.asarray(table)

    # Return output
    
    if index:
        return indices
    else:
        return table[indices]  


    

[docs]def match(x, table, kind = "exact"):
    """
    Matches the elements of `x` in `table`.
    
    For each element of `x`, a matching index will be returned. If an element of
    `x` has a matching index `i`, it means that this element can be found at position
    `i` of `table`.

    Parameters
    ----------
    x : array-like
        The values to be matched.
    table : 1D array-like
        The values to be matched against. Duplicate values are not allowed.
    kind : {"exact", "lower", "upper"}, optional
        Matching type: exact match (exact), exact or next smallest (lower),
        exact or next largest (upper). In case of exact match, the values of `x`
        must be found in `table` (otherwise, an error will be raised).
        The default is "exact".

    Returns
    -------
    array of int
        Array of indices with the same shape as `x`.

    Raises
    ------
    ValueError
        If `kind` is not valid.
        If a value of `x` cannot be found in `table` in case of exact match.
    """
    
    # Convert to numpy array
    
    x = np.asarray(x)
    table = np.asarray(table).ravel()
    
    # Control if the values are in the table in case of exact match
    
    if kind == "exact" and np.any(np.isin(x, table, invert = True)):
        raise ValueError("exact match is not possible: some values are not found")
            
    # Indices sorting the table
    
    sorter = np.argsort(table)
    
    # Find the indices matching the sorted values of table
    
    if kind == "exact":
        indices = np.searchsorted(table, x, sorter = sorter)
    elif kind == "lower":
        indices = np.maximum(np.searchsorted(table, x, sorter = sorter, side = "right") - 1, 0)
    elif kind == "upper":
        indices = np.minimum(np.searchsorted(table, x, sorter = sorter), len(table) - 1)
    else:
        raise ValueError("'kind' must be one of the following: 'exact', 'lower' or 'upper'")
        
    # Return output
    
    return sorter[indices]

[docs]def normalize_axis(axis, ndim, negative = False, sort = False, allow_duplicate = False):
    """
    Normalize an axis in order to have a positive value such that 0 <= normalized_axis < ndim.

    Parameters
    ----------
    axis : int or iterable of int or None
        Input axis. A single axis can be provided through an integer and
        multiple axis can be provided through an iterable of integers.
    ndim : int
        Number of array dimensions.
    negative : bool, optional
        If True, the normalized axis will be negative such that -ndim <= normalized_axis < 0
        (otherwise, it will be positive such that 0 <= normalized_axis < ndim). The default is False.
    sort : bool, optional
        If True and if multiple axis are provided, the normalized axis will be sorted
        in ascending order. The default is False.
    allow_duplicate : bool, optional
        If False and if multiple axis are provided, disallow an axis from being specified twice.
        The default is False.

    Returns
    -------
    int or tuple of int or None
        Normalized axis. The output can be an integer, a tuple of integers or a None value
        following the type of the input axis (respectively, integer, iterable of integers or None).
         
    Raises
    ------
    AxisError
        If any axis provided is out of range (i.e. if we don't have the following
        inequality: -ndim <= axis < ndim).
    ValueError
        If an axis is repeated multiple times (unless allow_duplicate is set to True).

    """
    
    def new_axis(axis):
        if not (-ndim <= axis < ndim):
            raise np.AxisError(axis, ndim)
            
        if negative:
            return axis if axis < 0 else axis - ndim
        else:
            return axis if axis >= 0 else axis + ndim
    
    if axis is None: # axis is a None value
        return None
    elif isinstance(axis, int): # axis is an integer
        return new_axis(axis)
    else: # axis is an iterable of integers
        axis = [new_axis(i) for i in axis]
        
        if not allow_duplicate and len(set(axis)) != len(axis):
            raise ValueError("An axis has been repeated multiple times")
            
        if sort:
            axis.sort() 
            
        return tuple(axis)

    """

    Parameters
    ----------
    x : array

    values : array-like

    ------
    IndexError

    ValueError
        If `values` cannot be broadcast with the elements of `x` to replace.

    if isinstance(indices, np.ndarray) and indices.ndim > 1:
        indices = indices.ravel()
        
    # Keep the number of dimensions of x if indices is an integer
    
    if keepdims:
        try:
            indices = [int(indices)]
        except:
            pass
            
    # Replace elements from x along the axis with given values
    
    if axis is None:
        x.ravel()[indices] = values
    elif axis >= 0:
        exec("x[" + axis*":," + "indices,...] = values")
    elif axis < 0:
        exec("x[...,indices" + (-axis-1)*",:" + "] = values")

    """
    Reduce a shape along an axis.

    Parameters
    ----------
    shape : tuple of int
        The shape to reduce.
    axis : int or tuple of int or None, optional

        The default is False.

        Reduced shape.
    """

    elif isinstance(axis, int):
        axis = [axis]
                
    # Reduce the shape along the given axis
    
    shape = list(shape)
    new_size = 1 if keepdims else 0
            
    for s in axis:
        shape[s] = new_size
        
    if not keepdims:
        shape = [s for s in shape if s != 0]
        
    # Return output
    
    return tuple(shape)
