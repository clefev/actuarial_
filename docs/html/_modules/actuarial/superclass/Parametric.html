from abc import ABCMeta, abstractmethod



[docs]class Parametric(metaclass = ABCMeta):
    """    
    Superclass for building parametric models.
    
    A parametric model is described by a set of parameters.
    The parameters are given through a `Parameters` instance.
    Therefore, each parameter has a name, type and potential constraints.
    
    The normal distribution is an example of parametric model. The parameters
    are the mean :math:`\mu` and the standard deviation :math:`\sigma` of the
    distribution. Both parameters are scalar type. The mean has no constraints
    while the standard deviation must be strictly positive.
    """
        
    # Initialization
    

[docs]    def __init__(self, parameters):
        """
        Parameters
        ----------
        parameters : Parameters
            Parameters of the instance. The parameters must be initialized,
            respect their constraints and be able to broadcast together.

        Raises
        ------
        TypeError
            If `parameters` is not a `Parameters` instance.
        ValueError
            If parameters cannot be broadcast together.
            If parameters are not valid when using the `control` method.
        """
        
        from actuarial.parameter import Parameters
        
        # Control the parameters

        if not isinstance(parameters, Parameters):
            raise TypeError("'parameters' must be a 'Parameters' instance")
        elif not parameters.is_broadcastable():
            raise ValueError(f"parameters cannot be broadcast together: their respective shapes are {parameters.shapes}")
                       
        parameters.control()
        self._parameters = parameters

            
    # Abstract
    
    @property
    @abstractmethod
    def _parametrization(self):
        """
        Parameters: Abstract property giving the parametrization used by the
        class. The parametrization is given through a `Parameters` instance where
        the parameters are not initialized. The `parameters` attribute of an
        instance should be an initialized copy of this parametrization.
        """
        
        raise NotImplementedError()

    # Properties
    
    @property
    def parameters(self):
        """
        Parameters: Parameters of the instance.
        """
        
        return self._parameters    
    
    # Setter    
    
    @parameters.setter
    def parameters(self, value):
        self._parameters = value

[docs]    def broadcast(self, *args):
        """
        Mimic the broadcast of the parameters with input arguments.

        Parameters
        ----------
        *args : array-like or object with shape attribute
            Inputs to broadcast with the parameters. If no argument is provided,
            only the parameters are broadcast together.

        Returns
        -------
        np.broadcast
            Broadcast result.
        """
        
        return self.parameters.broadcast(*args)

        

[docs]    @classmethod
    def get_parametrization(cls):
        """
        Return a deep copy of the parametrization used by the class.
        A parameterization indicates the name and type of parameters used by an
        instance of this class. The parameters of this parametrization are not
        initialized meaning that their value is set to None. Therefore, the
        `parameters` attribute of an instance should be an initialized copy of
        this parametrization.

        Returns
        -------
        Parameters
            Parametrization used by the class.
        """
        
        return cls._parametrization.copy(deep = True)
