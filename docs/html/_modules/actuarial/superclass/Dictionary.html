[docs]class Dictionary(dict):
    """
    Superclass for building dictionary.
    
    A dictionary is a collection of key-component pairs.
    The key is represented by a string and the component by an object.
    
    This superclass is based on the base python dictionary (inherits from the
    `dict` instance) for which some enhancements have been added:
        
        * Possibility to accept only one specific type of component within the
          dictionary (e.g., a dictionary contatining only integers as components).
        * Implementation of an `add` method to add new components to the dictionary.
        * Implementation of a `broadcast` method.
        * Improvement of the `pop` method (now allows to pop several components at once).
        * Iteration over the components of the dictionary instead of the keys (see `__iter__`).
        * Possibility to get a single component of the dictionary by either its
          key or its index, and possibility to get multiple components by either
          a slice or list of keys (see `__getitem__`).
    """
    
    # Class attributes
    
    _component_type = None
    
    # Initialization
    

[docs]    def __init__(self, **kwargs):
        """
        Parameters
        ----------
        **kwargs
            Components of the dictionary.
            The keyword is the key of the component and the argument must have
            the correct type accoring to the `component_type` attribute.

        Raises
        ------
        TypeError
            If an argument has not the correct type accoring to the `component_type` attribute.
        """
        
        self._control_kwargs(**kwargs)
        super().__init__(self, **kwargs)

        
    # Properties
    
    @property
    def components(self):
        """
        list: List of all components.
        """
        
        return list(super().values())
    
    @property
    def names(self):
        """
        list of str: Name of each component (key).
        """
        
        return list(super().keys())
    
    @property
    def pairs(self):
        """
        list of tuple: List containing a tuple for each key-component pair.
        """
        
        return list(super().items())
    
    @property
    def component_type(self):
        """
        type or None: Type of components accepted by the instance.
        If None, any types are accepted.
        """
        
        return self._component_type

[docs]    def _control_keys(self, *keys):
        """
        Control whether the provided keys are defined in the instance.        
        If a key doesn't exist, a `KeyError` will be raised.
        """
        
        for key in keys:
            if key not in self:
                raise KeyError(key)

                

[docs]    def _control_kwargs(self, **kwargs):
        """
        Control whether the provided arguments have the correct type accoring to
        the `component_type` attribute.        
        If an argument has not the correct type, a `TypeError` will be raised.
        """
        
        if self.component_type is not None:
            for key, component in kwargs.items():
                if not isinstance(component, self.component_type):
                    raise TypeError(f"'{key}' must be '{self.component_type.__name__}' instance")

                    

[docs]    def add(self, **kwargs):
        """
        Add new components to the instance.

        Parameters
        ----------
        **kwargs
            New components to add.
            The keyword is the key of the component to add and the argument
            must have the correct type accoring to the `component_type` attribute.
            If the key already exists, the component will be replaced by the
            provided argument.
            
        Returns
        -------
        Dictionary
            Instance.
            
        Raises
        ------
        TypeError
            If an argument has not the correct type accoring to the `component_type` attribute.
        """
        
        self._control_kwargs(**kwargs)
        
        for key, component in kwargs.items():
            self[key] = component
            
        return self

[docs]    def copy(self, deep = False):
        """
        Return a copy of the instance.

        Parameters
        ----------
        deep : bool, optional
            If True, a deep copy of the instance is returned and if False, a
            shallow copy is returned. A shallow copy creates a new instance but
            doesn't create a copy of nested objects while a deep copy does.
            The default is False (shallow copy).

        Returns
        -------
        Dictionary
            Copy of the instance.
        """
        
        if deep:
            return deepcopy(self)
        else:
            return copy(self)

                

[docs]    def pop(self, keys = -1):
        """
        Remove the specified components from the instance.
        The removed components are returned by the method.

        Parameters
        ----------
        keys : str or int or slice or list of str, optional
            Keys of the components to return and remove from the instance.
            It can be a single component if a string or integer is provided,
            or multiple components if a slice or list of strings is provided.
            In the last case, a new instance containing these components will
            be returned.
            The default returns and removes the last component.

        Returns
        -------
        any
            Components removed from the instance.
            If a string or integer is provided, a single component is returned.
            If a slice or list of strings is provided, multiple components are
            returned through a new instance.

        Raises
        ------
        TypeError
            If `keys` is not a string, integer, slice or list.
        IndexError
            If the index is out of range when an integer is provided.
        KeyError
            If the key of a component is not found when a string or list of strings is provided.
        """
        
        pop = lambda key: dict.pop(self, key)
            
        if isinstance(keys, str):
            return pop(keys)
        elif isinstance(keys, int):
            return pop(self.names[keys])
        elif isinstance(keys, slice):
            return self.__class__(**{key: pop(key) for key in self.names[keys]})
        elif isinstance(keys, (list, tuple)):
            self._control_keys(*keys)
            return self.__class__(**{key: pop(key) for key in keys})
        else:
            raise TypeError(f"cannot pop items from a '{type(keys).__name__}' instance: should be a string, integer, slice or list of strings")

[docs]    def __getitem__(self, keys):
        """
        Return a single component if a string or integer is provided,
        and return a new instance if a slice or list of strings is provided.
        """
                
        get = lambda key: dict.__getitem__(self, key)

        if isinstance(keys, str):
            return get(keys)
        elif isinstance(keys, int):
            return get(self.names[keys])
        elif isinstance(keys, slice):
            return self.__class__(**{key: get(key) for key in self.names[keys]})
        elif isinstance(keys, (list, tuple)):
            return self.__class__(**{key: get(key) for key in keys})
        else:
            raise TypeError(f"cannot get items from a '{type(keys).__name__}' instance: should be a string, integer, slice or list of strings")
