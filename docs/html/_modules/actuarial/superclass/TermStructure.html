[docs]class TermStructure:
    r"""
    Superclass for building discrete term structure models.
    
    A discrete term structure model is described by a finite set of values
    associated with maturities.
    
    A (discrete) yield curve is an example of term structure model.
    It represents a relationship between interest rates and maturities.
    Mathematically, it can be described by a collection :math:`\lbrace (\tau_i, r_i) \rbrace_{i=1,...,n}`
    where :math:`\tau_i`'s are the maturities, :math:`r_i`'s are the associated
    interest rates and :math:`n` is the finite number of observed pairs.
    """
    
    # Initialization
    

[docs]    def __init__(self, maturities, values):
        """        
        Parameters
        ----------
        maturities : array-like
            Maturities of the term structure.
            Must be along the first axis.
        values : array-like
            Values associated with the maturities of the term structure.
            Its size along the term structure axis must be equal to the number
            of maturities.

        Raises
        ------
        ValueError
            If `maturities` is not along the first axis or contain negative values.
            If `values` has not the same size as `maturities` along the term structure axis.
        """
        
        # Control if maturities is along the first axis and contains positive values
        
        maturities = np.asarray(maturities)
        
        if not util.is_along_axis(maturities):
            raise ValueError(f"'maturities' must be along axis 0: its shape is {maturities.shape}")
        elif np.any(maturities < 0):
            raise ValueError("'maturities' must be positive")
            
        self._maturities = maturities
            
        # Control if values has the same size as maturities along the axis
        
        values = np.asarray(values)

        if not util.is_along_axis(values, axis = self.axis, size = maturities.size):
            raise ValueError(f"'values' must have the same size as 'maturities' along the axis {self.axis}: its shape is {values.shape}")
            
        self._values = values

    @property
    def axis(self):
        """
        int: Axis along which the maturities are (given as a negative integer).
        """
        
        return -self.maturities.ndim
    
    @property
    def maturities(self):
        """
        array: Maturties.
        """
        
        return self._maturities

    @property
    def values(self):
        """
        array: Values associated with the maturities.
        """
        
        return self._values

[docs]    def append(self, maturity, value):
        """
        Append in-place a maturity and its associated value.
        
        Note that if the maturity to append already exists, the appending
        does not take place. In addition, the maturities are not sorted in ascending
        order after an appending.

        Parameters
        ----------
        maturity : scalar
            Maturity to append.
        value : scalar
            Value to append.

        Returns
        -------
        TermStructure
            Instance.
        """
        
        if maturity not in self:
            self.maturities, self.values = util.concatenate(self.maturities, maturity, axis = self.axis), util.concatenate(self.values, value, axis = self.axis)
            
        return self

        """
        Prepend in-place a maturity and its associated value.

        if maturity not in self:

[docs]    def sort(self, decreasing = False):
        """
        Sort in-place the maturities and values by increasing maturities.
        
        Parameters
        ----------
        decreasing : bool
            If True, the maturities and values will be sorted by decreasing maturities.
            The default is False.

        Returns
        -------
        TermStructure
            Instance.
        """
        
        indices = np.argsort(self.maturities.ravel())
        indices = indices[::-1] if decreasing else indices        
        self.maturities, self.values = util.take(self.maturities, indices, axis = self.axis), util.take(self.values, indices, axis = self.axis)
        return self
