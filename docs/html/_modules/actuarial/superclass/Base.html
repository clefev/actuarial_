[docs]    def _dict_to_print(self):
        """
        Return a dictionary with the attributes and properties to print when
        calling the `print` function on the object.
        
        The dictionary will contain all attributes and properties mentionned in
        the `_print` attribute of the class. Therefore, the `_print` attribute
        must be a tuple containing the name of all attributes and properties of
        the instance to print. If a name doesn't exist as an attribute or
        property, it will be skipped.
        
        Returns
        -------
        dict
            Dictionary with the attributes and properties of the instance to print.
        """
        
        keys = self._print if self._print else [key.strip("_") for key in self.__dict__] # get the list of properties to print
        keys = [key for key in keys if hasattr(self, key) and not isinstance(getattr(self, key), type(getattr(self, "__repr__")))] # check if the selected properties exist and are not methods
        return {key: getattr(self, key) for key in keys}

    

[docs]    def copy(self, deep = False):
        """
        Return a copy of the instance.

        Parameters
        ----------
        deep : bool, optional
            If True, a deep copy of the instance is returned and if False, a
            shallow copy is returned. A shallow copy creates a new instance but
            doesn't create a copy of nested objects while a deep copy does.
            The default is False (shallow copy).

        Returns
        -------
        Base
            Copy of the instance.
        """
        
        if deep:
            return deepcopy(self)
        else:
            return copy(self)

        

[docs]    def delete_attributes(self, *args):
        """
        Delete in-place the attributes from the instance.

        Parameters
        ----------
        *args : str
            Name of attributes to delete.
            If the name of an attribute does not exist within the instance,
            it will be skipped.
            
        Returns
        -------
        Base
            Instance.
        """
        
        for arg in args:
            try:
                delattr(self, arg)
            except AttributeError:
                pass
            
        return self

        

[docs]    def set_attributes(self, **kwargs):
        """
        Set in-place the attributes of the instance.

        Parameters
        ----------
        **kwargs : any
            Attributes to set. The name of the argument is the name of the
            attribute and the value of the argument is the corresponding value
            of this attribute. If the name doesn't exist, a new attribute will
            be created and if it already exists, its value will be replaced by
            the provided one.
            
        Returns
        -------
        Base
            Instance.
        """
        
        for key, value in kwargs.items():
            setattr(self, key, value)
            
        return self

[docs]    def __repr__(self):
        """
        Object representation in string format. The representation includes the
        name of the module where the object comes from, the name of the class
        and a list of attributes and properties given by the `_dict_to_print` method.
        """
        
        return self.__module__ + "." + type(self).__name__ + "\n" + util.dict_to_string(self._dict_to_print(), bullet = "\t")
