[docs]class Broadcast(metaclass = ABCMeta):
    """
    Superclass implementing broadcasting method to mimic how input aruguments
    are broadcast with the instance.
    The superclass also confers properties such as the shape, number of dimensions
    and size of the instance.
    """

            raise ValueError("instance cannot be broadcast to a single shape")

[docs]    @abstractmethod
    def broadcast(self, *args):
        """
        Abstract method to mimic the broadcast of the instance with input arguments.

        Parameters
        ----------
        *args : array-like or object with shape attribute
            Inputs to broadcast with the instance. It can be array-like or any
            object having a `shape` attribute.
            
        Returns
        -------
        np.broadcast
            Broadcast result.
        """
        
        return util.broadcast(*args)

[docs]    @staticmethod
    def _asarray(x):
        """
        Convert the input argument to a numpy array. 
        
        The conversion occurs only if the input argument is not a Broadcast instance
        and is not None.
        Otherwise, the original argument is returned as is.
        In other words, this method is equivalent to the `asarray` function of
        the numpy package except that Broadcast instance and None value are not
        converted.
        
        Parameters
        ----------
        x : array-like or Broadcast or None
            Input to convert to a numpy array.
            
        Returns
        -------
        array or Broadcast or None
            Input converted to a numpy array. If the input is a Broadcast instance
            or is None, the conversion doesn't occur and the input is returned as is.
        """
        
        if x is None or isinstance(x, Broadcast):
            return x
        else:
            return np.asarray(x) 

    

[docs]    def is_broadcastable(self, *args):
        """
        Indicate whether the input arguments can be broadcast with the instance
        to a single shape using the implemented `broadcast` method of the class.

        Parameters
        ----------
        *args : array-like or object with shape attribute
            Inputs to broadcast with the instance. It can be array-like or any
            object having a `shape` attribute.

        Returns
        -------
        bool
            True if the inputs can be broadcast with the implemented `broadcast`
            method of the class and False otherwise.
        """
        
        try:
            self.broadcast(*args)
            return True
        except ValueError:
            return False
