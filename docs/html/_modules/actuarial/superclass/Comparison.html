    def __comparison__(self, value, operator):
        """
        Abstract method describing how comparison operators (`==`, `!!=`, `<`, `>`, `<=` and `>=`)
        should behave on the object.
        
        The `operator` argument is a function representing a generic comparison
        operator.
        
        For example, the derived class could implement the following `__comparison__`
        method:
            
        .. code-block:: python
        
            def __comparison__(self, value, operator):
                return operator(self.attr, value)
            
        By implementing this single abstract method, all comparison operators mentioned
        above will be available (e.g., `object == value` will return the result of
        `object.attr == value`, `object != value` will return the result of
        `object.attr != value`, ...).
        """
        
        raise NotImplementedError()

    class Example1(Comparison):
        """
        Create a class `Example1` inheriting from the `Comparison` superclass and
        having an `amount` attribute representing a number.
        The application of comparison operators on the instance (`==`, `!=`, `<`, `>`, `<=` and `>=`)

        """
        
        def __init__(self, amount: float):
            self.amount = float(amount)
            
        def __comparison__(self, value, operator):
