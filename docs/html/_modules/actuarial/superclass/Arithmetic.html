[docs]    @abstractmethod
    def __arithmetic__(self, value, operator):
        """
        Abstract method describing how arithmetic operators (`+`, `-`, `*`, `/` and `**`)
        should behave on the object.
        
        The `operator` argument is a function representing a generic arithmetic
        operator.
        
        For example, the derived class could implement the following `__arithmetic__`
        method:
            
        .. code-block:: python
        
            def __arithmetic__(self, value, operator):
                return operator(self.attr, value)
            
        By implementing this single abstract method, all arithmetic operators mentioned
        above will be available (e.g., `object + value` will return the result of
        `object.attr + value`, `object - value` will return the result of
        `object.attr - value`, ...).
        """
        
        raise NotImplementedError()

    # Example 1
    
    class Example1(Arithmetic):
        """
        Create a class `Example1` inheriting from the `Arithmetic` superclass and
        having an `amount` attribute representing a number.
        The application of arithmetic operators on the instance (`+`, `-`, `*`, `/` and `**`)
        should return the result of this arithmetic operator applied on the `amount`
        attribute of the instance.
        """
        
        def __init__(self, amount: float):
            self.amount = float(amount)
            
        def __arithmetic__(self, value, operator):
            return operator(self.amount, value) # return the result of the arithmetic operator applied on the 'amount' attribute of the instance
        
    example1 = Example1(amount = 10)
    print(example1.amount)
    print(example1+2)
    print(example1-2)
    print(example1*2)
    print(example1/2)
    print(example1**2)
