[docs]class Countermonotonic(Copula):
    """
    The countermonotonic copula is a special bivariate copula characterising a
    perfect negative dependence between two components.
    
    The marginals :math:`U_1` and :math:`U_2` of the countermonotonic copula
    are described by a same uniform random variable :math:`U` as following:
        
    .. math::
        (U_1, U_2) = (U, 1-U)
        
    where the equality stands for the equality in distribution.
    It represents the extreme case where the dependency is as strongly negative
    as possible (i.e. achieves the Fréchet–Hoeffding lower bound).
    
    Therefore, a random vector :math:`(X_1, X_2)` is countermonotonic if it can
    be represented as
    
    .. math::
        (X_1, X_2) = (F_{X_1}^{-1}(U), F_{X_2}^{-1}(1-U))
        
    where the components :math:`X_1` and :math:`X_2` are respectively an increasing
    and decreasing function of a same random variable :math:`U`.        
    In this way, the two components move in the opposite direction which describes
    a perfect negative dependence.
    """

[docs]    @Copula.Decorators.cdf
    def cdf(self, x, keepdims = False):
        """
        Evaluate the cumulative distribution function.
        
        Note
        ----
        The cdf of the countermonotonic copula is given by
        
        .. math::
            C(u_1, u_2) = \max(u_1 + u_2 - 1, 0)
            
        where :math:`(u_1, u_2) \in [0, 1]^2`.

        Parameters
        ----------
        u : array-like
            Values (between 0 and 1) for which the cdf should be evaluated.
            The size along the last axis must be equal to the dimensionality
            of the copula.
        keepdims : bool, optional
            If True, the last axis which is reduced is left in the result as dimension
            with size one (otherwise, it will be removed). With this option, the
            result will broadcast correctly against the input array.
            The default is False.
        
        Returns
        -------
        scalar or array
            Evaluated cdf. Array with a float dtype and the same shape as `u`
            except that the last axis has a size 1 if `keepdims` is True or is
            removed if it is False.
            
        Raises
        ------
        ValueError        
            If the size of `u` along its last axis is not equal to the dimensionality
            of the copula or if `u` contains values that are not between 0 and 1.
        """
        
        return np.maximum(np.sum(x, axis = -1, keepdims = keepdims) - 1., 0.)

[docs]    @Copula.Decorators.random
    def random(self, size = None, seed = -1):
        """
        Draw *2*-dimensional random samples of joint uniform distributions.
        
        Note
        ----
        The following methodology is used to generate a *2*-dimensional random
        sample from the countermonotonic copula:
            
            1. Draw a single random sample :math:`u \in (0,1)` from an uniform distribution.
            2. Return :math:`(u_1, u_2) = (u, 1-u)`.

        Parameters
        ----------
        size : int or tuple of int or None, optional
            Number of samples to be drawn. Because each sample is *2*-dimensional,
            the output shape will have an additional axis with size *2* in last
            position. If None, a single *2*-dimensional sample is returned.
            The default is None.
        seed : int or array-like of int or SeedSequence or Generator or None, optional
            Seed to use for the random number generator. The default uses the
            random number generator associated with the instance.

        Returns
        -------
        array
            Random samples. Array of values between 0 and 1 with the same shape
            as `size` except that an additional axis with size *2* has been added
            in last position.
        """
        
        rng = self.get_rng(seed)
        u = rng.random(size + (1,))
        return np.concatenate((u, 1-u), axis = -1)
