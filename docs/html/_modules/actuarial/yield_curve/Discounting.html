[docs]class Discounting(Arithmetic, Base, Broadcast, Comparison):
    """
    Class to manage discount values between two maturities.
    
    A discount value is a value that can be used to compute discount factors
    between two maturities, namely `maturities1` and `maturities2`.
    There are different types of discount values such as discount factors,
    capitalization factors, compound rates, continuous rates and simple rates.    
    The type of discount values is provided by the `kind` attribute.
    A discount value can be easily converted to one type to another using the
    `convert` method.   
    """
    
    # Class attributes    

    _print = ("maturities1", "maturities2", "values", "kind")
    kinds = ("capitalization_factor", "compound_rate", "continuous_rate", "discount_factor", "simple_rate")

[docs]    def __init__(self, maturities1, maturities2, values, kind):
        """
        Parameters
        ----------
        maturities1 : array-like
            Starting maturities.
        maturities2 : array-like
            Ending maturities.
        values : array-like
            Discount values between the starting and ending maturities.
        kind : {"capitalization_factor", "compound_rate", "continuous_rate", "discount_factor", "simple_rate"}
            Type of discount values.
        
        Raises
        ------
        ValueError
            If `kind` is not valid.
            If `maturities1`, `maturities2` and `values` cannot be broadcast together.
        """
        
        # Control the kind
                
        if kind not in self.kinds:
            raise ValueError(f"'kind' must be one of the following: {', '.join(self.kinds)}")
            
        # Control if maturities and values can be broadcast together
        
        maturities1 = np.asarray(maturities1)
        maturities2 = np.asarray(maturities2)
        values = np.asarray(values)
        
        if not util.is_broadcastable(maturities1, maturities2, values):
            raise ValueError(f"'maturities1', 'maturities2' and 'values' cannot be broadcast together: their respective shapes are {maturities1.shape}, {maturities2.shape} and {values.shape}")

        # Define the instance attributes
        
        self._maturities1 = maturities1
        self._maturities2 = maturities2
        self._values = values
        self._kind = kind

[docs]    def convert(self, kind = "discount_factor"):
        """
        Convert in-place the discount values to a new type.

        Parameters
        ----------
        kind : {"capitalization_factor", "compound_rate", "continuous_rate", "discount_factor", "simple_rate"}, optional
            The discount values will be converted to this type.
            The default converts to discount factors.

        Returns
        -------
        Discounting
            Instance.
            
        Raises
        ------
        ValueError
            If `kind` is not valid.
        """
        
        # Return the instance if kind is already correct
        
        if self.kind == kind:
            return self
        
        # Control the kind
                
        if kind not in self.kinds:
            raise ValueError(f"'kind' must be one of the following: {', '.join(self.kinds)}")
                    
        # Convert the discount values
        
        with np.errstate(divide = "ignore", invalid = "ignore"): # ignore numpy warnings such as division by 0 (handled by np.where if any)
            
            if self.kind == "capitalization_factor":
        
                if kind == "compound_rate":
                    self.values = self.values**(1/self.maturities) - 1
                elif kind == "continuous_rate":
                    self.values = np.where(self.values == 1, 0, np.log(self.values)/self.maturities)
                elif kind == "discount_factor":
                    self.values = 1/self.values
                elif kind == "simple_rate":
                    self.values = np.where(self.values == 1, 0, (self.values - 1)/self.maturities)
        
            elif self.kind == "compound_rate":
                
                if kind == "capitalization_factor":
                    self.values = (1 + self.values)**self.maturities
                elif kind == "continuous_rate":
                    self.values = np.log(1 + self.values)
                elif kind == "discount_factor":
                    self.values = (1 + self.values)**(-self.maturities)
                elif kind == "simple_rate":
                    self.values = np.where(self.maturities == 0, self.values, ((1 + self.values)**self.maturities - 1)/self.maturities)
                
            elif self.kind == "continuous_rate":
                
                if kind == "capitalization_factor":
                    self.values = np.exp(self.values*self.maturities)
                elif kind == "compound_rate":
                    self.values = np.exp(self.values) - 1
                elif kind == "discount_factor":
                    self.values = np.exp(-self.values*self.maturities)
                elif kind == "simple_rate":
                    self.values = np.where(self.maturities == 0, self.values, (np.exp(self.values*self.maturities) - 1)/self.maturities)
                
            elif self.kind == "discount_factor":
                
                if kind == "capitalization_factor":
                    self.values = 1/self.values
                elif kind == "compound_rate":
                    self.values = self.values**(-1/self.maturities) - 1
                elif kind == "continuous_rate":
                    self.values = np.where(self.values == 1, 0, -np.log(self.values)/self.maturities)
                elif kind == "simple_rate":
                    self.values = np.where(self.values == 1, 0, (1/self.values - 1)/self.maturities)
                
            elif self.kind == "simple_rate":
                
                if kind == "capitalization_factor":
                    self.values = 1 + self.values*self.maturities
                elif kind == "compound_rate":
                    self.values = np.where(self.maturities == 0, self.values, (1 + self.values*self.maturities)**(1/self.maturities) - 1)
                elif kind == "continuous_rate":
                    self.values = np.where(self.maturities == 0, self.values, np.log(1 + self.values*self.maturities)/self.maturities)
                elif kind == "discount_factor":
                    self.values = 1/(1 + self.values*self.maturities)
                                
        # Update the type of discount values
        
        self.kind = kind
                
        # Return output
        
        return self
