[docs]class Mixture(Mixture_, YieldCurve):
    r"""
    A mixture yield curve is a yield curve derived from a collection of other
    yield curves.
    The other yield curves are called the components of the mixture yield curve.
    For each component, a weight can be associated.
    
    The value of a mixture yield curve is expressed as a linear combination
    of its components:
        
    .. math::
        r(\tau) = \sum_{k=1}^K w_k r_k(\tau)
        
    where :math:`r_k(\tau)` is the value of the k-th yield curve component
    at maturity :math:`\tau`, :math:`w_k` the associated weight and :math:`K`
    the number of mixture components.
    """
    
    # Class attributes
    
    _print = ("components", "weights", "kind")
    kinds = ("compound_rate", "continuous_rate", "simple_rate")

        """
        
        Mixture_.__init__(self, components, weights, component_type = YieldCurve)
        YieldCurve.__init__(self, kind)

    def evaluate(self, maturities, split = False):
        r"""
        Evaluate the value of the yield curve for the given maturities.
        
        Note
        ----
        The value of a mixture yield curve at maturity :math:`\tau` is given by
        
        .. math::
            r(\tau) = \sum_{k=1}^K w_k r_k(\tau)
            
        where :math:`r_k(\tau)` is the value of the k-th yield curve component
        at maturity :math:`\tau` and :math:`w_k` the associated weight.
        
        Note that the values of each component of the mixture yield curve are
        automatically converted to the same type as the mixture yield curve
        (`kind` attribute of the instance).

        Parameters
        ----------
        maturities : array-like
            Maturities for which the yield curve should be evaluated.
        split : bool, optional
            If True, the individual weighted components of the mixture yield curve
            are returned separately. Otherwise, they are aggregated.
            The default is False.

        Returns
        -------
        scalar or array or list
            Value of the yield curve for the given maturities.
            If `split` is True, a list with the weighted values of each individual
            component is returned.

        Raises
        ------
        ValueError
            If `maturities` cannot be broadcast with the instance.
        """
        
        values = [weight*component.to_discounting(maturities).convert(self.kind).values for component, weight in self]
        
        if split:
            return values
        else:
            return sum(values)
