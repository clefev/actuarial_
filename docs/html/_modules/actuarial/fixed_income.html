    def aggregate(self, maturities, strict = False):
        """
        Aggregate in-place cash-flows by maturity buckets. The value of all
        cash-flows belonging to a same maturity bucket will be aggregated in 
        one single cash-flow amount. If no cash flow belongs to a maturity bucket,
        the cash-flow amount will be set to 0 for this bucket.

        Parameters
        ----------
        maturities : 1D sorted array-like
            Sorted maturities defining the maturity buckets as follows:
                * Bucket 0: [0, maturities[0]]
                * Bucket 1: (maturities[0], maturities[1]]
                * Bucket 2: (maturities[1], maturities[2]]
                * ...
                * Bucket i: (maturities[i-1], maturities[i]]
        strict : bool, optional
            If True, the inequalities of buckets will change so that bucket i
            becomes [maturities[i-1], maturities[i]) i.e. left closed and right
            open interval. The default is False.

        Returns
        -------
        Instance
        """
        
        # Input maturities must be a 1D array
        
        maturities = np.asarray(maturities).ravel()
        
        # Append the instance final maturity to the input maturities
        
        maturity_max = np.max(self.maturities)
        
        if maturities[-1] < maturity_max:
            maturities = np.append(maturities, maturity_max)
            
        # Map the instance maturities to the maturity buckets
            
        indices = util.upper_bound(self.maturities.ravel(), maturities, index = True, strict = strict) # 1D array
        maturity_buckets = pd.Categorical(maturities[indices], maturities, ordered = True)
        
        # Aggregate the cash-flows by maturity buckets
        
        self.maturities, self.values = util.aggregate_by(self.values, maturity_buckets, axis = self.axis, sort = True)
        
        # Return output
        
        return self
    
    def cumulate(self):
        """
        Cumulate the cash-flows over the maturities.
        
        Note
        ----
        If the cash-flows are already cumulated ('cumulative' attribute set to True),
        then the instance is returned as is. 

        Returns
        -------
        Instance
            Instance with cumulated cash-flow values over the maturities.

        """
        
        if not self.cumulative:
            self.values = np.cumsum(self.values, axis = self.axis)
            self.cumulative = True
            
        return self
    
    def decumulate(self):
        """
        Decumulate the cash-flows over the maturities.
        
        Note
        ----
        If the cash-flows are already decumulated ('cumulative' attribute set to False),
        then the instance is returned as is. 

        Returns
        -------
        Instance
            Instance with decumulated cash-flow values over the maturities.

        """
        
        if self.cumulative:
            self.values = np.diff(self.values, axis = self.axis, prepend = 0)
            self.cumulative = False
            
        return self

        Returns
        -------

        """
        
        return self.values.sum(axis = self.axis, keepdims = keepdims)

    

[docs]    def weights(self, cumulative = False):
        r"""
        Compute the relative weight (or percentage) of each cash-flow.
        
        Note
        ----
        The weight of a cash-flow corresponds to its relative size compared
        to the total sum of cash-flows over the maturities:
            
        .. math::
            w_i = \frac{CF_i}{CF_\bullet} = \frac{CF_i}{\sum_{j=1}^n CF_j}.
            
        Note that by construction we have
        
        .. math::
            \sum_{i=1}^n w_i = 1.

        Parameters
        ----------
        cumulative : bool, optional
            If True, the weights are cumulated to reach 100% at the final maturity.
            The default is False.

        Returns
        -------
        array
            Weights.
            Array with a float dtype and the same shape as the `values` attribute.
        """
        
        if not cumulative:
            return (self.values / self.sum(keepdims = True)).astype(float)
        else:
            cumsum = np.cumsum(self.values, axis = self.axis)
            return (cumsum / util.take(cumsum, -1, axis= self.axis, keepdims = True)).astype(float)

[docs]    def to_discounting(self, discount_factors):
        """
        Return a `Discounting` instance containing the discount factors evaluated
        at each cash-flow maturity.       
        The resulting discount factors can therefore be used to discount the
        cash-flows of the instance.

        Parameters
        ----------
        discount_factors : array-like or Discounting or YieldCurve
            Discount factors to use when discounting the cash-flows.
            It can be an array, a Discounting instance or an object able to
            evaluate the discount factors.
            If an array or a Discounting instance is provided, it is assumed to
            contain the value of the discount factors for each cash-flow maturity.

        Returns
        -------
        Discounting
            Discounting instance containing discount factors evaluated at each
            cash-flow maturity.

        Raises
        ------
        ValueError
            If the resulting array of discount factors cannot be broadcast with
            the instance or is not a supershape of the `maturities` attribute.
        """
        
        # Get an array with the discount factors
        
        if isinstance(discount_factors, Discounting): # Discounting instance
            discount_factors = discount_factors.copy().convert("discount_factor").values
        elif hasattr(discount_factors, "discount_factor"): # instance able to evaluate the discount factors based on provided maturites
            discount_factors = discount_factors.discount_factor(self.maturities)
        else: # discount factors directly given as an array
            discount_factors = np.asarray(discount_factors)
            
        # Control if the discount factors can be broadcast with the instance and is a supershape of the maturities
        
        if not self.is_broadcastable(discount_factors):
            raise ValueError("'discount_factors' cannot be broadcast with the instance")
        elif not util.is_supershape(discount_factors.shape, self.maturities.shape):
            raise ValueError("'discount_factors' must be a supershape of the 'maturities' attribute")
            
        # Return output

        return Discounting(0, self.maturities, discount_factors, "discount_factor")

            

[docs]    def discount(self, discount_factors):
        r"""
        Transform in-place the cash-flows by applying discount factors to them.
        
        Note
        ----
        The new transformed cash-flows are given by
        
        .. math::
            CF_i^{new} = CF_i P(\tau_i)
            
        where :math:`P(\tau)` is the discount factor for the maturity :math:`\tau`.

        Parameters
        ----------
        discount_factors : array-like or Discounting or YieldCurve
            Discount factors to use when discounting the cash-flows.
            It can be an array, a Discounting instance or an object able to
            evaluate the discount factors.
            If an array or a Discounting instance is provided, it is assumed to
            contain the value of the discount factors for each cash-flow maturity.
            
        Returns
        -------
        TYPE
            Instance.
        """
        
        cash_flows, discount_factors = self.values, self.to_discounting(discount_factors).values
        self.values = cash_flows * discount_factors
        return self

        Compute the present value of cash-flows.
        
        Note
        ----
        The present value of cash-flows is given by
        
        .. math::
            PV = \sum_{i=1}^n CF_i P(\tau_i)
            
        where :math:`P(\tau)` is the discount factor for the maturity :math:`\tau`.

        Parameters
        ----------
        discount_factors : array-like or Discounting or YieldCurve, optional
            Discount factors to use when discounting the cash-flows.

            contain the value of the discount factors for each cash-flow maturity.
        keepdims : bool, optional
            If True, the axis which is reduced is left in the result as dimension
            with size one (otherwise, it will be removed). With this option, the
            result will broadcast correctly against the initial cash-flows.
            The default is False.
            
        Returns
        -------
        scalar or array
            Present value.

        cash_flows, discount_factors = self.values, self.to_discounting(discount_factors).values
        return np.sum(cash_flows * discount_factors, axis = self.axis, keepdims = keepdims)

    

[docs]    def duration(self, discount_factors, kind = "continuous_rate", dollar = False, keepdims = False):
        r"""
        Compute the modified duration of cash-flows.
        
        Note
        ----
        The **modified duration** is a price sensitivity measure. It provides
        the relative change in the present value of cash-flows following a
        parallel shift of the term structure of interest rates.
        It is mathematically defined as
        
        .. math::
            duration = -\frac{1}{PV} \frac{\partial PV}{\partial }
            
        where :math:`PV` is the current present value (price) of cash-flows.
        
        Depending on the type of interest rates and assuming fixed cash-flows,
        the modified duration is calculated as follows:
        
            * **Continuous interest rates:**
            
                .. math:: \frac{1}{PV} \sum_{i=1}^n \tau_i CF_i P(\tau_i)
                
            * **Compound interest rates:**
            
                .. math:: \frac{1}{PV} \sum_{i=1}^n \tau_i CF_i P(\tau_i)^{1 + \frac{1}{\tau_i}}
                
            * **Simple interest rates:**
            
                .. math:: \frac{1}{PV} \sum_{i=1}^n \tau_i CF_i P(\tau_i)^2
        
        Note that the duration is measured in unit of time (the same unit as the
        maturities of cash-flows, usually in years).
        In case of continuous interest rates, the modified duration
        corresponds to the so called **Macaulay duration**.
        
        The **dollar (modified) duration** is equal to the (modified) duration
        multiplied by the current present value of cash-flows (i.e. the same
        formula as above except that we don't divide by the present value).

        Parameters
        ----------
        discount_factors : array-like or Discounting or YieldCurve, optional
            Discount factors to use when discounting the cash-flows.
            It can be an array, a Discounting instance or an object able to
            evaluate the discount factors.
            If an array or a Discounting instance is provided, it is assumed to
            contain the value of the discount factors for each cash-flow maturity.
        kind : {"compound_rate", "continuous_rate", "simple_rate"}, optional
            Type of interest rates. The default assumes continuous rates.
        dollar : bool, optional
            If True, compute the dollar duration. The default is False.
        keepdims : bool, optional
            If True, the axis which is reduced is left in the result as dimension
            with size one (otherwise, it will be removed). With this option, the
            result will broadcast correctly against the initial cash-flows.
            The default is False.

        Returns
        -------
        scalar or array
            Modified duration.
        """
                
        maturities, cash_flows, discount_factors = self.maturities, self.values, self.to_discounting(discount_factors).values
        
        # Compute the modified dollar duration
        
        if kind == "continuous_rate":
            duration = maturities * cash_flows * discount_factors
        elif kind == "compound_rate":
            duration = maturities * cash_flows * discount_factors**(1 + np.divide(1, maturities, where = (maturities != 0)))
        elif kind == "simple_rate":
            duration = maturities * cash_flows * discount_factors**2
        else:
            raise ValueError("'kind' must be one of the following: compound_rate, continuous_rate, simple_rate")
            
        # Normalize by the present value
        
        if not dollar:
            duration = duration / self.present_value(discount_factors, keepdims = True)
            
        # Return output
        
        return np.sum(duration, axis = self.axis, keepdims = keepdims)

        r"""
        Compute the modified convexity of cash-flows.
        
        Note
        ----
        Depending on the type of interest rates and assuming fixed cash-flows,
        the modified convexity is calculated as follows:
        
            * **Continuous interest rates:**
            
                .. math:: \frac{1}{PV} \sum_{i=1}^n \tau_i^2 CF_i P(\tau_i)
                
            * **Compound interest rates:**
            
                .. math:: \frac{1}{PV} \sum_{i=1}^n (\tau_i^2 + \tau_i) CF_i P(\tau_i)^{1 + \frac{2}{\tau_i}}
                
            * **Simple interest rates:**
            
                .. math:: \frac{2}{PV} \sum_{i=1}^n \tau_i^2 CF_i P(\tau_i)^3
        
        The **dollar (modified) convexity** is equal to the (modified) convexity
        multiplied by the current present value of cash-flows (i.e. the same
        formula as above except that we don't divide by the present value).

        Parameters
        ----------
        discount_factors : array-like or Discounting or YieldCurve, optional
            Discount factors to use when discounting the cash-flows.
            It can be an array, a Discounting instance or an object able to
            evaluate the discount factors.
            If an array or a Discounting instance is provided, it is assumed to
            contain the value of the discount factors for each cash-flow maturity.
        kind : {"compound_rate", "continuous_rate", "simple_rate"}, optional
            Type of interest rates. The default assumes continuous rates.
        dollar : bool, optional
            If True, compute the dollar convexity. The default is False.
        keepdims : bool, optional
            If True, the axis which is reduced is left in the result as dimension
            with size one (otherwise, it will be removed). With this option, the
            result will broadcast correctly against the initial cash-flows.
            The default is False.

        Returns
        -------
        scalar or array
            Modified convexity.
        """
        
        maturities, cash_flows, discount_factors = self.maturities, self.values, self.to_discounting(discount_factors).values
        
        # Compute the modified dollar convexity
        
        if kind == "continuous_rate":
            convexity = maturities**2 * cash_flows * discount_factors
        elif kind == "compound_rate":
            convexity = (maturities**2 + maturities) * cash_flows * discount_factors**(1 + np.divide(2, maturities, where = (maturities != 0)))
        elif kind == "simple_rate":
            convexity = 2 * maturities**2 * cash_flows * discount_factors**3
        else:
            raise ValueError("'kind' must be one of the following: compound_rate, continuous_rate, simple_rate")
            
        # Normalize by the present value
        
        if not dollar:
            convexity = convexity / self.present_value(discount_factors, keepdims = True)
            
        # Return output
        
        return np.sum(convexity, axis = self.axis, keepdims = keepdims)

        r"""
        Compute the modified asymetry of cash-flows.
        
        Note
        ----
        Depending on the type of interest rates and assuming fixed cash-flows,
        the modified asymetry is calculated as follows:
        
            * **Continuous interest rates:**
            
                .. math:: \frac{1}{PV} \sum_{i=1}^n \tau_i^3 CF_i P(\tau_i)
                
            * **Compound interest rates:**
            
                .. math:: \frac{1}{PV} \sum_{i=1}^n (\tau_i^3 + 3\tau_i^2 + 2\tau_i) CF_i P(\tau_i)^{1 + \frac{3}{\tau_i}}
                
            * **Simple interest rates:**
            
                .. math:: \frac{6}{PV} \sum_{i=1}^n \tau_i^3 CF_i P(\tau_i)^4
        
        The **dollar (modified) asymetry** is equal to the (modified) asymetry
        multiplied by the current present value of cash-flows (i.e. the same
        formula as above except that we don't divide by the present value).

        Parameters
        ----------
        discount_factors : array-like or Discounting or YieldCurve, optional
            Discount factors to use when discounting the cash-flows.
            It can be an array, a Discounting instance or an object able to
            evaluate the discount factors.
            If an array or a Discounting instance is provided, it is assumed to
            contain the value of the discount factors for each cash-flow maturity.
        kind : {"compound_rate", "continuous_rate", "simple_rate"}, optional
            Type of interest rates. The default assumes continuous rates.
        dollar : bool, optional
            If True, compute the dollar asymetry. The default is False.
        keepdims : bool, optional
            If True, the axis which is reduced is left in the result as dimension
            with size one (otherwise, it will be removed). With this option, the
            result will broadcast correctly against the initial cash-flows.
            The default is False.

        Returns
        -------
        scalar or array
            Modified asymetry.
        """
        
        maturities, cash_flows, discount_factors = self.maturities, self.values, self.to_discounting(discount_factors).values
        
        # Compute the modified dollar asymetry
        
        if kind == "continuous_rate":
            asymetry = maturities**3 * cash_flows * discount_factors
        elif kind == "compound_rate":
            asymetry = (maturities**3 + 3*maturities**2 + 2*maturities) * cash_flows * discount_factors**(1 + np.divide(3, maturities, where = (maturities != 0)))
        elif kind == "simple_rate":
            asymetry = 6 * maturities**3 * cash_flows * discount_factors**4
        else:
            raise ValueError("'kind' must be one of the following: compound_rate, continuous_rate, simple_rate")
            
        # Normalize by the present value
        
        if not dollar:
            asymetry = asymetry / self.present_value(discount_factors, keepdims = True)
            
        # Return output
        
        return np.sum(asymetry, axis = self.axis, keepdims = keepdims)

    

[docs]    def yield_to_maturity(self, price, kind = "continuous_rate", tol = 1e-4, keepdims = False):
        r"""
        Compute the yield to maturity (YTM).
        
        Note
        ----
        The **yield to maturity** is the discount rate such that the sum of all
        future discounted cash-flows of a fixed income instrument is equal to
        its current price :math:`PV^*`.
        
        Depending on the type of interest rates, the yield to maturity consists
        to find the value of the variable :math:`y` such that the following
        equality holds:
        
            * **Continuous interest rates:**
        
                .. math:: \sum_{i=1}^n CF_i e^{-y \tau_i} = PV^*
                
            * **Compound interest rates:**
            
                .. math:: \sum_{i=1}^n \frac{CF_i}{(1 + y)^{\tau_i}} = PV^*
                
            * **Simple interest rates:**
            
                .. math:: \sum_{i=1}^n \frac{CF_i}{1 + y \tau_i} = PV^*

        Note that the yield to maturity is a particular case of the z-spread
        where the initial yield curve is flat and equal to zero.
        
        Since no close formula exists to compute the yield to maturity, a **Newton-Raphson**
        method is used. A tolerance parameter can be used to control the approximation
        error.

        Parameters
        ----------
        price : array-like
            Current price or target present value to reach.
        kind : {"compound_rate", "continuous_rate", "simple_rate"}, optional
            Type of interest rates. The default assumes continuous rates.
        tol : scalar, optional
            Absolute tolerance for the yield to maturity. The maximum absolute error between
            the computed yield to maturity and the exact one will be equal to this tolerance.
            The default uses a tolerance of 1 basis point (1e-4).
        keepdims : bool, optional
            If True, the axis which is reduced is left in the result as dimension
            with size one (otherwise, it will be removed). With this option, the
            result will broadcast correctly against the initial cash-flows.
            The default is False.

        Returns
        -------
        scalar or array
            Yield to maturity.

        Raises
        ------
        ValueError
            If `price` cannot be broadcast with the instance or
            has a size greater than one along the `axis` attribute of the instance.
        """
        
        return self.z_spread(price, discount_factors = np.ones(self.maturities.shape), kind = kind, tol = tol, keepdims = keepdims)

    

[docs]    def z_spread(self, price, discount_factors, kind = "continuous_rate", tol = 1e-4, keepdims = False):
        r"""
        Compute the z-spread.
        
        Note
        ----
        The **z-spread** is a spread or a parallel shift over an initial yield
        curve such that the sum of all future discounted cash-flows of a fixed
        income instrument is equal to its current price :math:`PV^*`.
        
        Depending on the type of interest rates, the z-spread consists to find
        the value of the variable :math:`z_{spread}` such that the following
        equality holds:
        
            * **Continuous interest rates:**
        
                .. math:: \sum_{i=1}^n CF_i e^{-(r(\tau_i) + z_{spread})\tau_i} = PV^*
                
            * **Compound interest rates:**
            
                .. math:: \sum_{i=1}^n \frac{CF_i}{(1 + r(\tau_i) + z_{spread})^{\tau_i}} = PV^*
                
            * **Simple interest rates:**
            
                .. math:: \sum_{i=1}^n \frac{CF_i}{1 + (r(\tau_i) + z_{spread})\tau_i} = PV^*

        Note that the initial yield curve :math:`r(\tau_i)` is derived from the
        provided discount factors which are converted to interest rates.
        
        Since no close formula exists to compute the z-spread, a **Newton-Raphson**
        method is used. A tolerance parameter can be used to control the approximation
        error.

        Parameters
        ----------
        price : array-like
            Current price or target present value to reach.
        discount_factors : array-like or Discounting or YieldCurve
            Discount factors to use when discounting the cash-flows.
            It can be an array, a Discounting instance or an object able to
            evaluate the discount factors.
            If an array or a Discounting instance is provided, it is assumed to
            contain the value of the discount factors for each cash-flow maturity.
        kind : {"compound_rate", "continuous_rate", "simple_rate"}, optional
            Type of interest rates. The default assumes continuous rates.
        tol : scalar, optional
            Absolute tolerance for the z-spread. The maximum absolute error between
            the computed z-spread and the exact one will be equal to this tolerance.
            The default uses a tolerance of 1 basis point (1e-4).
        keepdims : bool, optional
            If True, the axis which is reduced is left in the result as dimension
            with size one (otherwise, it will be removed). With this option, the
            result will broadcast correctly against the initial cash-flows.
            The default is False.

        Returns
        -------
        scalar or array
            Z-spread.

        Raises
        ------
        ValueError
            If `price` cannot be broadcast with the instance or
            has a size greater than one along the `axis` attribute of the instance.
        """
        
        # Control if the target price to reach can be broadcast correctly
        
        price = np.asarray(price)
        
        if not self.is_broadcastable(price):
            raise ValueError(f"'price' cannot be broadcast with the instance: their respective shapes are {price.shape} and {self.shape}")
        elif price.ndim >= -self.axis and price.shape[self.axis] != 1:
            raise ValueError(f"'price' must have a size 1 along the axis {self.axis}")
            
        # Get the initial spot rates from the provided discount factors
        
        spot_rates = self.to_discounting(discount_factors).convert(kind) # Discounting instance
        
        # Define the target function for which the root (z-spread) is sought
        
        def f(z_spread):
            return self.present_value(spot_rates + z_spread, keepdims = True) - price
        
        # Define the derivative of the target function (opposite of the dollar modified duration)
        
        def df(z_spread):
            return -self.duration(spot_rates + z_spread, kind = kind, dollar = True, keepdims = True)
        
        # Newton-Raphson method to find the root (z-spread) of the target function f (we use 0 as initial guess for the z-spread)
        
        z_spread = math.newton_raphson(f, df, 0., tol = tol)["root"]
        
        # Return output
        
        if keepdims:
            return z_spread
        else:
            return np.squeeze(z_spread, self.axis)[()]
