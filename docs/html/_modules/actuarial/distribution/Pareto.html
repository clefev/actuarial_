[docs]class Pareto(Parametric, UnivariateDistribution):
    r"""
    The **Pareto distribution** is a continuous probability distribution defined
    on a semi-infinite interval :math:`[A,+\infty)` and parametrized by a positive
    shape parameter :math:`\alpha`.
    This distribution is characterized by a heavy tail.
    It is a particular case of the generalized Pareto distribution which has one
    additional parameter for more flexibility.
    It is therefore a suitable choice to model the cost of individual large claims
    for which the flexibility of the generalized Pareto distribution is not necessary.
    """
    
    # Class attributes
    
    _parametrization = Parameters(
        shape = Array(min = 0, strict = True),
        min = Array(min = 0, strict = True))
    dtype = np.dtype(float)

[docs]    @UnivariateDistribution.Decorators.cdf
    def cdf(self, x):
        r"""
        Evaluate the cumulative distribution function.
        
        Note
        ----
        The cdf of a Pareto distribution is given by
        
        .. math::
            F(x) = 1 - \left(\frac{x}{A}\right)^{-\alpha}
            
        for :math:`x \geq A`.           
        
        Parameters
        ----------
        x : array-like
            Values for which the cdf should be evaluated.

        Returns
        -------
        scalar or array
            Evaluated cdf. Array with a float dtype and the same shape as the
            broadcast of `x` with the distribution parameters.
                                
        Raises
        ------
        ValueError
            If `x` cannot be broadcast with the distribution parameters.
        """
        
        shape, min = self.parameters.values
        return 1 - (math.pmax(x, min)/min)**(-shape)

    

[docs]    @UnivariateDistribution.Decorators.pdf
    def pdf(self, x):
        r"""
        Evaluate the probability density function.
        
        Note
        ----
        The pdf of a Pareto distribution is given by
        
        .. math::
            f(x) = \frac{\alpha A^{\alpha}}{x^{\alpha + 1}}
            
        for :math:`x \geq A`.           

        Parameters
        ----------
        x : array-like
            Values for which the pdf should be evaluated.

        Returns
        -------
        scalar or array
            Evaluated pdf. Array with a float dtype and the same shape as the
            broadcast of `x` with the distribution parameters.
                                    
        Raises
        ------
        ValueError
            If `x` cannot be broadcast with the distribution parameters.
        """
        
        shape, min = self.parameters.values
        return np.where(x >= min, shape/x * (min/x)**shape, 0.)

    

[docs]    @UnivariateDistribution.Decorators.quantile
    def quantile(self, p):
        r"""
        Evaluate the quantile of the distribution.
        
        Note
        ----
        The quantile of a Pareto distribution is given by
        
        .. math::
            F^{-1}(p) = A(1 - p)^{-\alpha}
            
        for :math:`p \in (0,1)`.

        Parameters
        ----------
        p : array-like
            Values (between 0 and 1) for which the quantile should be evaluated.
        
        Returns
        -------
        scalar or array
            Evaluated quantile. Array with a float dtype and the same shape as the
            broadcast of `p` with the distribution parameters.
            
        Raises
        ------
        ValueError
            If `p` cannot be broadcast with the distribution parameters or
            contains values not between 0 and 1.
        """
        
        shape, min = self.parameters.values
        return min*(1 - p)**(-1/shape)

    def random(self, size = None, seed = -1):
        r"""
        Draw random samples of the distribution.

        Parameters
        ----------
        size : int or tuple of int or None, optional
            Number of samples to be drawn. If None, the instance shape will be used.
            The default is None.
        seed : int or array-like of int or SeedSequence or Generator or None, optional
            Seed to use for the random number generator. The default uses the
            random number generator associated with this instance.

        Returns
        -------
        scalar or array
            Random samples. Array with a float dtype and the same shape as `size`.
            
        Raises
        ------
        ValueError
            If `size` is not a supershape of the instance shape.
        """
        
        rng = self.get_rng(seed)
        return self.quantile(rng.random(size))
