    """
    Abstract class for constructing an univariate distribution.
    
    An univariate distribution describes the probability distribution of a single
    random variable.
    The random variable can be discrete or continuous.
    
    This abstract class implements basic methods such as
    
        * Evaluate the probability density/mass function (`pdf`).
        * Evaluate the cumulative distribution function (`cdf`).
        * Evaluate the quantile (`quantile`).
        * Draw random samples (`random`).
        * Compute the log-likelihood of a sample (`log_likelihood`).
        
    Note that the type of values returned by the random variable (integers or floats)
    is provided by the `dtype` attribute of the instance.
    """

[docs]    def log_likelihood(self, x, weights = None, axis = None, keepdims = False):
        r"""
        Compute the log-likelihood.
        
        Note
        ----
        The log-likelihood of a sample :math:`x = (x_1, ..., x_n)` is calculated as
        
        .. math::
            L = \sum_{i=1}^n w_i \ln\left(f(x_i)\right)
            
        where :math:`w = (w_1, ..., w_n)` is a vector of weights associated with
        each :math:`x_i`.

        Parameters
        ----------
        x : array-like
            Observed sample values from which the log-likelihood should be calculated.
        weights : array-like or None, optional
            Weight associated with each value.
            The default assumes a weight of 1 associated with each observation.
        axis : int or None, optional
            Axis along which the sum of the individual log-likelihood is computed.
            If None, the sum is computed over all the dimensions. The default is None.
        keepdims : bool, optional
            If True, the axes which are reduced are left in the result as dimensions
            with size one (otherwise, they will be removed). With this option,
            the result will broadcast correctly against the input array `x`.
            The default is False.
            
        Returns
        -------
        scalar or array
            Log-likelihood.
        """
  
        if weights is None:
            return np.sum(np.log(self.pdf(x)), axis = axis, keepdims = keepdims)
        else:
            return np.sum(weights*np.log(self.pdf(x)), axis = axis, keepdims = keepdims)

[docs]        def cdf(func):
            """
            Decorator for the implemented `cdf` method of an `UnivariateDistribution` instance.
            
            This decorator implements the following:
                
                * Convert the values for which the cdf should be evaluated to a
                  numpy array.
                * Control whether these values can be broadcast with the instance.
                * Ensure that the returned output is an array with a float dtype
                  and reduce it to a scalar when possible.
    
            Returns
            -------
            callable
                Decorator.
                
            Raises
            ------
            ValueError
                If the values for which the cdf should be evaluated cannot be
                broadcast with the instance.
            """
            
            @wraps(func)
            def wrapper(self, x, *args, **kwargs):
                
                # Control if x can be broadcast with the instance
                
                x = np.asarray(x)
                
                if not self.is_broadcastable(x):
                    raise ValueError(f"'x' cannot be broadcast with the instance: their respective shapes are {x.shape} and {self.shape}")
                
                # Evaluate the cdf
                
                cdf = func(self, x, *args, **kwargs)
                
                # Return output
                
                return np.array(cdf, dtype = float)[()]
            
            wrapper._undecorated = func            
            return wrapper

[docs]        def quantile(func):
            """
            Decorator for the implemented `quantile` method of an `UnivariateDistribution` instance.
            
            This decorator implements the following:
                
                * Convert the values for which the quantile should be evaluated to a
                  numpy array.
                * Control whether these values can be broadcast with the instance
                  and are between 0 and 1.
                * Ensure that the returned output is an array with the same dtype
                  as the `dtype` attribute of the instance and reduce it to a scalar
                  when possible. 
    
            Returns
            -------
            callable
                Decorator.
                
            Raises
            ------
            ValueError
                If the values for which the quantile should be evaluated cannot
                be broadcast with the instance or contain values not between 0 and 1.
            """
            
            @wraps(func)
            def wrapper(self, p, *args, **kwargs):
                
                # Control if p can be broadcast with the instance and is between 0 and 1
                
                p = np.asarray(p)
                
                if not self.is_broadcastable(p):
                    raise ValueError(f"'p' cannot be broadcast with the instance: their respective shapes are {p.shape} and {self.shape}")
                elif np.any(p > 1) or np.any(p < 0):
                    raise ValueError("'p' must contain values between 0 and 1")
                    
                # Evaluate the quantile
                
                quantile = func(self, p, *args, **kwargs)
                
                # Return output
                
                return np.array(quantile, dtype = self.dtype)[()]
                
            wrapper._undecorated = func            
            return wrapper

        

[docs]        def random(func):
            """
            Decorator for the implemented `random` method of an `UnivariateDistribution` instance.
            
            This decorator implements the following:
                
                * Convert the size of the samples to a tuple of integers.
                * Control if the size of the samples is a supershape of the instance shape.
                * Ensure that the returned output is an array with the same dtype
                  as the `dtype` attribute of the instance and reduce it to a scalar
                  when possible.
    
            Returns
            -------
            callable
                Decorator.
                
            Raises
            ------
            ValueError
                If the size of the samples to be drawn is not a supershape of the
                instance shape.
            """
            
            @wraps(func)
            def wrapper(self, size = None, *args, **kwargs):
                
                # Convert size to a tuple of integers
    
                if size is None:
                    size = self.shape
                else:
                    try:
                        size = (int(size),)
                    except TypeError:
                        size = tuple([int(s) for s in size])
    
                # Control if size is a supershape of the instance shape
                
                if not util.is_supershape(size, self.shape):
                    raise ValueError(f"'size' must be a supershape of the instance shape: their respective shapes are {size} and {self.shape}")
                
                # Generate random samples of the distribution
                
                random = func(self, size, *args, **kwargs)
                    
                # Return output
                
                return np.array(random, dtype = self.dtype)[()]
            
            wrapper._undecorated = func            
            return wrapper
