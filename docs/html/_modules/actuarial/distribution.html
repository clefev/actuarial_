    def log_likelihood(self, x, weights = None, axis = None, keepdims = False):
        """
        Compute the log-likelihood.

        Parameters
        ----------
        x : array-like
            Observed values from which the log-likelihood will be calculated.
        weights : array-like or None, optional
            Weight associated with each value. If None, the weight is assumed to
            be 1 for each observation. The default is None.
        axis : int or None, optional
            Axis along which the sum of the individual log-likelihood is computed.
            If None, the sum is computed over all the dimensions. The default is None.
        keepdims : bool, optional
            If True, the axes which are reduced are left in the result as dimensions
            with size one (otherwise, they will be removed). With this option,
            the result will broadcast correctly against the input array `x`.
            The default is False.
            
        Returns
        -------
        scalar or array:
            Computed log-likelihood.
        """
                
        if weights is None:
            return np.sum(np.log(self.pdf(x)), axis = axis, keepdims = keepdims)
        else:
            return np.sum(weights*np.log(self.pdf(x)), axis = axis, keepdims = keepdims)

    class Decorators:
        """
        Container class containing decorators to use for the implemented

        def cdf(func):
            """
            Decorator for the implemented `cdf` method of an univariate distribution.
            
            This decorator implements the following:
                
                * Control wether the values for which the cdf should be evaluated
                  can be broadcast with the instance.
                * Ensure that the returned output is an array with a float dtype
                  and reduce it to a scalar when possible.
    
            Parameters
            ----------
            func : callable
                Method evaluating the cdf. The first argument of this method
                must be the values for which the cdf should be evaluated.
                Other additional arguments are allowed.
    
            Returns
            -------
            callable
                Decorator.
            """
            
            @wraps(func)
            def wrapper(self, x, *args, **kwargs):
                
                # Control if x can be broadcast with the instance
                
                x = np.asarray(x)
                
                if not self.is_broadcastable(x):
                    raise ValueError(f"'x' cannot be broadcast with the instance: their respective shapes are {x.shape} and {self.shape}")
                
                # Evaluate the cdf
                
                cdf = func(self, x, *args, **kwargs)
                
                # Return output
                
                return np.array(cdf, dtype = float)[()]
            
            wrapper._undecorated = func            
            return wrapper
        
        def pdf(func):
            """
            Decorator for the implemented `pdf` method of an univariate distribution.
            
            This decorator implements the following:
                
                * Control wether the values for which the pdf should be evaluated
                  can be broadcast with the instance.
                * Ensure that the returned output is an array with a float dtype
                  and reduce it to a scalar when possible. 
                
            Parameters
            ----------
            func : callable
                Method evaluating the pdf. The first argument of this method
                must be the values for which the pdf should be evaluated.
                Other additional arguments are allowed.
    
            Returns
            -------
            callable
                Decorator.
            """
            
            @wraps(func)
            def wrapper(self, x, *args, **kwargs):
                
                # Control if x can be broadcast with the instance
                
                x = np.asarray(x)
                
                if not self.is_broadcastable(x):
                    raise ValueError(f"'x' cannot be broadcast with the instance: their respective shapes are {x.shape} and {self.shape}")
                
                # Evaluate the pdf
                
                pdf = func(self, x, *args, **kwargs)
                    
                # Return output
                
                return np.array(pdf, dtype = float)[()]
                
            wrapper._undecorated = func            
            return wrapper
        
        def quantile(func):
            """
            Decorator for the implemented `quantile` method of an univariate distribution.
            
            This decorator implements the following:
                
                * Control wether the values for which the quantile should be evaluated
                  can be broadcast with the instance and are between 0 and 1.
                * Ensure that the returned output is an array with the same dtype
                  as the `dtype` attribute of the instance and reduce it to a scalar
                  when possible. 
                    
            Parameters
            ----------
            func : callable
                Method evaluating the quantile. The first argument of this method
                must be the values (between 0 and 1) for which the quantile should
                be evaluated. Other additional arguments are allowed.
    
            Returns
            -------
            callable
                Decorator.
            """
            
            @wraps(func)
            def wrapper(self, p, *args, **kwargs):
                
                # Control if p can be broadcast with the instance and is between 0 and 1
                
                p = np.asarray(p)
                
                if not self.is_broadcastable(p):
                    raise ValueError(f"'p' cannot be broadcast with the instance: their respective shapes are {p.shape} and {self.shape}")
                elif np.any(p < 0) or np.any(p > 1):
                    raise ValueError("'p' must be between 0 and 1")
                    
                # Evaluate the quantile
                
                quantile = func(self, p, *args, **kwargs)
                
                # Return output
                
                return np.array(quantile, dtype = self.dtype)[()]
                
            wrapper._undecorated = func            
            return wrapper
        
        def random(func):
            """
            Decorator for the implemented `random` method of an univariate distribution.
            
            This decorator implements the following:
                
                * Convert the size of the samples to a tuple of integers.
                * Control if the size of the samples is a supershape of the instance shape.
                * Ensure that the returned output is an array with the same dtype
                  as the `dtype` attribute of the instance and reduce it to a scalar
                  when possible.
    
            Parameters
            ----------
            func : callable
                Method generating the random samples. The first argument of this
                method must be the size of the samples to be drawn.
                Other additional arguments are allowed.
    
            Returns
            -------
            callable
                Decorator.
            """
            
            @wraps(func)
            def wrapper(self, size = None, *args, **kwargs):
                
                # Convert size to a tuple of integers
    
                if size is None:
                    size = self.shape
                else:
                    try:
                        size = (int(size),)
                    except TypeError:
                        size = tuple([int(s) for s in size])
    
                # Control if size is a supershape of the instance shape
                
                if not util.is_supershape(size, self.shape):
                    raise ValueError(f"'size' must be a supershape of the instance shape: their respective shapes are {size} and {self.shape}")
                
                # Generate random samples of the distribution
                
                random = func(self, size, *args, **kwargs)
                    
                # Return output
                
                return np.array(random, dtype = self.dtype)[()]
            
            wrapper._undecorated = func            
            return wrapper

        ------
        ValueError
            If `x` cannot be broadcast with the distribution parameters.
        """

    def cdf(self, x):
        r"""
        Evaluate the cumulative distribution function.
        
        Note
        ----
        The cdf of an empirical distribution is given by
        
        .. math::
            \hat{F}_n(x) = \frac{\text{number of elements in the sample } \leq x}{n} = \frac{1}{n} \sum_{i=1}^n \mathbb{1}_{x_i \leq x}
            
        for :math:`x \in \mathbb{R}`. If all :math:`x_i`'s are distint, we have
        
        .. math::
            \hat{F}_n(x_{(i)}) = \frac{i}{n}
    
        where :math:`x_{(i)}` is the i-th order statistics.
        
        Parameters
        ----------
        x : array-like
            Values for which the cdf should be evaluated.

        Returns
        -------
        scalar or array
            Evaluated cdf. Array with a float dtype and the same shape as `x`.
        """
        
        mask = (x >= self.values[0]) # boolean array indicating elements of x having a non-zero cdf (i.e. elements greater than the minimum value)
        indices = np.searchsorted(self.values, x, side = "right") - 1 # match each element of x having a non-zero cdf with the sorted unique values (lower bound match)
        cdf = np.full(x.shape, 0.)
        cdf[mask] = self.cumulative_probabilities[indices]
        return cdf

[docs]    @UnivariateDistribution.Decorators.quantile
    def quantile(self, p, maxiter = 100, tol = 1e-4):
        r"""
        Evaluate the quantile of the distribution.

        Note
        ----
        No close formula exists to compute the quantile of a mixture distribution,
        a bisection method is used in order to approximate it. The bisection
        finds the value :math:`x^*` such that the cdf of the mixture distribution
        :math:`F(x^*)` is equal to :math:`p`. This value is then returned as the
        quantile.
        
        Parameters
        ----------
        p : array-like
            Values (between 0 and 1) for which the quantile should be evaluated.
        maxiter : int, optional
            Maximum number of iterations for the bisection algorithm.
            The default is 100.
        tol : float, optional
            Tolerance of the bisection algorithm. The maximum error between
            the computed quantile and the exact quantile will be equal to this
            tolerance. The default is 1e-4.

        Returns
        -------
        scalar or array
            Evaluated quantile. Array with the same shape as the broadcast of
            `p` with the individual distributions and weights.
            
        Raises
        ------
        ValueError
            If `p` cannot be broadcast with the individual distributions and
            weights, or contains values not between 0 and 1.
        """
        
        # Quantiles of each distribution: the minimum and maximum quantile are used as initial interval for the bisection algorithm
        
        quantiles = [component.quantile(p) for component in self.components]
        lower = math.pmin(*quantiles) - 1e-6
        upper = math.pmax(*quantiles) + 1e-6
        
        # Target function for which the root (quantile) must be found
        
        f = lambda x: np.round(self.cdf(x) - p, 10)
                
        # Bisection algorithm to find the quantile
        
        bisection = math.bisection(f, lower, upper, maxiter = maxiter, tol = tol, f_tol = -1)
        quantile = bisection["root"]
        
        # Return output
        
        return quantile

    def random(self, size = None, seed = -1):
        r"""
        Draw random samples of the distribution.
        
        Note
        ----
        A random sample of a mixture distribution is drawn as follows:
            
            * Select the individual distribution to generate by drawing randomly
              an integer :math:`i \in \lbrace1,...,n\rbrace` where :math:`p(i)=w_i`.
            * Draw a random sample for the selected i-th distribution :math:`X_i`.

        Parameters
        ----------
        size : int or tuple of int or None, optional
            Number of samples to be drawn. If None, the instance shape will be used.
            The default is None.
        seed : int or array-like of int or SeedSequence or Generator or None, optional
            Seed to use for the random number generator. The default uses the
            random number generator associated with this instance.

        Returns
        -------
        scalar or array
            Random samples. Array with the same shape as `size`.

        Raises
        ------
        ValueError
            If `size` is not a supershape of the instance shape.
        """
        
        # Control the shape of components and weights
        
        if util.broadcast(*self.components).shape not in ((), (1,)):
            raise ValueError("the shape of each individual distribution must be an empty tuple or (1,)")
        elif util.broadcast(*self.weights).ndim > 0:
            raise ValueError("'weights' attribute must be a list of scalars")
                
        # Get the random number generator
        
        rng = self.get_rng(seed)
        random = np.empty(size)
        
        # Simulate the occurrences of each distribution according to their weights
        
        indices = rng.choice(len(self), p = self.weights, size = size)
        
        # Simulate the values of each distribution
                
        for i, component in enumerate(self.components):
            mask = (indices == i)
            n = np.count_nonzero(mask)     
            if n > 0:
                 random[mask] = component.random(size = n, seed = self.give_seed())
                
        # Return output
        
        return random
