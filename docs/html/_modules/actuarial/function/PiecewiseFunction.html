[docs]    def __init__(self, intervals, function, right_closed = False, shift = False):
        """
        Parameters
        ----------
        intervals : array-like
            Partition :math:`x_0 < x_1, ... < x_n` dividing the domain of the
            piecewise function. 
            The domain of the piecewise function will be divided into the following
            subintervals: :math:`[x_0,x_1)`, :math:`[x_1,x_2)`, ..., :math:`[x_{n-1},x_n)`
            and :math:`[x_n,\infty)`.
            The provided values must be strictly increasing.
        function : Function
            Parametric function to be applied on each interval of the piecewise function.
            Parameters of the function must be given for each interval. The first
            interval of the piecewise function [intervals[0], intervals[1]] will
            use the function parameters from index 0, the second interval [intervals[1], intervals[2]]
            will use parameters from index 1, ... and the last interval [intervals[N], inf]
            will use parameters from index N.
        right_closed : bool, optional
            Indicate whether the function :math:`f_i` should be used on the
            right-closed interval :math:`(x_i,x_{i+1}]` (True) or the left-closed
            interval :math:`[x_i,x_{i+1})` (False).
            The default assumes a left-closed interval (False).            
        shift : bool, optional
            Indicate whether the function :math:`f_i` should be horizontally shifted.
            If True, the function :math:`f_i` at point :math:`x` will be evaluated
            as :math:`f_i(x-x_i)` instead of :math:`f_i(x)`.
            The default assumes no shift (False).

        Raises
        ------
        TypeError
            If `function` is not a `Function` and `Parametric` instance.
        ValueError
            If `intervals` is not along the first axis.
            If `function` has not the same size as `intervals` along the partition axis.
        """
            
        # Control if intervals is along the first axis
        
        intervals = np.asarray(intervals)
        
        if not util.is_along_axis(intervals):
            raise ValueError(f"'intervals' must be along axis 0: its shape is {intervals.shape}")
            
        self._intervals = intervals
            
        # Control if function is a Function and Parametric instance and has the same size as intervals along the axis
        
        if not (isinstance(function, Function) and isinstance(function, Parametric)):
            raise TypeError("'function' must be a 'Function' and 'Parametric' instance")
        elif not util.is_along_axis(function, self.axis, size = intervals.size):
            raise ValueError(f"'function' must have the same size as 'intervals' along the axis {self.axis}: its shape is {function.shape}")
            
        self._function = function
            
        # Other piecewise options
        
        self._right_closed = bool(right_closed)
        self._shift = bool(shift)

    @property
    def shift(self):
        """
        bool: Indicate whether the function :math:`f_i` should be horizontally
        shifted.
        If True, the function :math:`f_i` at point :math:`x` will be evaluated
        as :math:`f_i(x-x_i)` instead of :math:`f_i(x)`.
        """
        
        return self._shift
    
    @property
    @memorize
    def cumulative_integral(self):
        """
        array: Compute the cumulative integrals :math:`\int_{x_0}^{x_0} f(x) dx`,
        :math:`\int_{x_0}^{x_1} f(x) dx`, ..., :math:`\int_{x_0}^{x_n} f(x) dx`
        (used for preprocessing purpose when using the `integrate` method).
        """
        
        if self.shift:
            cumulative_integral = self.get_function(slice(-1)).integrate(0, self.to_partition()[1:].values - self.to_partition()[:-1].values).cumsum(axis = self.axis)
        else:
            cumulative_integral = self.get_function(slice(-1)).integrate(self.to_partition()[:-1].values, self.to_partition()[1:].values).cumsum(axis = self.axis)
        
        return util.concatenate(0, cumulative_integral, axis = self.axis)

    def broadcast(self, *args):
        """
        Mimic the broadcast of the piecewise function with input arrays.
        
        For this broadcast, the size of the function parameters along the partition
        axis is reduced (considered as 1).

        Parameters
        ----------
        *args : array-like or object with shape attribute
            Inputs to broadcast with the piecewise function.

        Returns
        -------
        np.broadcast
            Broadcast result.
        """
        
        shape = list(self.function.shape) # get the shape of the parameters of the underlying function
        shape[self.axis] = 1 # reduce the size along the 'axis' attribute
        return util.broadcast(np.empty(shape, dtype = []), *args)
    

[docs]    @Function.Decorators.evaluate
    def evaluate(self, x):
        r"""
        Evaluate the value of the function.
        
        Note
        ----
        The value of a piecewise function is given by
        
        .. math::
            f(x) = \begin{cases}
                        f_0(x) & \text{if } x < x_1, \\
                        f_1(x) & \text{if } x_1 \leq x < x_2, \\
                        ~ \vdots \\
                        f_{n-1}(x) & \text{if } x_{n-1} \leq x < x_n, \\
                        f_n(x) & \text{if } x \geq x_n.
                    \end{cases}
                    
        If the `right_closed` attribute is set to True (and not False), the
        inequalities above should change: ':math:`<`' becomes ':math:`\leq`' and
        ':math:`\leq`' becomes ':math:`<`'.
        
        If the `shift` attribute is set to True (and not False), the function
        :math:`f_i(x)` above should be replaced by :math:`f_i(x-x_i)`.

        Parameters
        ----------
        x : array-like
            Values for which the function should be evaluated.

        Returns
        -------
        scalar or array
            Evaluated function.

        Raises
        ------
        ValueError
            If `x` cannot be broadcast with the instance.
        """
        
        indices = self.to_partition().index(x)
        x = x - self.intervals.ravel()[indices] if self.shift else x
        return self.get_function(indices).evaluate(x)

    def derivative(self, x, order = 1):
        r"""
        Compute the derivative.
        
        Note
        ----
        The derivative of a piecewise function is given by
        
        .. math::
            f'(x) = \begin{cases}
                        f_0'(x) & \text{if } x < x_1, \\
                        f_1'(x) & \text{if } x_1 \leq x < x_2, \\
                        ~ \vdots \\
                        f_{n-1}'(x) & \text{if } x_{n-1} \leq x < x_n, \\
                        f_n'(x) & \text{if } x \geq x_n.
                    \end{cases}
                    
        If the `right_closed` attribute is set to True (and not False), the
        inequalities above should change: ':math:`<`' becomes ':math:`\leq`' and
        ':math:`\leq`' becomes ':math:`<`'.
        
        If the `shift` attribute is set to True (and not False), the function
        :math:`f_i(x)` above should be replaced by :math:`f_i(x-x_i)`.

        Parameters
        ----------
        x : array-like
            Values for which the derivative should be evaluated.
        order : int, optional
            Order of the derivative. The default is 1.

        Returns
        -------
        scalar or array
            Evaluated derivative.
            
        Raises
        ------
        ValueError
            If `x` cannot be broadcast with the instance.
            If `order` is not a positive integer.
        """
        
        indices = self.to_partition().index(x)
        x = x - self.intervals.ravel()[indices] if self.shift else x
        return self.get_function(indices).derivative(x, order)

    def integrate(self, lower, upper):
        r"""
        Compute the definite integral.
        
        Note
        ----
        xxxxxxxxxxxxxxxxxxxxxxxxxxxx
            
        Parameters
        ----------
        lower : array-like
            Lower bound of the definite integral interval.
        upper : array-like
            Upper bound of the definite integral interval.
        
        Returns
        -------
        scalar or array
            Evaluated integral.

        Raises
        ------
        ValueError
            If `lower` and `upper` cannot be broadcast with the instance.
        """

        # Get interval indices associated with lower and upper
        
        intervals = self.intervals.ravel()
        lower, upper = np.broadcast_arrays(lower, upper)
        indices_below_lower = util.lower_bound(lower, intervals, index = True) # indices of the interval below lower
        indices_below_upper = util.lower_bound(upper, intervals, index = True) # indices of the interval below upper
        indices_above_lower = util.upper_bound(lower, intervals, index = True) # indices of the interval above lower

        # Compute the definite integral between lower and upper
                
        if self.shift:
            integral = (self.get_function(indices_below_lower).integrate(lower - intervals[indices_below_lower], intervals[indices_above_lower] - intervals[indices_below_lower]) + # integral on [lower, interval above lower]
                        self.to_partition().take(self.cumulative_integral, indices_below_upper) - self.to_partition().take(self.cumulative_integral, indices_above_lower) + # integral on [interval above lower, interval below upper]
                        self.get_function(indices_below_upper).integrate(0, upper - intervals[indices_below_upper])) # integral on [interval below upper, upper]
        else:
            integral = (self.get_function(indices_below_lower).integrate(lower, intervals[indices_above_lower]) + # integral on [lower, interval above lower]
                        self.to_partition().take(self.cumulative_integral, indices_below_upper) - self.to_partition().take(self.cumulative_integral, indices_above_lower) + # integral on [interval above lower, interval below upper]
                        self.get_function(indices_below_upper).integrate(intervals[indices_below_upper], upper)) # integral on [interval below upper, upper]
        
        # Return output
        
        return integral

    
    def get_function(self, indices = None):
        """
        Get the parametric function associated with the given partition indices.
        
        xxxxxxxxxxxxxxxxxxxxxxxx

        Parameters
        ----------
        indices : int or slice or array-like or None, optional
            The parameters of the function will be given for these partition indices.
            If None, the `function` attribute is returned.
            The default is None.
            
        Returns
        -------
        function : Function
            Parametric function where the parameters are associated with the given
            partition indices.
        """
        
        if indices is None:
            return self.function
        else:
            parameters = self.function.get_parametrization().set(*[self.to_partition().take(value, indices) for value in self.function.parameters.values])
            return self.function.copy().set_attributes(parameters = parameters)
        
    def to_partition(self):
        """
        Return a `Partition` instance.

        Returns
        -------
        Partition
            Partition :math:`x_0 < x_1 < ... < x_n`.
        """
        
        return Partition(self.intervals, right_closed = self.right_closed)

    @classmethod
    def interpolate(cls, x, y, kind = "linear"):
        """
        Perform a piecewise interpolation given observed values for the x-axis
        and the y-axis.

        Parameters
        ----------
        x : array-like
            Observed values for the x-axis. The values must be strictly increasing
            and along the first axis (i.e. only the first axis can have a size greater
            than one, the other ones must have a size equal to one).
        y : array-like
            Observed values for the y-axis. Must have the same size as x along
            the interpolation axis.
        kind : {"exponential", "linear", "logarithm", "lower", "upper"}, optional
            Type of interpolation. The default is "linear".

        Raises
        ------
        ValueError
            If kind is not valid.
            If x is not along the first axis.
            If x has not at least two values.
            If y has not the same size as x along the interpolation axis.

        Returns
        -------
        PiecewiseFunction
            PiecewiseFunction instance such that the value of the function at x
            is equal to y.

        """

        # Control the kind
        
        kinds = ("exponential", "linear", "logarithm", "lower", "upper")
        
        if kind not in kinds:
            raise ValueError(f"kind must be one of the following: {', '.join(kinds)}")
            
        # Convert x and y to numpy array
        
        x, y = np.asarray(x), np.asarray(y)
        
        # Control if x is along the first axis and have at least two values
        
        if not util.is_along_axis(x):
            raise ValueError(f"x must be along axis 0: its shape is {x.shape}")
        elif x.size < 2 and kind not in ("lower", "upper"):
            raise ValueError("x must have at least two values for this kind of interpolation")
            
        # Control if y has the same size as x along the interpolation axis
        
        if not util.is_along_axis(y, -x.ndim, x.size):
            raise ValueError(f"y must have the same size as x along the axis {-x.ndim}: its shape is {y.shape}")
        
        # Calibrate and build the interpolated piecewise function
        
        axis = -x.ndim
          
        if kind == "exponential":
            intervals = x[:-1]
            rate = np.diff(np.log(y), axis = axis) / np.diff(x, axis = axis)
            scale = util.take(y, slice(-1), axis = axis)
            piecewise_function = cls(intervals, Exponential(scale = scale, rate = rate), shift = True)
        elif kind == "linear":
            intervals = x[:-1]
            intercept = util.take(y, slice(-1), axis = axis)
            slope = np.diff(y, axis = axis) / np.diff(x, axis = axis)
            piecewise_function = cls(intervals, Linear(slope = slope, intercept = intercept), shift = True)
        elif kind == "logarithm":
            intervals = x[:-1]
            shift = util.take(y, slice(-1), axis = axis)
            scale = np.diff(y, axis = axis) / np.diff(np.log(x), axis = axis)
            rate = 1/intervals
            piecewise_function = cls(intervals, Logarithm(shift = shift, scale = scale, rate = rate))
        elif kind == "lower":
            intervals = x
            constant = y
            piecewise_function = cls(intervals, Constant(constant = constant))
        elif kind == "upper":
            intervals = util.concatenate(x[0]-1, x[:-1], axis = axis)
            constant = y
            piecewise_function = cls(intervals, Constant(constant = constant), right_closed = True)
        
        # Return the PiecewiseFunction instance
        
        return piecewise_function
