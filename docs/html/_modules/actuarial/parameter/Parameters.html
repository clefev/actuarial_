[docs]class Parameters(Dictionary, Base, Broadcast):
    """
    Class representing a collection of parameters.
    
    This class inherits from the `Dictionary` superclass.
    Therefore, it represents a collection of key-parameter pairs.
    The key is a string representing the name of the parameter and the parameter
    is a `Parameter` instance.
    This class allows to control, update and reset several parameters at once.
    """
    
    # Class attribute
    
    _component_type = Parameter

    @property
    def shapes(self):
        """
        dict of tuple: Shape of each parameter.
        """
        
        return {key: parameter.shape for key, parameter in self.items()}
    
    @property
    def values(self):
        """
        list: Value of each parameter.
        """
        
        return [parameter.value for parameter in self]

[docs]    def control(self):
        """
        Control whether the value of each parameter is valid or not.

        For each parameter, the associated `control` method will be called.
        If a parameter is not valid, an error will be raised.
        If the instance contains a parameter called `min` and a parameter called
        `max`, an additional control is performed to ensure that `min < max`.

        Raises
        ------
        ValueError
            If a parameter is not valid.
            If the instance contains a parameter called `min` and a parameter
            called `max` for which `min >= max`.
        """
        
        # Control each individual parameter through its 'control' method
        
        for key, parameter in self.items():
            parameter.control(name = key)
            
        # Control min-max parameters to ensure min < max
            
        if "min" in self and "max" in self:
            if np.any(self["min"].value >= self["max"].value):
                raise ValueError("'max' must be greater than 'min'")

            

[docs]    def reset(self):
        """
        Reset the value of each parameter.
        
        The value of each parameter will be set to None meaning that they are
        not initialized. 
        
        Returns
        -------
        Parameters
            Instance.        
        """
        
        for parameter in self:
            parameter.reset()
            
        return self

    

[docs]    def set(self, *args, **kwargs):
        """
        Set the value of the parameters.

        Parameters
        ----------
        *args : array-like or Parameter or None
            New parameter values. The first argument is the new value
            of the first parameter, the second argument is the new value of the
            second parameter, ... The number of arguments provided must be equal
            to the number of parameters of the instance.
        **kwargs : array-like or Parameter or None
            New parameter values given as key-value pairs.
            The keyword is the key of the parameter to update.
            If the key doesn't exist, an error will be raised.

        Returns
        -------
        Parameters
            Instance.
            
        Raises
        ------
        ValueError
            If the number of arguments provided by `args` is not the same as the
            number of parameters of the instance.
            If the key of a parameter doesn't exist.
        """
        
        # Control the keys from kwargs argument
        
        self._control_keys(*kwargs.keys())
        
        # Update parameters given by args
        
        if args:
            if len(args) == len(self):
                for key, value in zip(self.keys(), args):
                    self[key].set(value)
            else:
                raise ValueError(f"{len(args)} values are given through 'args' instead of {len(self)} associated with the following parameters: {', '.join(self.keys())}")
        
        # Update parameters given by kwargs
            
        for key, value in kwargs.items():
            self[key].set(value)
            
        # Return output
            
        return self
