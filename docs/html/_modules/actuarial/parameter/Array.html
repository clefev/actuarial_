    """    
    Array parameter.
    
    An array parameter is a parameter for which its value is an array (numpy array).
    It accepts the following contraints:
        
        * Particular dtype (e.g., to restrict to an array of integers or floats).
        * Lower bound (minimum value below which the array elements cannot be).
        * Upper bound (maximum value above which the array elements cannot be).
    """
    
    # Class attributes
    
    _print = ("value", "min", "max")
    
    # Initialization
    

[docs]    def __init__(self, dtype = None, min = None, max = None, strict = False):
        """
        Parameters
        ----------
        dtype : np.dtype, optional
            Dtype of the parameter.
            The default accepts any dtype.
        min : scalar, optional
            Lower bound of the parameter (minimum accepted value).
            The default assumes no lower bound.
        max : scalar, optional
            Upper bound of the parameter (maximum accepted value).
            The default assumes no upper bound.
        strict : bool, optional
            If True, the value must be in the open interval `(min, max)`.
            Otherwise, it can be in the closed interval `[min, max]`.
            The default is False (closed interval).

        Raises
        ------
        ValueError
            If `min` and `max` are not scalars.
            If `max` is not greater than `min`.
        """
        
        # Control the lower bound
        
        if min is not None:
            min = np.asarray(min)[()]
            if min.ndim > 0:
                raise ValueError(f"'min' must be a scalar: not an array with shape {min.shape}")
                
        self._min = min
                
        # Control the upper bound
                
        if max is not None:
            max = np.asarray(max)[()]
            if max.ndim > 0:
                raise ValueError(f"'max' must be a scalar: not an array with shape {max.shape}")
                
        self._max = max
                
        # Control if the upper bound is greater than the lower bound
        
        strict = bool(strict)
        
        if min is not None and max is not None:
            if strict and not np.all(min < max):
                raise ValueError("'max' must be greater than 'min'")
            elif not np.all(min <= max):
                raise ValueError("'max' must be greater than or equal to 'min'")
            
        self._strict = strict
        
        # Control if dtype can be interpreted as a valid data type
        
        if dtype is not None:
            dtype = np.dtype(dtype)
            
        self._dtype = dtype

            
    # Properties
    
    @property
    def min(self):
        """
        scalar or None: Lower bound of the parameter.
        If None, no lower bound is applied.
        """
        
        return self._min
    
    @property
    def max(self):
        """
        scalar or None: Upper bound of the parameter.
        If None, no upper bound is applied.
        """
        
        return self._max
    
    @property
    def bounds(self):
        """
        tuple: Tuple with the lower bound and upper bound.
        """
        
        return (self.min, self.max)
    
    @property
    def dtype(self):
        """
        np.dtype or None: Data type of the parameter.
        If None, all data types are accepted.
        """
        
        return self._dtype
    
    @property
    def strict(self):
        """
        bool: Indicate whether the value of the parameter must be strictly between
        the lower bound and upper bound.
        """
        
        return self._strict

    def _dict_to_print(self):
        
        dictionary = super()._dict_to_print()
        
        if self.min is None:
            del dictionary["min"]
            
        if self.max is None:
            del dictionary["max"]
            
        return dictionary

    def control(self, value = None, name = "value"):
        """
        Control whether the value of the parameter is valid or not.
        
        The value of the parameter is valid if it satisfies all the following
        conditions:
            
            * Correct dtype.
            * Greater than the lower bound.
            * Less than the upper bound.

        Parameters
        ----------
        value : array-like or None, optional
            The value to control.
            If None, the value of the instance will be used instead.
            The default is None.
        name : str, optional
            Name of the parameter.
            If an error is raised, this name will be used.
            The default is "value".

        Raises
        ------
        ValueError
            If the parameter is not initialized (set to None).
            If one of the condition above is not fulfilled.
        """
            
        # Control if the value has a correct dtype
        
        if self.dtype is not None:
            if not np.issubdtype(value.dtype, self.dtype):
                raise ValueError(f"'{name}' must have the following dtype: {type(self.dtype).__name__}")
            
        # Control if the value is greater than the lower bound
        
        if self.min is not None:
            if self.strict and not np.all(value > self.min):
                raise ValueError(f"'{name}' must be greater than {self.min}")
            elif not np.all(value >= self.min):
                raise ValueError(f"'{name}' must be greater than or equal to {self.min}")
                
        # Control if the value is less than the upper bound
        
        if self.max is not None:
            if self.strict and not np.all(value < self.max):
                raise ValueError(f"'{name}' must be less than {self.max}")
            elif not np.all(value <= self.max):
                raise ValueError(f"'{name}' must be less than or equal to {self.max}")
