[docs]    @catch_maturities()
    def capitalization_factor(self, maturities1, maturities2):
        r"""
        Compute the inflation capitalization factor between two maturities.
        
        Note
        ----
        The inflation capitalization factor between maturities :math:`\tau_1`
        and :math:`\tau_2` is given by
        
        .. math::
            \frac{PI(\tau_2)}{PI(\tau_1)}
            
        where :math:`PI(\tau)` is the price index at maturity :math:`\tau`.
        The capitalization factor is the mathematical inverse of the discount factor.

        Parameters
        ----------
        maturities1 : array-like
            Starting maturities :math:`\tau_1`.
        maturities2 : array-like
            Ending maturities :math:`\tau_2`.

        Returns
        -------
        scalar or array
            Inflation capitalization factor between `maturities1` and `maturities2`.
            
        Raises
        ------
        ValueError
            If `maturities1` and `maturities2` cannot be broadcast with the instance.
        """

        return self.price_index(maturities2) / self.price_index(maturities1)

    def inflation(self, maturities1, maturities2):
        r"""
        Compute the rate of inflation between two maturities.
        
        Note
        ----
        The rate of inflation between maturities :math:`\tau_1` and :math:`\tau_2`
        is given by
        
        .. math::
            inflation(\tau_1, \tau_2) = \frac{PI(\tau_2)}{PI(\tau_1)} - 1

        Parameters
        ----------
        maturities1 : array-like
            Starting maturities :math:`\tau_1`.
        maturities2 : array-like
            Ending maturities :math:`\tau_2`.

        Returns
        -------
        scalar or array
            Rate of inflation between `maturities1` and `maturities2`.
            
        Raises
        ------
        ValueError
            If `maturities1` and `maturities2` cannot be broadcast with the instance.
        """
 
        return self.price_index(maturities2) / self.price_index(maturities1) - 1

[docs]        def price_index(func):
            """
            Decorator for the implemented `price_index` method of an `Indexation`
            instance.
            
            This decorator implements the following:
                
                * Convert the maturities for which the price index should be
                  evaluated to a numpy array.
                * Control whether these maturities can be broadcast with the instance.
                * Ensure that the returned output is an array and reduce it to
                  a scalar when possible.
    
            Parameters
            ----------
            func : callable
                Method evaluating the price index. The first argument of this method
                must be the maturities for which the price index should be evaluated.
                Other additional arguments are allowed.
    
            Returns
            -------
            callable
                Decorator.
                
            Raises
            ------
            ValueError
                If the maturities for which the price index should be evaluated
                cannot be broadcast with the instance.
            """
            
            @wraps(func)
            def wrapper(self, maturities, *args, **kwargs):
                
                # Control if maturities can be broadcast with the instance
                
                maturities = np.asarray(maturities)
                
                if not self.is_broadcastable(maturities):
                    raise ValueError(f"'maturities' cannot be broadcast with the instance: their respective shapes are {maturities.shape} and {self.shape}")
                    
                # Evaluate the price index
                
                price_index = func(self, maturities, *args, **kwargs)
                
                # Return output
                    
                return np.asarray(price_index)[()]
            
            wrapper._undecorated = func            
            return wrapper

[docs]    @Indexation.Decorators.price_index
    def price_index(self, maturities):
        r"""
        Evaluate the price index for the given maturities.
        
        Note
        ----
        Given a constant indexation rate :math:`g`, the price index at maturity
        :math:`\tau` is given by
        
        .. math::
            PI(\tau) =  \begin{cases}
                            PI(0) (1 + g)^\tau & \text{if } g \text{ is compounded}, \\
                            PI(0) (1 + g \tau) & \text{if } g \text{ is not compounded}.
                        \end{cases}

        Parameters
        ----------
        maturities : array-like
            Maturities for which the price index should be evaluated.

        Returns
        -------
        scalar or array
            Evaluated price index.
            
        Raises
        ------
        ValueError
            If `maturities` cannot be broadcast with the instance.
        """
        
        rate = self.parameters["rate"].value
        
        if self.compound:
            return self.base*(1 + rate)**maturities
        else:
            return self.base*(1 + rate*maturities)

    # Class attributes
    
    _print = ("maturities", "values", "kind")
    kinds = ("exponential", "linear", "logarithm", "lower", "upper")
    
    # Constructor
    
    def __init__(self, maturities, values, kind = "lower"):
        r"""    
        .. rubric:: Initialization
            
        Parameters
        ----------
        maturities : array-like
            Maturities associated with the values of the price index.
        values : array-like
            Values of the price index.
        kind : {"exponential", "linear", "lower", "upper"}, optional
            Type of interpolation when evaluating the price index between two
            consecutive maturities. The default is "lower".
            
        Raises
        ------
        ValueError
            If `kind` is not valid.
            If `maturities` is not along the first axis.
            If `values` has not the same size as `maturities` along the instance axis.  
        """
        
        TermStructure.__init__(self, maturities, values)
        
        if kind not in self.kinds:
            raise ValueError(f"'kind' must be one of the following: {', '.join(self.kinds)}")
            
        self.kind = kind
        
    # Properties
    
    @property
    def kind(self):
        """
        str: Type of interpolation when evaluating the price index between two
        consecutive maturities.
        """
        
        return self._kind
    
    @property
    @memorize
    def piecewise_function(self):
        """
        PiecewiseFunction: Piecewise function used when interpolating the price
        index between two consecutive maturities.
        """
        
        return PiecewiseFunction.interpolate(x = self.maturities, y = self.values, kind = self.kind)

    # Setter
        
    @kind.setter
    def kind(self, value):
        self.delete_attributes("_piecewise_function")
        self._kind = value

[docs]    def normalize(self, maturity = 0, base = 100):
        r"""
        Normalize in-place the price index. The value of the price index at the
        provided reference maturity will be equal to the base value.
        
        Note
        ----
        The new normalized price index at any maturity :math:`\tau` is obtained
        as follows
        
        .. math::
            PI^{normalized}(\tau) = PI(\tau) \frac{base}{PI(\tau^*)}
        
        where :math:`PI(\tau)` is the price index before normalization and
        :math:`base` is the provided base value to reach at the provided
        reference maturity :math:`\tau^*`.

        Parameters
        ----------
        maturity : scalar, optional
            Reference maturity. The default is 0.
        base : scalar, optional
            Base value. The default is 100.

        Returns
        -------
        PriceIndex
            Instance.
        """
        
        self.values = self.values / self.price_index(maturity) * base
        return self

    

[docs]    @Indexation.Decorators.price_index
    def price_index(self, maturities):
        r"""
        Evaluate the price index for the given maturities.
        
        Note
        ----
        The price index at maturity :math:`\tau \in [\tau_i,\tau_{i+1})` is
        given by
        
        .. math::
            PI(\tau) =  PI(\tau_i)
            
        in case of lower interpolation,
        
        .. math::
            PI(\tau) = \alpha(\tau) PI(\tau_i) + (1 - \alpha(\tau)) PI(\tau_{i+1})
            
        in case of linear interpolation where
        :math:`\alpha(\tau) = \frac{\tau_{i+1} - \tau}{\tau_{i+1} - \tau_i}`, or
        
        .. math::
            PI(\tau) = PI(\tau_i) e^{\beta_i (\tau - \tau_i)}
            
        in case of exponential interpolation where 
        :math:`\beta_i = \frac{\ln(PI(\tau_{i+1})) - \ln(PI(\tau_i))}{\tau_{i+1} - \tau_i}`.
            
        Parameters
        ----------
        maturities : array-like
            Maturities for which the price index should be evaluated.

        Returns
        -------
        scalar or array
            Evaluated price index.
            
        Raises
        ------
        ValueError
            If `maturities` cannot be broadcast with the instance.
        """
        
        return self.piecewise_function(maturities)
