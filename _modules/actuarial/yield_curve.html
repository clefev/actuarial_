[docs]    def convert(self, kind = "discount_factor"):
        """
        Convert in-place the discount values to a new type.

        Parameters
        ----------
        kind : {"capitalization_factor", "compound_rate", "continuous_rate", "discount_factor", "simple_rate"}, optional
            The discount values will be converted to this type.
            The default converts to discount factors.

        Returns
        -------
        Discounting
            Instance.
            
        Raises
        ------
        ValueError
            If `kind` is not valid.
        """
        
        # Return self if kind is already correct
        
        if self.kind == kind:
            return self
        
        # Control the kind
                
        if kind not in self.kinds:
            raise ValueError(f"'kind' must be one of the following: {', '.join(self.kinds)}")
                    
        # Convert the discount values
        
        with np.errstate(divide = "ignore", invalid = "ignore"): # ignore numpy warnings such as division by 0 (handled by np.where if any)
            
            if self.kind == "capitalization_factor":
        
                if kind == "compound_rate":
                    self.values = self.values**(1/self.maturities) - 1
                elif kind == "continuous_rate":
                    self.values = np.where(self.values == 1, 0, np.log(self.values)/self.maturities)
                elif kind == "discount_factor":
                    self.values = 1/self.values
                elif kind == "simple_rate":
                    self.values = np.where(self.values == 1, 0, (self.values - 1)/self.maturities)
        
            elif self.kind == "compound_rate":
                
                if kind == "capitalization_factor":
                    self.values = (1 + self.values)**self.maturities
                elif kind == "continuous_rate":
                    self.values = np.log(1 + self.values)
                elif kind == "discount_factor":
                    self.values = (1 + self.values)**(-self.maturities)
                elif kind == "simple_rate":
                    self.values = np.where(self.maturities == 0, self.values, ((1 + self.values)**self.maturities - 1)/self.maturities)
                
            elif self.kind == "continuous_rate":
                
                if kind == "capitalization_factor":
                    self.values = np.exp(self.values*self.maturities)
                elif kind == "compound_rate":
                    self.values = np.exp(self.values) - 1
                elif kind == "discount_factor":
                    self.values = np.exp(-self.values*self.maturities)
                elif kind == "simple_rate":
                    self.values = np.where(self.maturities == 0, self.values, (np.exp(self.values*self.maturities) - 1)/self.maturities)
                
            elif self.kind == "discount_factor":
                
                if kind == "capitalization_factor":
                    self.values = 1/self.values
                elif kind == "compound_rate":
                    self.values = self.values**(-1/self.maturities) - 1
                elif kind == "continuous_rate":
                    self.values = np.where(self.values == 1, 0, -np.log(self.values)/self.maturities)
                elif kind == "simple_rate":
                    self.values = np.where(self.values == 1, 0, (1/self.values - 1)/self.maturities)
                
            elif self.kind == "simple_rate":
                
                if kind == "capitalization_factor":
                    self.values = 1 + self.values*self.maturities
                elif kind == "compound_rate":
                    self.values = np.where(self.maturities == 0, self.values, (1 + self.values*self.maturities)**(1/self.maturities) - 1)
                elif kind == "continuous_rate":
                    self.values = np.where(self.maturities == 0, self.values, np.log(1 + self.values*self.maturities)/self.maturities)
                elif kind == "discount_factor":
                    self.values = 1/(1 + self.values*self.maturities)
                                
        # Update the type of discount values
        
        self.kind = kind
                
        # Return output
        
        return self

    def to_discounting(self, maturities1, maturities2):
        """to_discounting([maturities1=0,] maturities2)
        Return a `Discounting` instance for the provided maturities.

        Parameters
        ----------
        maturities1 : array-like, optional
            Starting maturities. The default is 0.
        maturities2 : array-like or None
            Ending maturities. If None, the `maturities` attribute will be used
            if it exists.

        Returns
        -------
        Discounting
            Discounting instance containing discount values for the provided
            maturities. If `maturities1` is different from 0, the returned discount
            values correspond to discount factors while if `maturities1` is 0,
            they correspond to the `kind` attribute of the instance.

        Raises
        ------
        ValueError
            If `maturities1` and `maturities2` cannot be broadcast with the instance.
            If `maturities2` is None and the instance has no `maturities` attribute.
        """

        if maturities1 is not None and (maturities1 == 0).all(): # maturities1 = 0
            
            if maturities2 is not None:
                return Discounting(0, maturities2, self.evaluate(maturities2), self.kind)
            elif isinstance(self, TermStructure):
                return Discounting(0, self.maturities, self.values, self.kind)
            elif hasattr(self, "maturities"):
                return Discounting(0, self.maturities, self.evaluate(self.maturities), self.kind)
            else:
                raise ValueError("'maturities2' argument cannot be None")
                
        else: # maturities1 > 0
        
            return Discounting(maturities1, maturities2, self.discount_factor(maturities2)/self.discount_factor(maturities1), "discount_factor")

[docs]        def evaluate(func):
            """
            Decorator for the implemented `evaluate` method of a `YieldCurve`
            instance.
                        
            This decorator implements the following:
                
                * Convert the maturities for which the yield curve should be
                  evaluated to a numpy array.
                * Control whether these maturities can be broadcast with the instance.
                * Reduce the output to a scalar when possible.
    
            Returns
            -------
            callable
                Decorator.
    
            Raises
            ------
            ValueError
                If the maturities cannot be broadcast with the instance.
            """
            
            @wraps(func)
            def wrapper(self, maturities, *args, **kwargs):
                    
                # Control if maturities can be broadcast with the instance
                
                maturities = np.asarray(maturities)
                    
                if not self.is_broadcastable(maturities):
                    raise ValueError(f"'maturities' cannot be broadcast with the instance: their respective shapes are {maturities.shape} and {self.shape}")
                        
                # Evaluate the yield curve
                
                values = func(self, maturities, *args, **kwargs)
                
                # Return output
                
                if isinstance(values, np.ndarray):
                    return values[()]
                else:
                    return values
                
            wrapper._undecorated = func            
            return wrapper

[docs]    @YieldCurve.Decorators.evaluate
    def evaluate(self, maturities, forward = False, split = False):
        r"""
        Evaluate the values of the yield curve for the given maturities.
        
        Note
        ----
        The Nelson-Siegel formula for the **spot rate** at maturity :math:`\tau`
        is given by
        
        .. math::
            r(\tau) = \beta_0 + \beta_1 \frac{1 - e^{-\frac{\tau}{\lambda}}}{\frac{\tau}{\lambda}} + \beta_2 \left( \frac{1 - e^{-\frac{\tau}{\lambda}}}{\frac{\tau}{\lambda}} - e^{-\frac{\tau}{\lambda}} \right),

        and the **instantaneaous forward rate** is given by
        
        .. math::
            f(\tau) = \beta_0 + \beta_1 e^{-\frac{\tau}{\lambda}} + \beta_2 \frac{\tau}{\lambda} e^{-\frac{\tau}{\lambda}}.

        Parameters
        ----------
        maturities : array-like
            Maturities for which the yield curve should be evaluated.
        forward : bool, optional
            If True, the instantaneaous forward rate is returned.
            Otherwise, the spot rate is returned.
            The default is False.
        split : bool, optional
            If True, the three additive components of the Nelson-Siegel model
            (level, slope and curvature components) are returned separately
            through a tuple.
            Otherwise, they are aggregated.
            The default is False.

        Returns
        -------
        scalar or array or tuple
            Values of the yield curve for the given maturities.
            If `split` is True, a tuple with the three additive components of
            the Nelson-Siegel model is returned.

        Raises
        ------
        ValueError
            If `maturities` cannot be broadcast with the instance.
        """

        # Get the Nelson-Siegel parameters
        
        beta0, beta1, beta2, scale = self.parameters.values
        
        # Get the predictors multiplying the beta parameters
        
        x0, x1, x2 = self.predictors(maturities, scale, forward = forward)
        
        # Nelson-Siegel formula
        
        if split:
            return (beta0*x0)[()], (beta1*x1)[()], (beta2*x2)[()]
        else:
            return beta0 + beta1*x1 + beta2*x2



[docs]    def predictors(self, maturities, scale = None, forward = False):
        r"""
        Compute the predictors of the Nelson-Siegel model.
        
        Note
        ----
        The Nelson-Siegel model can be seen as a linear model with beta coefficients.
        
        The **spot rate** can be written as
        
        .. math::
            r(\tau) = \beta_0 x_0(\tau; \lambda) + \beta_1 x_1(\tau; \lambda) + \beta_2 x_2(\tau; \lambda)
            
        where the predictors :math:`x_i(\tau,\lambda)` depend on the maturity
        :math:`\tau`, scale parameter :math:`\lambda` and are given by
        
        .. math::
            \begin{cases}
                x_0(\tau; \lambda) = 1 \\
                x_1(\tau; \lambda) = \frac{1 - e^{-\frac{\tau}{\lambda}}}{\frac{\tau}{\lambda}}  \\
                x_2(\tau; \lambda) = \frac{1 - e^{-\frac{\tau}{\lambda}}}{\frac{\tau}{\lambda}} - e^{-\frac{\tau}{\lambda}}
            \end{cases}
            
        The **instantaneous forward rate** can be written as
        
        .. math::
            f(\tau) = \beta_0 z_0(\tau; \lambda) + \beta_1 z_1(\tau; \lambda) + \beta_2 z_2(\tau; \lambda)
            
        where the predictors :math:`z_i(\tau,\lambda)` are given by
        
        .. math::
            \begin{cases}
                z_0(\tau; \lambda) = 1 \\
                z_1(\tau; \lambda) = e^{-\frac{\tau}{\lambda}} \\
                z_2(\tau; \lambda) = \frac{\tau}{\lambda} e^{-\frac{\tau}{\lambda}}
            \end{cases}

        Parameters
        ----------
        maturities : array-like
            Maturities for which the yield curve should be evaluated.
        scale : array-like or None, optional
            Scale parameter of the Nelson-Siegel model. The default uses the
            scale parameter of the instance.
        forward : bool, optional
            If True, the predictors of the instantaneaous forward rate are returned.
            Otherwise, the predictors of the spot rate are returned.
            The default is False.

        Returns
        -------
        tuple of array
            The three predictors of the Nelson-Siegel model. Each predictor is
            an array for which the shape corresponds to the broadcast of the
            maturities with scale parameter.
            
        Raises
        ------
        ValueError
            If `maturities` and `scale` cannot be broadcast together.
        """
        
        # Control if maturities and scale can be broadcast together
        
        maturities = np.asarray(maturities)
        scale = self.parameters["scale"].value if scale is None else np.asarray(scale)
        
        if not util.is_broadcastable(maturities, scale):
            raise ValueError(f"'maturities' and 'scale' parameter cannot be broadcast together: their respective shapes are {maturities.shape} and {scale.shape}")
        
        # Scale maturities and compute its exponential decay function
        
        scaled_maturities = maturities/scale
        exp_scaled_maturities = np.exp(-scaled_maturities)
        
        # Predictors
        
        if forward: # instantaneaous forward rates
            return np.broadcast_to(1, scaled_maturities.shape), exp_scaled_maturities, scaled_maturities*exp_scaled_maturities
        else: # spot rates
            factor = np.ones(scaled_maturities.shape)
            np.divide(1 - exp_scaled_maturities, scaled_maturities, where = (maturities != 0), out = factor) # if a maturity is equal to 0, then factor is equal to 1
            return np.broadcast_to(1, scaled_maturities.shape), factor, factor - exp_scaled_maturities

[docs]    @classmethod
    def fit(cls, yield_curve, scales):
        r"""  
        Fit a Nelson-Siegel model.
        
        Note
        ----
        The fitting method uses a grid search method for the scale parameter and
        an ordinary least-squares (OLS) method for the beta parameters.
        
        Given a scale parameter :math:`\lambda`, the beta parameters of the
        Nelson-Siegel model are fitted by ordinary least-squares method.
        It means that the beta parameters :math:`\beta_0, \beta_1` and :math:`\beta_2`
        are obtained by minimizing the sum of squared error (SSE) given by
        
        .. math::
            SSE = \sum_j \left( \beta_0 + \beta_1 x_1(\tau_j;\lambda) + \beta_2 x_2(\tau_j;\lambda) - r^{obs}(\tau_j) \right)^2     
            
        where :math:`r^{obs}(\tau_j)` are the observed spot continuous rates
        to be fitted.
        
        In order to fit the scale parameter, a grid serach method is used.
        It means that a grid of scale parameters :math:`(\lambda_1, ..., \lambda_K)`
        must be provided to the algorithm. For each scale parameter :math:`\lambda_k`,
        the beta parameters are fitted by OLS method and the sum of squared
        error :math:`SSE_k` is calculated. The optimal scale parameter is the
        one for which the SSE is the smallest.                  

        Parameters
        ----------
        yield_curve : RawYieldCurve
            Raw yield curve to be fitted. The curve will be automatically converted
            to spot continuous rates for the fitting.
        scales : 1D array-like
            Grid search to use for the scale parameter. If only one number is
            provided, the scale parameter will be forced to this one. Otherwise,
            the optimal scale parameter will be chosen from the grid.

        Returns
        -------
        NelsonSiegel
            Fitted instance of a Nelson-Siegel model.

        Raises
        ------
        TypeError
            If `yield_curve` is not a `RawYieldCurve` instance.
        ValueError
            If the length of `yield_curve` is not at least 3.
        """
        
        # Control if yield_curve is a RawYieldCurve instance with at least 3 observed maturities
        
        if not isinstance(yield_curve, RawYieldCurve):
            raise TypeError("'yield_curve' argument must be a RawYieldCurve instance")
        elif len(yield_curve) <= 3:
            raise ValueError("the length of 'yield_curve' must be at least 3")
            
        # Convert raw yield curve to continuous rate
        
        yield_curve = yield_curve.copy().convert("continuous_rate")
        
        # Reshape values of the raw yield curve (must be a matrix with len(yield_curve) rows)
        
        values = np.moveaxis(yield_curve.values, yield_curve.axis, 0).reshape(len(yield_curve), -1) # matrix with len(yield_curve) rows and n columns
        
        # Initialize the parameters and residuals
        
        n = values.shape[1] # number of curves to be fitted
        parameters = cls.get_parametrization().set(np.full(n, np.nan), np.full(n, np.nan), np.full(n, np.nan), np.full(n, np.nan))
        best_residuals = np.full(n, np.inf)
        
        # Calculate the predictors x0, x1 and x2
        
        scales = np.array(scales).ravel() # 1D array
        x0, x1, x2 = cls.predictors(cls, yield_curve.maturities.reshape(-1,1), scales) # matrices with len(yield_curve) rows and len(scales) columns
        
        # Fit the parameters using a grid search method for the scale parameter and a least-squares method (OLS) for the beta parameters

        for i, scale in enumerate(scales): # loop on each scale parameter (grid search)
            
            # Matrix of predictors

            X = np.hstack((x0[:,[i]], x1[:,[i]], x2[:,[i]])) # matrix with len(yield_curve) rows and 3 columns
            
            # Optimal beta parameters using a least-squares method
            
            betas, current_residuals, *_ = np.linalg.lstsq(X, values, rcond = None) # betas is a matrix with 3 rows and n columns and residuals is a vector of length n
        
            # Update optimal residuals (sum of squared errors)
            
            mask = (current_residuals < best_residuals) # boolean vector indicating if current residuals are better than previous best ones
            best_residuals[mask] = current_residuals[mask]
            
            # Update optimal parameters
            
            parameters["beta0"][mask] = betas[0, mask]
            parameters["beta1"][mask] = betas[1, mask]
            parameters["beta2"][mask] = betas[2, mask]
            parameters["scale"][mask] = scale
            
        # Reshape parameters and residuals to be aligned with the initial shape of the provided raw yield curve
            
        shape = list(yield_curve.shape)
        
        if parameters.size == 1:
            shape = ()
        elif yield_curve.ndim == -yield_curve.axis:
            del shape[yield_curve.axis]
        else:
            shape[yield_curve.axis] = 1
            
        parameters.set(*[value.reshape(shape) for value in parameters.values])
        best_residuals = best_residuals.reshape(shape)
            
        # Return output
        
        return cls(**parameters).set_attributes(residuals = best_residuals)

        Note
        ----
        Let :math:`\lbrace(\tau_i,r_i)\rbrace_{i=1,...,n}` be a set of rates to interpolate.

        .. math::
            r(\tau) =   \begin{cases}
                            r_1 & \text{if } \tau < \tau_1, \\
                            \alpha(\tau) r_i + (1-\alpha(\tau)) r_{i+1} & \text{if } \tau_i \leq \tau < \tau_{i+1}, \\
                            r_n & \text{if } \tau \geq \tau_n.
                        \end{cases}

        Returns
        -------
        scalar or array
            Values of the yield curve for the given maturities.

        Raises
        ------
        ValueError
            If `maturities` cannot be broadcast with the instance.
        """

    def evaluate(self, maturities = None):
        r"""
        Evaluate the values of the yield curve for the given maturities.
        
        Note
        ----
        Let :math:`\lbrace(\tau_i,P_i)\rbrace_{i=0,...,n}` be a set of zero-coupon
        bonds price (discount factors) to interpolate. We assume that we have the
        initial condition :math:`\tau_0=0` and :math:`P_0=1`.
        
        The raw interpolated zero-coupon bond price at maturity :math:`\tau \in [\tau_i,\tau_{i+1})`
        is given by
        
        .. math::
            P(\tau) = P_i e^{-\mu_i(\tau-\tau_i)}
                        
        where :math:`\mu_i` is the constant instantaneous forward rate on the
        interval :math:`[\tau_i,\tau_{i+1})` given by
        
        .. math::
            \mu_i = - \frac{\ln(P_{i+1}) - \ln(P_i)}{\tau_{i+1} - \tau_i}.
        
        Note that above the maximum maturity :math:`\tau_n`, an extrapolation
        method is used. The extrapolation assumes that the instantaneous forward
        rate :math:`\mu_{n-1}` calculated for the interval :math:`[\tau_{n-1},\tau_n)`
        continues to apply above :math:`\tau_n`.

        ----------
        maturities : array-like
            Maturities for which the yield curve should be evaluated.

        Returns
        -------
        scalar or array
            Values of the yield curve for the given maturities.

        Raises
        ------
        ValueError
            If `maturities` cannot be broadcast with the instance.
        """
        
        discount_factor = self.piecewise_function(maturities)
        return Discounting(0, maturities, discount_factor, "discount_factor").convert(self.kind).values

    def evaluate(self, maturities, split = False):
        r"""
        Evaluate the values of the yield curve for the given maturities.
        
        Note
        ----
        The value of a mixture yield curve at maturity :math:`\tau` is given by
        
        .. math::
            r(\tau) = \sum_{k=1}^K w_k r_k(\tau)
            
        where :math:`r_k(\tau)` is the value of the k-th yield curve component
        and :math:`w_k` the associated weight.

        Parameters
        ----------
        maturities : array-like
            Maturities for which the yield curve should be evaluated.
        split : bool, optional
            If True, the individual components of the mixture yield curve are
            returned separately. Otherwise, they are aggregated.
            The default is False.

        Returns
        -------
        scalar or array or list
            Values of the yield curve for the given maturities.
            If `split` is True, a list with the weighted values of each individual
            component is returned.

        Raises
        ------
        ValueError
            If `maturities` cannot be broadcast with the instance.
        """
        
        values = [weight*component.to_discounting(maturities).convert(self.kind).values for component, weight in self]
        
        if split:
            return values
        else:
            return sum(values)
