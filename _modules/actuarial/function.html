[docs]        @staticmethod
        def evaluate(func):
            """
            Decorator for the implemented `evaluate` method of a `Function` instance.
            
            This decorator implements the following:

                * Convert the values for which the function should be evaluated
                  to a numpy array.
                * Control whether these values can be broadcast with the instance.
                * Ensure that the returned output is an array and reduce it to
                  a scalar when possible.
    
            Parameters
            ----------
            func : callable
                Method evaluating the function. The first argument of this method
                must be the values for which the function should be evaluated.
                Other additional arguments are allowed.
    
            Returns
            -------
            callable
                Decorator.
                
            Raises
            ------
            ValueError
                If the values for which the function should be evaluated
                cannot be broadcast with the instance.
            """
            
            @wraps(func)
            def wrapper(self, x, *args, **kwargs):
                
                # Control if x can be broadcast with the instance
                
                x = np.asarray(x)
            
                if not self.is_broadcastable(x):
                    raise ValueError(f"'x' cannot be broadcast with the instance: their respective shapes are {x.shape} and {self.shape}")
                    
                # Evaluate the function
                
                value = func(self, x, *args, **kwargs)
                
                # Return output
                
                return np.array(value)[()]        
            
            wrapper._undecorated = func            
            return wrapper

        

[docs]        @staticmethod
        def derivative(func):        
            """
            Decorator for the implemented `derivative` method of a `Function` instance.
            
            This decorator implements the following:

                * Convert the values for which the derivative should be evaluated
                  to a numpy array.
                * Control whether these values can be broadcast with the instance.
                * Control whether the order of the derivative is an integer greater
                  than or equal to 1.
                * Ensure that the returned output is an array and reduce it to
                  a scalar when possible.
    
            Parameters
            ----------
            func : callable
                Method evaluating the derivative. The two first arguments of
                this method must be the values for which the derivative should
                be evaluated and the order of the derivative.
                Other additional arguments are allowed.
    
            Returns
            -------
            callable
                Decorator.
                
            Raises
            ------
            ValueError
                If the values for which the derivative should be evaluated
                cannot be broadcast with the instance.
                If the order of the derivative is not an integer or is not
                greater than or equal to 1.
            """
            
            @wraps(func)
            def wrapper(self, x, order = 1, *args, **kwargs):
                
                # Control if x can be broadcast with the instance
                
                x = np.asarray(x)
                
                if not self.is_broadcastable(x):
                    raise ValueError(f"'x' cannot be broadcast with the instance: their respective shapes are {x.shape} and {self.shape}")
                    
                # Control if order is a non-negative integer
                
                if not isinstance(order, int) or order < 1:
                    raise ValueError("'order' must be a non-negative integer")
                    
                # Evaluate the derivative
                
                derivative = func(self, x, order, *args, **kwargs)
                
                # Return output
                
                return np.array(derivative)[()]
            
            wrapper._undecorated = func            
            return wrapper

        

[docs]        @staticmethod
        def integrate(func):
            """
            Decorator for the implemented `integrate` method of a `Function` instance.
            
            This decorator implements the following:

                * Convert the lower bound and upper bound for which the definite
                  integral should be evaluated to numpy arrays.
                * Control whether these bounds can be broadcast with the instance.
                * Ensure that the returned output is an array and reduce it to
                  a scalar when possible.
    
            Parameters
            ----------
            func : callable
                Method evaluating the definite integral. The two first arguments
                of this method must be the lower bound and upper bound for which
                the definite integral should be evaluated.
                Other additional arguments are allowed.
    
            Returns
            -------
            callable
                Decorator.
                
            Raises
            ------
            ValueError
                If the lower bound and upper bound for which the definite
                integral should be evaluated cannot be broadcast with the
                instance.
            """
            
            @wraps(func)
            def wrapper(self, lower, upper, *args, **kwargs):
                
                # Control if lower and upper can be broadcast with the instance
                
                lower, upper = np.asarray(lower), np.asarray(upper)
                
                if not self.is_broadcastable(lower, upper):
                    raise ValueError(f"'lower' and 'upper' cannot be broadcast with the instance: their respective shapes are {lower.shape}, {upper.shape} and {self.shape}")        
                
                # Evaluate the integral
                
                integral = func(self, lower, upper, *args, **kwargs)
                
                # Return output
                
                return np.array(integral)[()] 
            
            wrapper._undecorated = func            
            return wrapper

        +----------------------+--------------------------------------------------------------------------------+

        |**Properties**        |                                                                                |
        +----------------------+--------------------------------------------------------------------------------+

        +----------------------+--------------------------------------------------------------------------------+
    """
    
    _parametrization = Parameters(constant = Array())
    
    def __init__(self, constant = 0):

    r"""
    .. rubric:: Description
    
    A linear function is a function whose graph is a straight line, that is,
    a polynomial function of degree zero or one.
    
        +----------------------+--------------------------------------------------------------------------------+
        |**Form**              |:math:`f(x) = ax + b`                                                           |
        +----------------------+--------------------------------------------------------------------------------+
        |**Parameters**        || `slope` = :math:`a \in \mathbb{R}`                                            |
        |                      || `intercept` = :math:`b \in \mathbb{R}`                                        |
        +----------------------+--------------------------------------------------------------------------------+
        |**Domain**            |:math:`\mathbb{R}`                                                              |
        +----------------------+--------------------------------------------------------------------------------+
        |**Range**             |:math:`\mathbb{R}`                                                              |
        +----------------------+--------------------------------------------------------------------------------+
        |**Properties**        |                                                                                |
        +----------------------+--------------------------------------------------------------------------------+
        |**Other comments**    |                                                                                |
        +----------------------+--------------------------------------------------------------------------------+
    """
    
    _parametrization = Parameters(slope = Array(), intercept = Array())
    
    def __init__(self, slope = 1, intercept = 0):
        """
        .. rubric:: Initialization

        Parameters
        ----------
        slope : array-like, optional
            Slope parameter. The default is 1.
        intercept : array-like, optional
            Intercept parameter. The value of the function in 0 will be equal
            to this intercept. The default is 0.
            
        Raises
        ------
        ValueError
            If parameters cannot be broadcast together.
        """
        
        super().__init__(self.get_parametrization().set(slope, intercept))

[docs]    @Function.Decorators.derivative
    def derivative(self, x, order = 1):
        r"""
        Evaluate the derivative of the function.
        
        Note
        ----
        The derivative of a linear function is constant and equal to its slope
        while highest order derivatives are zero:
        
        .. math::
            \frac{d^n f(x)}{dx^n} = \begin{cases}
                                        a & \text{if } n = 1, \\
                                        0 & \text{if } n > 1
                                    \end{cases}
                                    
        for all :math:`x \in \mathbb{R}`.

        Parameters
        ----------
        x : array-like
            Values for which the derivative should be evaluated.
        order : int, optional
            Order of the derivative. The default is 1.

        Returns
        -------
        scalar or array
            Evaluated derivative.
            
        Raises
        ------
        ValueError
            If the values for which the derivative should be evaluated
            cannot be broadcast with the instance.
            If the order of the derivative is not an integer or is not
            greater than or equal to 1.
        """
        
        slope, intercept = self.parameters.values
        if order == 1:
            return np.full(self.broadcast(x).shape, slope)
        else:
            return np.full(self.broadcast(x).shape, 0)

    

[docs]    @Function.Decorators.integrate
    def integrate(self, lower, upper):
        r"""
        Evaluate the definite integral of the function.
        
        Note
        ----
        The definite integral of a linear function is given by
        
        .. math::
            \int_{x_1}^{x_2} f(x) dx = \frac{1}{2} a (x_2^2 - x_1^2) + b(x_2 - x_1)
            
        for all :math:`x_1, x_2 \in \mathbb{R}`.
            
        Parameters
        ----------
        lower : array-like
            Lower bound of the definite integral interval.
        upper : array-like
            Upper bound of the definite integral interval.
        
        Returns
        -------
        scalar or array
            Evaluated integral.

        Raises
        ------
        ValueError
            If the lower bound and upper bound for which the definite integral
            should be evaluated cannot be broadcast with the instance.
        """
        
        slope, intercept = self.parameters.values
        return 0.5*slope*(upper**2 - lower**2) + intercept*(upper - lower)

class Logarithm(Parametric, Function):
    r"""
    .. rubric:: Description
    
    Shifted and scaled logarithm function.
    
        +----------------------+--------------------------------------------------------------------------------+
        |**Form**              |:math:`f(x) = a\ln(bx) + c`                                                     |
        +----------------------+--------------------------------------------------------------------------------+
        |**Parameters**        || `scale` = :math:`a \in \mathbb{R}`                                            |
        |                      || `rate` = :math:`b \in \mathbb{R}`                                             |
        |                      || `shift` = :math:`c \in \mathbb{R}`                                            |
        +----------------------+--------------------------------------------------------------------------------+
        |**Domain**            |:math:`\mathbb{R^+_0}`                                                          |
        +----------------------+--------------------------------------------------------------------------------+
        |**Range**             |:math:`\mathbb{R}`                                                              |
        +----------------------+--------------------------------------------------------------------------------+
        |**Properties**        |                                                                                |
        +----------------------+--------------------------------------------------------------------------------+
        |**Other comments**    |                                                                                |
        +----------------------+--------------------------------------------------------------------------------+
    """
    
    _parametrization = Parameters(scale = Array(), rate = Array(min = 0, strict = True), shift = Array())
    
    def __init__(self, scale = 1, rate = 1, shift = 0):
        """
        Initialize a logarithm function f(x) = shift + scale*log(rate*x).

        Parameters
        ----------
        scale : array_like, optional
            Scale parameter multiplying the logarithm. The default is 1.
        rate : array_like, optional
            Rate of the logarithm. The default is 1.
        shift : array_like, optional
            Shift parameter translating the logarithm. The default is 0.

        """
        
        super().__init__(self.get_parametrization().set(scale, rate, shift))
        
    @Function.Decorators.derivative
    def derivative(self, x, order = 1):
        scale, rate, shift = self.parameters.values
        if order == 1:
            return scale/x
        else:
            return (-1)**(order-1) * np.math.factorial(order-1) * scale/x**order
        
    @Function.Decorators.evaluate
    def evaluate(self, x):
        scale, rate, shift = self.parameters.values
        return shift + scale*np.log(rate*x)
    
    @Function.Decorators.integrate
    def integrate(self, lower, upper):
        r"""
        Evaluate the definite integral of the function.
        
        Note
        ----
        The definite integral of a (shifted and scaled) logarithm function is
        given by
        
        .. math::
            \int_{x_1}^{x_2} f(x) dx = a(x_2 \ln(bx_2) - x_1 \ln(bx_1)) + (c - a)(x_2 - x_1)
            
        for all :math:`x_1, x_2 \in \mathbb{R}`.
            
        Parameters
        ----------
        lower : array-like
            Lower bound of the definite integral interval.
        upper : array-like
            Upper bound of the definite integral interval.
        
        Returns
        -------
        scalar or array
            Evaluated integral.

        Raises
        ------
        ValueError
            If the lower bound and upper bound for which the definite integral
            should be evaluated cannot be broadcast with the instance.
        """
        
        scale, rate, shift = self.parameters.values
        return scale*(upper*np.log(rate*upper) - lower*np.log(rate*lower)) + (shift - scale)*(upper - lower) 
        
           
class PiecewiseFunction(Function):
    
    # Class attributes    
    
    _print = ("intervals", "function", "side", "shift", "keepdims")    
    
    # Constructor
    
    def __init__(self, intervals, function, side = "right", shift = False, keepdims = False):
        """
        Initialize a piecewise function.

        Parameters
        ----------
        intervals : array_like
            Intervals dividing the domain of the piecewise function. The domain of the piecewise
            function will be divided into the following intervals: [intervals[0], intervals[1]],
            [intervals[1], intervals[2]], ..., [intervals[N-1], intervals[N]], [intervals[N], inf].
            Therefore if N values are provided, the piecewise function will be divided into
            N intervals. The provided values must be strictly increasing.
        function : Function
            Parametric function to be applied on each interval of the piecewise function.
            Parameters of the function must be given for each interval. The first
            interval of the piecewise function [intervals[0], intervals[1]] will
            use the function parameters from index 0, the second interval [intervals[1], intervals[2]]
            will use parameters from index 1, ... and the last interval [intervals[N], inf]
            will use parameters from index N.
        side : {"left", "right"}, optional
            DESCRIPTION. The default is "right".            
        shift : bool, optional
            DESCRIPTION. The default is False.
        keepdims : bool, optional
            ... The default is False.

        Raises
        ------
        TypeError
            If `function` is not a `Function` instance.
        ValueError
            If side is not correct.
            If intervals are not along the first axis.
            If function has not the same size as intervals along the piecewise axis.

        """
            
        # Control if intervals are along the first axis
        
        intervals = np.asarray(intervals)
        
        if not util.is_along_axis(intervals):
            raise ValueError(f"'intervals' must be along axis 0: its shape is {intervals.shape}")
            
        self._intervals = intervals
            
        # Control if function is a Function and Parametric instance and has the same size as intervals along the axis
        
        if not isinstance(function, Function):
            raise TypeError("'function' must be a 'Function' instance")
        elif not util.is_along_axis(function, self.axis, intervals.size):
            raise ValueError(f"'function' must have the same size as 'intervals' along the axis {self.axis}: its shape is {function.shape}")
            
        self._function = function
            
        # Control the side
        
        if side not in ("left", "right"):
            raise ValueError("side must be 'left' or 'right'")
            
        self._side = side
        self._shift = bool(shift)        
        self._keepdims = bool(keepdims)        
        
    # Properties    
        
    @property
    def axis (self):
        """
        int: Axis along which the intervals are (given as a negative integer).
        """
        
        return -self.intervals.ndim
    
    @property
    @memorize
    def cumulative_integral(self):
        
        if self.shift:
            cumulative_integral = self.get_function(slice(-1)).integrate(0, self.get_intervals(slice(1, None)) - self.get_intervals(slice(-1))).cumsum(axis = self.axis)
        else:
            cumulative_integral = self.get_function(slice(-1)).integrate(self.get_intervals(slice(-1)), self.get_intervals(slice(1, None))).cumsum(axis = self.axis)
        
        return util.concatenate(0, cumulative_integral, axis = self.axis)
    
    @property
    def function(self):
        """
        Function: Parametric function to be applied on each interval
        of the piecewise function (the parameters of the parametric function are
        given for each interval of the piecewise function).
        """
        
        return self._function
    
    @property
    def intervals(self):
        """
        array: Intervals dividing the domain of the piecewise function.
        """
        
        return self._intervals
    
    @property
    def keepdims(self):
        """
        bool: ...
        """
        
        return self._keepdims
    
    @property
    def shift(self):
        
        return self._shift
    
    @property
    def side(self):
        
        return self._side    
    
    # Setter    
        
    @function.setter
    def function(self, value):
        self.delete_attributes("_cumulative_integral")
        self._function = value
        
    @intervals.setter
    def intervals(self, value):
        self.delete_attributes("_cumulative_integral")
        self._intervals = np.asarray(value)
        
    @keepdims.setter
    def keepdims(self, value):
        self._keepdims = bool(value)
        
    @shift.setter
    def shift(self, value):
        self.delete_attributes("_cumulative_integral")
        self._shift = bool(value)
        
    @side.setter
    def side(self, value):
        self._side = value     
        
    # Methods    
    
    def broadcast(self, *args):
        """
        Mimics the broadcast of function parameters with input arrays.
        
        Note
        ----
        For this broadcast, the size of function parameters along the intervals
        (that is, along the 'axis' attribute) is reduced (considered as 1).

        Parameters
        ----------
        *args : array_like
            The arrays to broadcast with the function parameters. If no array is
            provided, only the function parameters are broadcast together.

        Returns
        -------
        np.broadcast
            Broadcast result.

        """
        
        shape = list(self.function.shape)
        shape[self.axis] = 1 # reduce the size along the 'axis' attribute        
        return util.broadcast(np.empty(shape, dtype = []), *args)
    
    def get_function(self, indices = None):
        """
        Get the parametric function.

        Parameters
        ----------
        indices : int or slice or array_like or None, optional
            The parameters of the function will be given for these interval
            indices using the 'take' method. If None, the 'function' attribute is
            returned with all parameters. The default is None.
            
        Returns
        -------
        function : Function
            Parametric function where parameters are associated with the given
            interval indices.

        """
        
        if indices is None:
            return self.function
        else:
            return self.function.copy().set_attributes(parameters = self.get_parameters(indices))
        
    def get_intervals(self, indices = None):
        """
        Get the intervals.

        Parameters
        ----------
        indices : int or slice or 1D array_like or None, optional
            The intervals will be given for these indices. If None, the 'intervals'
            attribute is returned. The default is None.

        Returns
        -------
        intervals : array
            The array has the same number of dimensions as the 'intervals' attribute
            and its size depends on the provided 'indices' input.

        """
        
        if indices is None:
            return self.intervals
        else:
            return util.take(self.intervals, indices, self.axis, keepdims = True)
    
    def get_parameters(self, indices = None):
        """
        Get the function parameters.

        Parameters
        ----------
        indices : int or slice or array_like or None, optional
            The parameters will be given for these interval indices using the 'take'
            method. If None, all parameters are returned. The default is None.

        Returns
        -------
        parameters : Parameters
            Function parameters associated with the given interval indices.

        """
        
        if indices is None:
            return self.function.parameters
        else:
            return self.function.get_parametrization().set(*[self.take(value, indices) for value in self.function.parameters.values])

    def indices(self, x):
        """
        Return interval indices associated with input values.
        
        Note
        ----
        The interval index of a value x is the index i satisfying interval[i] <= x < interval[i+1]
        if the side attribute is 'right' or interval[i] < x <= interval[i+1] if it is 'left'.
        If the value x is below interval[0], its index will be truncated to 0. 

        Parameters
        ----------
        x : array_like
            Values for which interval indices should be found.
            
        Raises
        ------
        ValueError
            If x cannot be broadcast with the function parameters.

        Returns
        -------
        indices : int or array of int
            Integer if x is a scalar or array of integers with the same shape
            as x otherwise.

        """
                
        x = np.asarray(x)
        
        if not self.is_broadcastable(x):
            raise ValueError(f"x cannot be broadcast with the Function instance: their respective shapes are {x.shape} and {self.shape}")
            
        return util.lower_bound(x, self.intervals.ravel(), strict = True if self.side == "left" else False, index = True)  
    
    def take(self, parameter, indices):
        """
        Take the values of the parameter for the given interval indices.

        Parameters
        ----------
        parameter : array
            Parameter for which the values must be extracted.
        indices : int or slice or array_like, optional
            The values of the parameter will be given for these interval indices.

        Raises
        ------
        ValueError
            If indices cannot be broadcast with the function parameters.

        Returns
        -------
        parameter : array
            If the number of dimensions of the parameter argument is smaller than
            the number of dimensions of intervals, then parameter is returned as is.
            If the size of the parameter argument along the intervals (that is, along
            the 'axis' attribute) is equal to one, then parameter is returned as is.
            Otherwise, the values of the parameter will be given along the intervals
            (that is, along the 'axis' attribute).

        """
        
        ndim = self.intervals.ndim

        if isinstance(indices, slice):
            if parameter.ndim < ndim or parameter.shape[self.axis] == 1: # parameter has no dimension along the axis
                return parameter
            else:
                return util.take(parameter, indices, self.axis)
        else:
            indices = np.asarray(indices)

            if not self.is_broadcastable(indices):
                raise ValueError(f"indices cannot be broadcast with the parameters: their respective shapes are {indices.shape} and {self.shape}")
            
            if parameter.ndim < ndim or parameter.shape[self.axis] == 1: # parameter has no dimension along the axis
                pass
            elif (indices.ndim < ndim and indices.size == 1) or (indices.ndim == ndim and util.is_along_axis(indices, self.axis)): # indices is only along the axis
                parameter = util.take(parameter, indices.ravel(), self.axis)
            else:
                shape_parameter, shape_indices = list(parameter.shape), list(indices.shape)
                shape_parameter[self.axis] = 1
                try:
                    shape_indices[self.axis] = 1
                except:
                    pass
                new_shape_parameter, new_shape_indices = np.broadcast_shapes(shape_indices, parameter.shape), np.broadcast_shapes(shape_parameter, indices.shape)
                broadcast_parameter, broadcast_indices = np.broadcast_to(parameter, new_shape_parameter), np.broadcast_to(indices, new_shape_indices)
                parameter = np.take_along_axis(broadcast_parameter, broadcast_indices, self.axis)  

            if self.keepdims or indices.ndim >= ndim: # don't remove the axis
                return parameter
            elif parameter.size == 1: # remove all axis i.e. return a scalar
                return parameter[(0,)*parameter.ndim]
            else: # remove the axis if it exists
                try:
                    return np.squeeze(parameter, self.axis)
                except:
                    return parameter
            
    # Function methods
    
    @Function.Decorators.derivative
    def derivative(self, x, order = 1):
        indices = self.indices(x)
        x = x - self.intervals.ravel()[indices] if self.shift else x
        return self.get_function(indices).derivative(x, order)
    
    @Function.Decorators.evaluate
    def evaluate(self, x):
        indices = self.indices(x)
        x = x - self.intervals.ravel()[indices] if self.shift else x
        return self.get_function(indices).evaluate(x)
    
    @Function.Decorators.integrate
    def integrate(self, lower, upper):
        
        # Get interval indices associated with lower and upper
        
        intervals = self.intervals.ravel()
        lower, upper = np.broadcast_arrays(lower, upper)
        indices_below_lower = util.lower_bound(lower, intervals, index = True) # indices of the interval below lower
        indices_below_upper = util.lower_bound(upper, intervals, index = True) # indices of the interval below upper
        indices_above_lower = util.upper_bound(lower, intervals, index = True) # indices of the interval above lower

        # Compute the definite integral between lower and upper
                
        if self.shift:
            integral = (self.get_function(indices_below_lower).integrate(lower - intervals[indices_below_lower], intervals[indices_above_lower] - intervals[indices_below_lower]) + # integral on [lower, interval above lower]
                        self.take(self.cumulative_integral, indices_below_upper) - self.take(self.cumulative_integral, indices_above_lower) + # integral on [interval above lower, interval below upper]
                        self.get_function(indices_below_upper).integrate(0, upper - intervals[indices_below_upper])) # integral on [interval below upper, upper]
        else:
            integral = (self.get_function(indices_below_lower).integrate(lower, intervals[indices_above_lower]) + # integral on [lower, interval above lower]
                        self.take(self.cumulative_integral, indices_below_upper) - self.take(self.cumulative_integral, indices_above_lower) + # integral on [interval above lower, interval below upper]
                        self.get_function(indices_below_upper).integrate(intervals[indices_below_upper], upper)) # integral on [interval below upper, upper]
        
        # Return output
        
        return integral 
    
    # Class methods    
    
    @classmethod
    def interpolate(cls, x, y, kind = "linear", keepdims = False):
        """
        Perform a piecewise interpolation given observed values for the x-axis
        and the y-axis.

        Parameters
        ----------
        x : array_like
            Observed values for the x-axis. The values must be strictly increasing
            and along the first axis (i.e. only the first axis can have a size greater
            than one, the other ones must have a size equal to one).
        y : array_like
            Observed values for the y-axis. Must have the same size as x along
            the interpolation axis.
        kind : {"exponential", "linear", "logarithm", "lower", "upper"}, optional
            Type of interpolation. The default is "linear".
        keepdims : bool, optional
            ... The default is False.

        Raises
        ------
        ValueError
            If kind is not valid.
            If x is not along the first axis.
            If x has not at least two values.
            If y has not the same size as x along the interpolation axis.

        Returns
        -------
        PiecewiseFunction
            PiecewiseFunction instance such that the value of the function at x
            is equal to y.

        """

        # Control the kind
        
        kinds = ("exponential", "linear", "logarithm", "lower", "upper")
        
        if kind not in kinds:
            raise ValueError(f"kind must be one of the following: {', '.join(kinds)}")
            
        # Convert x and y to numpy array
        
        x, y = np.asarray(x), np.asarray(y)
        
        # Control if x is along the first axis and have at least two values
        
        if not util.is_along_axis(x):
            raise ValueError(f"x must be along axis 0: its shape is {x.shape}")
        elif x.size < 2 and kind not in ("lower", "upper"):
            raise ValueError("x must have at least two values for this kind of interpolation")
            
        # Control if y has the same size as x along the interpolation axis
        
        if not util.is_along_axis(y, -x.ndim, x.size):
            raise ValueError(f"y must have the same size as x along the axis {-x.ndim}: its shape is {y.shape}")
        
        # Calibrate and build the interpolated piecewise function
        
        axis = -x.ndim
          
        if kind == "exponential":
            intervals = x[:-1]
            rate = np.diff(np.log(y), axis = axis) / np.diff(x, axis = axis)
            scale = util.take(y, slice(-1), axis = axis)
            piecewise_function = cls(intervals, Exponential(scale = scale, rate = rate), shift = True, keepdims = keepdims)
        elif kind == "linear":
            intervals = x[:-1]
            intercept = util.take(y, slice(-1), axis = axis)
            slope = np.diff(y, axis = axis) / np.diff(x, axis = axis)
            piecewise_function = cls(intervals, Linear(slope = slope, intercept = intercept), shift = True, keepdims = keepdims)
        elif kind == "logarithm":
            intervals = x[:-1]
            shift = util.take(y, slice(-1), axis = axis)
            scale = np.diff(y, axis = axis) / np.diff(np.log(x), axis = axis)
            rate = 1/intervals
            piecewise_function = cls(intervals, Logarithm(shift = shift, scale = scale, rate = rate), keepdims = keepdims)
        elif kind == "lower":
            intervals = x
            constant = y
            piecewise_function = cls(intervals, Constant(constant = constant), keepdims = keepdims)
        elif kind == "upper":
            intervals = util.concatenate(x[0]-1, x[:-1], axis = axis)
            constant = y
            piecewise_function = cls(intervals, Constant(constant = constant), side = "left", keepdims = keepdims)
        
        # Return the PiecewiseFunction instance
        
        return piecewise_function
        
    # Special methods  

    def __contains__(self, x):
        """
        Check if a value is used as an interval (contained in the 'intervals' attribute).
        """
        
        return x in self.intervals
    
    def __len__(self):
        """
        Number of intervals (length of the 'intervals' attribute).
        """
        
        return len(self.intervals)
