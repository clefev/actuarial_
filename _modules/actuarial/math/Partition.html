[docs]    def __init__(self, values, right_closed = False):
        """
        Parameters
        ----------
        values : array-like
            Values :math:`x_0 < x_1 < ... < x_n` of the partition.
        right_closed : bool, optional
            Indicate whether the subintervals of the partition are right-closed
            :math:`(x_i,x_{i+1}]` or left-closed :math:`[x_i,x_{i+1})`.
            The default assumes left-closed intervals (False).

        Raises
        ------
        ValueError
            If `values` is not along the first axis.
        """
        
        values = np.asarray(values)
        
        if not util.is_along_axis(values):
            raise ValueError(f"'values' must be along axis 0: its shape is {values.shape}")
            
        self._values = values
        self._right_closed = bool(right_closed)

        
    # Properties
    
    @property
    def values(self):
        r"""
        array: Values :math:`x_0 < x_1 < ... < x_n` of the partition.
        """
        
        return self._values
    
    @property
    def min(self):
        r"""
        scalar: Minimum value :math:`x_0` of the partition.
        """
        
        return self.values.ravel()[0]
    
    @property
    def max(self):
        r"""
        scalar: Maximum value :math:`x_n` of the partition.
        """
        
        return self.values.ravel()[-1]
    
    @property
    def length(self):
        r"""
        scalar: Length :math:`x_n - x_0` of the partition.
        """
        
        return self.max - self.min
    
    @property
    def lengths(self):
        r"""
        array: Length of each subinterval of the partition.
        The length of the i-th subinterval is equal to :math:`x_i - x_{i-1}`.
        """
        
        return np.diff(self.values, axis = 0)
    
    @property
    def norm(self):
        r"""
        scalar: Norm of the partition.
        The norm of a partition is the length of the longest subintervals:
        :math:`\max\lbrace |x_i - x_{i-1}| : i=1, ..., n \rbrace`.
        """
        
        return np.max(self.lengths)
    
    @property
    def right_closed(self):
        r"""
        bool: Indicate whether the subintervals of the partition are right-closed
        :math:`(x_i,x_{i+1}]` or left-closed :math:`[x_i,x_{i+1})`.
        """
        
        return self._right_closed
    
    @property
    def axis(self):
        """
        int: Axis along which the values of the partition are (given as a negative integer).
        """
        
        return -self.values.ndim

    def values(self, value):

    @right_closed.setter
    def right_closed(self, value):
        self._right_closed = bool(value)

[docs]    def control(self, values = None):
        """
        Control whether the values of the partition are strictly increasing.

        Parameters
        ----------
        values : array-like or None, optional
            Values of the partition to control.
            If None, the values of the instance will be used instead.
            The default is None.

        Raises
        ------
        ValueError
            If `values` is not along the first axis.
            If `values` is not strictly increasing.            
        """
        
        values = self.values if values is None else np.asarray(values)
        
        if not util.is_along_axis(values):
            raise ValueError(f"'values' must be along axis 0: its shape is {values.shape}")
        elif not (np.diff(values, axis = 0) > 0).all():
            raise ValueError("'values' must be strictly increasing")

    

[docs]    def index(self, x):
        r"""
        Return the partition index associated with input values.
        
        Note
        ----
        The partition index associated with a value :math:`x` is the index :math:`i`
        satisfying either
        
        .. math::
            x_i \leq x < x_{i+1}
        
        in case of left-closed interval or
        
        .. math::
            x_i < x \leq x_{i+1}
            
        in case of right-closed interval.
        
        Note that if the value :math:`x` is below :math:`x_0`, the returned index
        will be truncated to :math:`i=0`. And if it is above :math:`x_n`, the
        returned index will be :math:`i=n`.
        Therefore, the returned index is always an integer between 0 and :math:`n`.

        Parameters
        ----------
        x : array-like
            Values for which interval indices should be provided.

        Returns
        -------
        scalar or array
            Scalar or array with an integer dtype and the same shape as `x`.

        Raises
        ------
        ValueError
            If `x` cannot be broadcast with the instance.
        """
        
        return util.lower_bound(x, self.values.ravel(), strict = self.right_closed, index = True)

    

[docs]    def take(self, x, indices):
        
        # Convert to numpy arrays
        
        x = np.asarray(x)
        indices = np.asarray(indices) if not isinstance(indices, slice) else indices
                
        # Control if x has a size > 1 along the partition axis
        
        if x.ndim < self.ndim or x.shape[self.axis] == 1:
            if not util.is_broadcastable(x, indices):
                raise ValueError(f"'x' and 'indices' cannot be broadcast together: their respective shapes are {x.shape} and {indices.shape}")
            else:
                return x
        
        # Take the values of x along the partition axis
        
        if isinstance(indices, slice):
            return util.take(x, indices, self.axis)
        else:            
            if (indices.ndim < self.ndim and indices.size == 1) or (indices.ndim == self.ndim and util.is_along_axis(indices, self.axis)): # indices is only along the axis
                return util.take(x, indices.ravel(), self.axis)
            else:
                shape_x, shape_indices = list(x.shape), list(indices.shape)
                shape_x[self.axis] = 1
                try:
                    shape_indices[self.axis] = 1
                except:
                    pass
                new_shape_x, new_shape_indices = np.broadcast_shapes(shape_indices, x.shape), np.broadcast_shapes(shape_x, indices.shape)
                x, indices = np.broadcast_to(x, new_shape_x), np.broadcast_to(indices, new_shape_indices)
                return np.take_along_axis(x, indices, self.axis)

        """
        Compare the values of the partition when using comparison operators
        (`==`, `!!=`, `<`, `>`, `<=` and `>=`) on the instance.
        """
        
        return operator(self.values, value)

        """

        The `value` argument must be a scalar.
        """
        
        return value in self.values

    

[docs]    def __getitem__(self, indices):
        """
        Return a (shallow) copy of the instance where we get a subset of values
        in the partition.
        
        The `indices` argument provides the partition indices to extract.
        It must be a slice or 1D array-like of integers or booleans.
        """
        
        return self.copy().set_attributes(values = util.take(self.values, indices, self.axis, keepdims = True))

[docs]    def __setitem__(self, indices, values):
        """
        Replace the values of the partition.
        
        The `indices` argument provides the partition indices to replace.
        It must be an integer, slice or 1D array-like of integers or booleans.
        The `values` argument provides the new points of the partition.
        """
        
        util.put(self.values, indices, values, axis = self.axis, keepdims = True)
        self.values = self.values # signal that the values have changed by using the setter
